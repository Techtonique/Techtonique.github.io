 17    """Arcsinh transformation with offset for zero values"""
 18    return np.arcsinh(x + 1)

 26    Machine learning based reserving model
 27    
 28    Parameters
 29    ----------
 30
 31    model : object, optional
 32        model to use (must implement fit and predict methods), default is RidgeCV
 33
 34    level: a float;
 35        Confidence level for prediction intervals. Default is 95,
 36        equivalent to a miscoverage error of 5 (%)
 37
 38    replications: an integer;
 39        Number of replications for simulated conformal (default is `None`),
 40        for type_pi = "bootstrap" or "kde"
 41
 42    conformal_method: a string
 43        conformal prediction method "splitconformal" or "localconformal"
 44
 45    type_pi: a string;
 46        type of prediction interval: currently `None`
 47        split conformal prediction without simulation, "kde" or "bootstrap"
 48
 49    use_factors : bool, default=False
 50        Whether to treat origin and development years as categorical variables
 51
 52    random_state : int, default=42
 53        Random state for reproducibility
 54    """
 55    
 56    def __init__(self, 
 57                 model=None, 
 58                 level=95,
 59                 replications=None,
 60                 conformal_method="splitconformal",
 61                 type_pi=None,
 62                 use_factors=False,
 63                 random_state=42):
 64        if model is None:
 65            model = RidgeCV(alphas=[10**i for i in range(-5, 5)])
 66        assert conformal_method in ("splitconformal", "localconformal"),\
 67              "must have conformal_method in ('splitconformal', 'localconformal')"    
 68        self.conformal_method = conformal_method
 69        self.model = PredictionInterval(model, level=level, 
 70                                        type_pi=type_pi, 
 71                                        type_split="sequential",
 72                                        method=conformal_method,
 73                                        replications=replications)
 74        self.level = level 
 75        self.replications = replications
 76        self.type_pi = type_pi 
 77        self.use_factors = use_factors
 78        self.origin_col = None
 79        self.development_col = None
 80        self.value_col = None
 81        self.max_dev = None
 82        self.origin_years = None
 83        self.cumulated = None 
 84        self.latest_ = None
 85        self.ultimate_ = None
 86        self.ultimate_lower_ = None
 87        self.ultimate_upper_ = None
 88        self.ibnr_mean_ = None
 89        self.ibnr_lower_ = None
 90        self.ibnr_upper_ = None
 91        self.X_test_ = None 
 92        self.full_data_ = None 
 93        self.full_data_upper_ = None 
 94        self.full_data_lower_ = None 
 95        self.full_data_sims_ = []
 96        self.scaler = StandardScaler()
 97        self.origin_encoder = OneHotEncoder(sparse_output=False, handle_unknown='ignore')
 98        self.dev_encoder = OneHotEncoder(sparse_output=False, handle_unknown='ignore')

100    def fit(self, data, origin_col="origin", 
101            development_col="development", 
102            value_col="values", 
103            cumulated=True):
104        """
105        Fit the model to the triangle data
106        
107        Parameters
108        ----------
109
110        data : pandas.DataFrame
111            Input data with origin, development, and value columns
112
113        origin_col : str, default="origin"
114            Name of the origin year column
115
116        development_col : str, default="development"
117            Name of the development year column
118
119        value_col : str, default="values"
120            Name of the value column
121
122        cumulated: bool, default=True
123            If the triangle is cumulated
124            
125        Returns
126        -------
127
128        self : object
129            Returns self
130        """
131        # Store column names
132        self.origin_col = origin_col
133        self.development_col = development_col
134        self.value_col = value_col
135        self.cumulated = cumulated
136        
137        df = data.copy()
138        df["dev"] = df[development_col] - df[origin_col] + 1
139        df["calendar"] = df[origin_col] + df["dev"] - 1
140        df = df.sort_values([origin_col, "dev"])
141        # If data is cumulated, convert to incremental first
142        if self.cumulated:
143            # Calculate incremental values
144            df[value_col] = df.groupby(origin_col)[value_col].diff().fillna(method='bfill')
145
146        self.max_dev = df["dev"].max()
147        self.origin_years = df[origin_col].unique()
148        
149        # Create full grid of all possible combinations
150        full_grid = pd.MultiIndex.from_product(
151            [self.origin_years, range(1, self.max_dev + 1)],
152            names=[origin_col, "dev"]
153        ).to_frame(index=False)
154        
155        # Merge with original data
156        full_data = pd.merge(
157            full_grid, 
158            df[[origin_col, "dev", value_col]], 
159            on=[origin_col, "dev"], 
160            how="left"
161        )
162        
163        # Calculate calendar year
164        full_data["calendar"] = full_data[origin_col] + full_data["dev"] - 1
165        
166        # Calculate latest values for each origin year
167        self.latest_ = full_data.groupby(origin_col)[value_col].last()
168        
169        # Apply transformations
170        if self.use_factors:
171            # One-hot encode origin and development years
172            origin_encoded = self.origin_encoder.fit_transform(full_data[[origin_col]])
173            dev_encoded = self.dev_encoder.fit_transform(full_data[["dev"]])
174            
175            # Create feature names for the encoded columns
176            origin_feature_names = [f"origin_{year}" for year in self.origin_years]
177            dev_feature_names = [f"dev_{i}" for i in range(1, self.max_dev + 1)]
178            
179            # Add encoded features to the dataframe
180            full_data = pd.concat([
181                full_data,
182                pd.DataFrame(origin_encoded, columns=origin_feature_names, index=full_data.index),
183                pd.DataFrame(dev_encoded, columns=dev_feature_names, index=full_data.index)
184            ], axis=1)
185            
186            # Add calendar year as a feature
187            full_data["log_calendar"] = np.log(full_data["calendar"])
188            feature_cols = origin_feature_names + dev_feature_names + ["log_calendar"]
189        else:
190            # Use log transformations
191            full_data["log_origin"] = np.log(full_data[origin_col])
192            full_data["log_dev"] = np.log(full_data["dev"])
193            full_data["log_calendar"] = np.log(full_data["calendar"])
194            feature_cols = ["log_origin", "log_dev", "log_calendar"]
195        
196        # Transform response if not NaN
197        full_data[f"arcsinh_{value_col}"] = full_data[value_col].apply(
198            lambda x: arcsinh(x) if pd.notnull(x) else x
199        )
200        
201        full_data["to_predict"] = full_data[value_col].isna()
202
203        self.full_data_ = deepcopy(full_data)
204        self.full_data_lower_ = deepcopy(full_data)
205        self.full_data_upper_ = deepcopy(full_data)
206        
207        train_data = full_data[~full_data["to_predict"]]
208        test_data = full_data[full_data["to_predict"]]
209
210        # Prepare features for training
211        X_train = train_data[feature_cols].values
212        X_test = test_data[feature_cols].values
213        
214        # Scale features
215        X_train_scaled = self.scaler.fit_transform(X_train)
216        self.X_test_ = self.scaler.transform(X_test)
217        
218        y_train = train_data[f"arcsinh_{value_col}"].values
219
220        self.model.fit(X_train_scaled, y_train)
221        
222        return self

224    def predict(self):
225        """
226        Make predictions for the missing values in the triangle
227        
228        Returns
229        -------
230
231        DescribeResult
232            Named tuple containing mean, lower, and upper triangles
233        """
234        preds = self.model.predict(self.X_test_, return_pi=True)            
235        
236        to_predict = self.full_data_["to_predict"]        
237
238        # Transform predictions back to original scale
239        mean_pred = inv_arcsinh(preds.mean)
240        lower_pred = inv_arcsinh(preds.lower)
241        upper_pred = inv_arcsinh(preds.upper)
242        
243        # Store predictions in the full data
244        self.full_data_.loc[to_predict, self.value_col] = mean_pred
245        self.full_data_lower_.loc[to_predict, self.value_col] = lower_pred
246        self.full_data_upper_.loc[to_predict, self.value_col] = upper_pred
247
248        # Calculate IBNR based on predicted values (in incremental form)
249        test_data = self.full_data_[to_predict]
250        
251        # Group by origin year and sum predictions
252        self.ibnr_mean_ = test_data.groupby(self.origin_col)[self.value_col].sum()
253        self.ibnr_lower_ = self.full_data_lower_[to_predict].groupby(self.origin_col)[self.value_col].sum()
254        self.ibnr_upper_ = self.full_data_upper_[to_predict].groupby(self.origin_col)[self.value_col].sum()
255
256        # If data was originally cumulated, convert predictions back to cumulative
257        if self.cumulated:
258            for df in [self.full_data_, self.full_data_lower_, self.full_data_upper_]:
259                # Calculate cumulative values
260                df[self.value_col] = df.groupby(self.origin_col)[self.value_col].cumsum()
261
262        # Calculate triangles using utility function
263        mean_triangle = df_to_triangle(
264            self.full_data_,
265            origin_col=self.origin_col,
266            development_col="dev",
267            value_col=self.value_col
268        )
269        lower_triangle = df_to_triangle(
270            self.full_data_lower_,
271            origin_col=self.origin_col,
272            development_col="dev",
273            value_col=self.value_col
274        )
275        upper_triangle = df_to_triangle(
276            self.full_data_upper_,
277            origin_col=self.origin_col,
278            development_col="dev",
279            value_col=self.value_col
280        )
281
282        # Calculate ultimate values
283        if self.cumulated:
284            # For cumulative data, ultimate is the last value in each origin year
285            self.ultimate_ = self.full_data_.groupby(self.origin_col)[self.value_col].last()
286            self.ultimate_lower_ = self.full_data_lower_.groupby(self.origin_col)[self.value_col].last()
287            self.ultimate_upper_ = self.full_data_upper_.groupby(self.origin_col)[self.value_col].last()
288        else:
289            # For incremental data, ultimate is latest + IBNR
290            self.ultimate_ = self.latest_ + self.ibnr_mean_
291            self.ultimate_lower_ = self.latest_ + self.ibnr_lower_
292            self.ultimate_upper_ = self.latest_ + self.ibnr_upper_
293
294        DescribeResult = namedtuple("DescribeResult", 
295                                    ("mean", "lower", "upper"))
296        return DescribeResult(mean_triangle.T, 
297                                lower_triangle.T, 
298                                upper_triangle.T)

313        DescribeResult = namedtuple("DescribeResult",

315           
316        return DescribeResult(self.ibnr_mean_, self.ibnr_lower_, self.ibnr_upper_)

336        Returns

338
339        pandas.DataFrame
340            Ultimate loss estimates (mean, lower, upper) indexed by origin year
341        """
342        if self.ultimate_ is None:
343            raise ValueError("Model must be fitted before getting ultimate values")
344    
345        DescribeResult = namedtuple("DescribeResult",

352    def get_summary(self):
353        """
354        Get a summary of reserving results including latest values, ultimate estimates,
355        and IBNR values with confidence intervals.
356        
357        Returns
358        -------
359        
360        dict
361            Dictionary containing two keys:
362            - 'ByOrigin': DataFrame with results by origin year
363            - 'Totals': Series with total values
364        """
365        if self.ultimate_ is None:
366            raise ValueError("Model must be fitted before getting summary")
367
368        # Get latest values
369        latest = self.get_latest()
370        
371        # Get ultimate values
372        ultimate = self.get_ultimate()
373        
374        # Get IBNR values
375        ibnr = self.get_ibnr()
376        
377        # Create summary by origin
378        summary_by_origin = pd.DataFrame({
379            'Latest': latest,
380            'Mean Ultimate': ultimate.mean,
381            'Mean IBNR': ibnr.mean,
382            f'IBNR {self.level}%': ibnr.upper,
383            f'Ultimate Lo{self.level}': ultimate.lower,
384            f'Ultimate Hi{self.level}': ultimate.upper
385        })
386        
387        # Calculate totals
388        totals = pd.Series({
389            'Latest': latest.sum(),
390            'Mean Ultimate': ultimate.mean.sum(),
391            'Mean IBNR': ibnr.mean.sum(),
392            f'Total IBNR {self.level}%': ibnr.upper.sum()
393        })
394        
395        return {
396            'ByOrigin': summary_by_origin,
397            'Totals': totals
398        }
