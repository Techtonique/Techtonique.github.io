 69    def fit(self, X, y) -> "BoosterClassifier":
 70        """Fit the booster model.
 71        
 72        Parameters:
 73
 74            X: Input data.
 75
 76            y: Target data.
 77            
 78        Returns:
 79
 80            self: The fitted boosting model.
 81        """
 82        if isinstance(X, pd.DataFrame):
 83            X = X.values
 84        if isinstance(y, pd.DataFrame):
 85            y = y.values        
 86        y = np.asarray([int(x) for x in y]).ravel() 
 87        Y = OneHotEncoder().fit_transform(y.reshape(-1, 1)).toarray()
 88        self.classes_ = np.unique(y)
 89        self.n_classes_ = len(self.classes_)
 90        
 91        # Store the results of the list comprehension
 92        self.boosters_ = []
 93        for i in range(self.n_classes_):
 94            booster = _RustBooster(
 95                self.base_estimator,
 96                self.n_estimators,
 97                self.learning_rate,
 98                self.n_hidden_features,
 99                self.direct_link,
100                weights_distribution=self.weights_distribution
101            )
102            booster.fit_boosting(X, Y[:, i], dropout=self.dropout, seed=self.random_state)
103            self.boosters_.append(booster)            
104        return self

122    def predict_proba(self, X) -> np.ndarray:
123        """Make probability predictions with the boosting model.
124        
125        Parameters:
126
127            X: Input data.
128            
129        Returns:
130        
131            preds: Probability predictions.
132        """
133        if isinstance(X, pd.DataFrame):
134            X = X.values
135        raw_preds = np.asarray([booster.predict_boosting(X) for booster in self.boosters_])
136        shifted_preds = raw_preds - np.max(raw_preds, axis=0)
137        exp_preds = np.exp(shifted_preds)
138        return exp_preds / np.sum(exp_preds, axis=0)
