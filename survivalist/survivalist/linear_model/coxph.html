 31class BreslowEstimator:
 32    """Breslow's estimator of the cumulative hazard function.
 33
 34    Attributes
 35    ----------
 36    cum_baseline_hazard_ : :class:`sksurv.functions.StepFunction`
 37        Cumulative baseline hazard function.
 38
 39    baseline_survival_ : :class:`sksurv.functions.StepFunction`
 40        Baseline survival function.
 41
 42    unique_times_ : ndarray
 43        Unique event times.
 44    """
 45
 46    def fit(self, linear_predictor, event, time):
 47        """Compute baseline cumulative hazard function.
 48
 49        Parameters
 50        ----------
 51        linear_predictor : array-like, shape = (n_samples,)
 52            Linear predictor of risk: `X @ coef`.
 53
 54        event : array-like, shape = (n_samples,)
 55            Contains binary event indicators.
 56
 57        time : array-like, shape = (n_samples,)
 58            Contains event/censoring times.
 59
 60        Returns
 61        -------
 62        self
 63        """
 64        risk_score = np.exp(linear_predictor)
 65        order = np.argsort(time, kind="mergesort")
 66        risk_score = risk_score[order]
 67        uniq_times, n_events, n_at_risk, _ = _compute_counts(event, time, order)
 68
 69        divisor = np.empty(n_at_risk.shape, dtype=float)
 70        value = np.sum(risk_score)
 71        divisor[0] = value
 72        k = 0
 73        for i in range(1, len(n_at_risk)):
 74            d = n_at_risk[i - 1] - n_at_risk[i]
 75            value -= risk_score[k: (k + d)].sum()
 76            k += d
 77            divisor[i] = value
 78
 79        assert k == n_at_risk[0] - n_at_risk[-1]
 80
 81        y = np.cumsum(n_events / divisor)
 82        self.cum_baseline_hazard_ = StepFunction(uniq_times, y)
 83        self.baseline_survival_ = StepFunction(uniq_times, np.exp(-y))
 84        self.unique_times_ = uniq_times
 85        return self
 86
 87    def get_cumulative_hazard_function(self, linear_predictor):
 88        """Predict cumulative hazard function.
 89
 90        Parameters
 91        ----------
 92        linear_predictor : array-like, shape = (n_samples,)
 93            Linear predictor of risk: `X @ coef`.
 94
 95        Returns
 96        -------
 97        cum_hazard : ndarray, shape = (n_samples,)
 98            Predicted cumulative hazard functions.
 99        """
100        risk_score = np.exp(linear_predictor)
101        n_samples = risk_score.shape[0]
102        funcs = np.empty(n_samples, dtype=object)
103        for i in range(n_samples):
104            funcs[i] = StepFunction(
105                x=self.cum_baseline_hazard_.x,
106                y=self.cum_baseline_hazard_.y,
107                a=risk_score[i],
108            )
109        return funcs
110
111    def get_survival_function(self, linear_predictor):
112        """Predict survival function.
113
114        Parameters
115        ----------
116        linear_predictor : array-like, shape = (n_samples,)
117            Linear predictor of risk: `X @ coef`.
118
119        Returns
120        -------
121        survival : ndarray, shape = (n_samples,)
122            Predicted survival functions.
123        """
124        risk_score = np.exp(linear_predictor)
125        n_samples = risk_score.shape[0]
126        funcs = np.empty(n_samples, dtype=object)
127        for i in range(n_samples):
128            funcs[i] = StepFunction(
129                x=self.baseline_survival_.x,
130                y=np.power(self.baseline_survival_.y, risk_score[i]),
131            )
132        return funcs

135class CoxPHOptimizer:
136    """Negative partial log-likelihood of Cox proportional hazards model"""
137
138    def __init__(self, X, event, time, alpha, ties):
139        # sort descending
140        o = np.argsort(-time, kind="mergesort")
141        self.x = X[o, :]
142        self.event = event[o]
143        self.time = time[o]
144        self.alpha = alpha
145        self.no_alpha = np.all(self.alpha < np.finfo(self.alpha.dtype).eps)
146        self._is_breslow = ties == "breslow"
147
148    def nlog_likelihood(self, w):
149        """Compute negative partial log-likelihood
150
151        Parameters
152        ----------
153        w : array, shape = (n_features,)
154            Estimate of coefficients
155
156        Returns
157        -------
158        loss : float
159            Average negative partial log-likelihood
160        """
161        time = self.time
162        n_samples = self.x.shape[0]
163        breslow = self._is_breslow
164        xw = np.dot(self.x, w)
165
166        loss = 0
167        risk_set = 0
168        k = 0
169        while k < n_samples:
170            ti = time[k]
171            numerator = 0
172            n_events = 0
173            risk_set2 = 0
174            while k < n_samples and ti == time[k]:
175                if self.event[k]:
176                    numerator += xw[k]
177                    risk_set2 += np.exp(xw[k])
178                    n_events += 1
179                else:
180                    risk_set += np.exp(xw[k])
181                k += 1
182
183            if n_events > 0:
184                if breslow:
185                    risk_set += risk_set2
186                    loss -= (
187                        numerator - n_events * np.log(risk_set)
188                    ) / n_samples
189                else:
190                    numerator /= n_events
191                    for _ in range(n_events):
192                        risk_set += risk_set2 / n_events
193                        loss -= (numerator - np.log(risk_set)) / n_samples
194
195        # add regularization term to log-likelihood
196        return loss + np.sum(self.alpha * np.square(w)) / (2.0 * n_samples)
197
198    def update(self, w, offset=0):
199        """Compute gradient and Hessian matrix with respect to `w`."""
200        time = self.time
201        x = self.x
202        breslow = self._is_breslow
203        exp_xw = np.exp(offset + np.dot(x, w))
204        n_samples, n_features = x.shape
205
206        gradient = np.zeros((1, n_features), dtype=w.dtype)
207        hessian = np.zeros((n_features, n_features), dtype=w.dtype)
208
209        inv_n_samples = 1.0 / n_samples
210        risk_set = 0
211        risk_set_x = np.zeros((1, n_features), dtype=w.dtype)
212        risk_set_xx = np.zeros((n_features, n_features), dtype=w.dtype)
213        k = 0
214        # iterate time in descending order
215        while k < n_samples:
216            ti = time[k]
217            n_events = 0
218            numerator = 0
219            risk_set2 = 0
220            risk_set_x2 = np.zeros_like(risk_set_x)
221            risk_set_xx2 = np.zeros_like(risk_set_xx)
222            while k < n_samples and ti == time[k]:
223                # preserve 2D shape of row vector
224                xk = x[k: k + 1]
225
226                # outer product
227                xx = np.dot(xk.T, xk)
228
229                if self.event[k]:
230                    numerator += xk
231                    risk_set2 += exp_xw[k]
232                    risk_set_x2 += exp_xw[k] * xk
233                    risk_set_xx2 += exp_xw[k] * xx
234                    n_events += 1
235                else:
236                    risk_set += exp_xw[k]
237                    risk_set_x += exp_xw[k] * xk
238                    risk_set_xx += exp_xw[k] * xx
239                k += 1
240
241            if n_events > 0:
242                if breslow:
243                    risk_set += risk_set2
244                    risk_set_x += risk_set_x2
245                    risk_set_xx += risk_set_xx2
246
247                    z = risk_set_x / risk_set
248                    gradient -= (numerator - n_events * z) * inv_n_samples
249
250                    a = risk_set_xx / risk_set
251                    # outer product
252                    b = np.dot(z.T, z)
253
254                    hessian += n_events * (a - b) * inv_n_samples
255                else:
256                    numerator /= n_events
257                    for _ in range(n_events):
258                        risk_set += risk_set2 / n_events
259                        risk_set_x += risk_set_x2 / n_events
260                        risk_set_xx += risk_set_xx2 / n_events
261
262                        z = risk_set_x / risk_set
263                        gradient -= (numerator - z) * inv_n_samples
264
265                        a = risk_set_xx / risk_set
266                        # outer product
267                        b = np.dot(z.T, z)
268
269                        hessian += (a - b) * inv_n_samples
270
271        if not self.no_alpha:
272            gradient += self.alpha * inv_n_samples * w
273
274            diag_idx = np.diag_indices(n_features)
275            hessian[diag_idx] += self.alpha * inv_n_samples
276
277        self.gradient = gradient.ravel()
278        self.hessian = hessian

281class VerboseReporter:
282    def __init__(self, verbose):
283        self.verbose = verbose
284
285    def end_max_iter(self, i):
286        if self.verbose > 0:
287            print(
288                f"iter {i + 1:>6d}: reached maximum number of iterations. Stopping."
289            )
290
291    def end_converged(self, i):
292        if self.verbose > 0:
293            print(f"iter {i + 1:>6d}: optimization converged")
294
295    def update(self, i, delta, loss_new):
296        if self.verbose > 2:
297            print(f"iter {i + 1:>6d}: update = {delta}")
298        if self.verbose > 1:
299            print(f"iter {i + 1:>6d}: loss = {loss_new:.10f}")
300
301    def step_halving(self, i, loss):
302        if self.verbose > 1:
303            print(
304                f"iter {
305                    i:>6d}: loss increased, performing step-halving. loss = {loss:.10f}"
306            )

309class CoxPHSurvivalAnalysis(BaseEstimator, SurvivalAnalysisMixin):
310    """Cox proportional hazards model.
311
312    There are two possible choices for handling tied event times.
313    The default is Breslow's method, which considers each of the
314    events at a given time as distinct. Efron's method is more
315    accurate if there are a large number of ties. When the number
316    of ties is small, the estimated coefficients by Breslow's and
317    Efron's method are quite close. Uses Newton-Raphson optimization.
318
319    See [1]_, [2]_, [3]_ for further description.
320
321    Parameters
322    ----------
323    alpha : float, ndarray of shape (n_features,), optional, default: 0
324        Regularization parameter for ridge regression penalty.
325        If a single float, the same penalty is used for all features.
326        If an array, there must be one penalty for each feature.
327        If you want to include a subset of features without penalization,
328        set the corresponding entries to 0.
329
330    ties : {'breslow', 'efron'}, optional, default: 'breslow'
331        The method to handle tied event times. If there are
332        no tied event times all the methods are equivalent.
333
334    n_iter : int, optional, default: 100
335        Maximum number of iterations.
336
337    tol : float, optional, default: 1e-9
338        Convergence criteria. Convergence is based on the negative log-likelihood::
339
340        |1 - (new neg. log-likelihood / old neg. log-likelihood) | < tol
341
342    verbose : int, optional, default: 0
343        Specifies the amount of additional debug information
344        during optimization.
345
346    Attributes
347    ----------
348    coef_ : ndarray, shape = (n_features,)
349        Coefficients of the model
350
351    cum_baseline_hazard_ : :class:`sksurv.functions.StepFunction`
352        Estimated baseline cumulative hazard function.
353
354    baseline_survival_ : :class:`sksurv.functions.StepFunction`
355        Estimated baseline survival function.
356
357    n_features_in_ : int
358        Number of features seen during ``fit``.
359
360    feature_names_in_ : ndarray of shape (`n_features_in_`,)
361        Names of features seen during ``fit``. Defined only when `X`
362        has feature names that are all strings.
363
364    unique_times_ : array of shape = (n_unique_times,)
365        Unique time points.
366
367    See also
368    --------
369    sksurv.linear_model.CoxnetSurvivalAnalysis
370        Cox proportional hazards model with l1 (LASSO) and l2 (ridge) penalty.
371
372    References
373    ----------
374    .. [1] Cox, D. R. Regression models and life tables (with discussion).
375           Journal of the Royal Statistical Society. Series B, 34, 187-220, 1972.
376    .. [2] Breslow, N. E. Covariance Analysis of Censored Survival Data.
377           Biometrics 30 (1974): 8999.
378    .. [3] Efron, B. The Efficiency of Coxs Likelihood Function for Censored Data.
379           Journal of the American Statistical Association 72 (1977): 557565.
380    """
381
382    _parameter_constraints: dict = {
383        "alpha": [Interval(numbers.Real, 0, None, closed="left"), np.ndarray],
384        "ties": [StrOptions({"breslow", "efron"})],
385        "n_iter": [Interval(numbers.Integral, 1, None, closed="left")],
386        "tol": [Interval(numbers.Real, 0, None, closed="left")],
387        "verbose": ["verbose"],
388    }
389
390    def __init__(
391        self, alpha=0, *, ties="breslow", n_iter=100, tol=1e-9, verbose=0
392    ):
393        self.alpha = alpha
394        self.ties = ties
395        self.n_iter = n_iter
396        self.tol = tol
397        self.verbose = verbose
398
399        self._baseline_model = BreslowEstimator()
400
401    @property
402    def cum_baseline_hazard_(self):
403        return self._baseline_model.cum_baseline_hazard_
404
405    @property
406    def baseline_survival_(self):
407        return self._baseline_model.baseline_survival_
408
409    @property
410    def unique_times_(self):
411        return self._baseline_model.unique_times_
412
413    def fit(self, X, y):
414        """Minimize negative partial log-likelihood for provided data.
415
416        Parameters
417        ----------
418        X : array-like, shape = (n_samples, n_features)
419            Data matrix
420
421        y : structured array, shape = (n_samples,)
422            A structured array containing the binary event indicator
423            as first field, and time of event or time of censoring as
424            second field.
425
426        Returns
427        -------
428        self
429        """
430        self._validate_params()
431
432        X = self._validate_data(X, ensure_min_samples=2, dtype=np.float64)
433        event, time = check_array_survival(X, y)
434
435        if isinstance(self.alpha, (numbers.Real, numbers.Integral)):
436            alphas = np.empty(X.shape[1], dtype=float)
437            alphas[:] = self.alpha
438        else:
439            alphas = self.alpha
440
441        alphas = check_array(
442            alphas,
443            ensure_2d=False,
444            ensure_min_samples=0,
445            estimator=self,
446            input_name="alpha",
447        )
448        if np.any(alphas < 0):
449            raise ValueError(f"alpha must be positive, but was {self.alpha!r}")
450        if alphas.shape[0] != X.shape[1]:
451            raise ValueError(
452                f"Length alphas ({alphas.shape[0]}) must match number of features ({
453                    X.shape[1]})."
454            )
455
456        optimizer = CoxPHOptimizer(X, event, time, alphas, self.ties)
457
458        verbose_reporter = VerboseReporter(self.verbose)
459        w = np.zeros(X.shape[1])
460        w_prev = w
461        i = 0
462        loss = float("inf")
463        while True:
464            if i >= self.n_iter:
465                verbose_reporter.end_max_iter(i)
466                warnings.warn(
467                    (
468                        "Optimization did not converge: Maximum number of iterations has been exceeded."
469                    ),
470                    stacklevel=2,
471                    category=ConvergenceWarning,
472                )
473                break
474
475            optimizer.update(w)
476            delta = solve(
477                optimizer.hessian,
478                optimizer.gradient,
479                overwrite_a=False,
480                overwrite_b=False,
481                check_finite=False,
482            )
483
484            if not np.all(np.isfinite(delta)):
485                raise ValueError(
486                    "search direction contains NaN or infinite values"
487                )
488
489            w_new = w - delta
490            loss_new = optimizer.nlog_likelihood(w_new)
491            verbose_reporter.update(i, delta, loss_new)
492            if loss_new > loss:
493                # perform step-halving if negative log-likelihood does not decrease
494                w = (w_prev + w) / 2
495                loss = optimizer.nlog_likelihood(w)
496                verbose_reporter.step_halving(i, loss)
497                i += 1
498                continue
499
500            w_prev = w
501            w = w_new
502
503            res = np.abs(1 - (loss_new / loss))
504            if res < self.tol:
505                verbose_reporter.end_converged(i)
506                break
507
508            loss = loss_new
509            i += 1
510
511        self.coef_ = w
512        self._baseline_model.fit(np.dot(X, self.coef_), event, time)
513        return self
514
515    def predict(self, X):
516        """Predict risk scores.
517
518        Parameters
519        ----------
520        X : array-like, shape = (n_samples, n_features)
521            Data matrix.
522
523        Returns
524        -------
525        risk_score : array, shape = (n_samples,)
526            Predicted risk scores.
527        """
528        check_is_fitted(self, "coef_")
529
530        X = self._validate_data(X, reset=False)
531
532        return np.dot(X, self.coef_)

534    def predict_cumulative_hazard_function(self, X, return_array=False):
535        """Predict cumulative hazard function.
536
537        The cumulative hazard function for an individual
538        with feature vector :math:`x` is defined as
539
540        .. math::
541
542            H(t \\mid x) = \\exp(x^\\top \\beta) H_0(t) ,
543
544        where :math:`H_0(t)` is the baseline hazard function,
545        estimated by Breslow's estimator.
546
547        Parameters
548        ----------
549        X : array-like, shape = (n_samples, n_features)
550            Data matrix.
551
552        return_array : boolean
553            If set, return an array with the cumulative hazard rate
554            for each `self.unique_times_`, otherwise an array of
555            :class:`sksurv.functions.StepFunction`.
556
557        Returns
558        -------
559        cum_hazard : ndarray
560            If `return_array` is set, an array with the cumulative hazard rate
561            for each `self.unique_times_`, otherwise an array of length `n_samples`
562            of :class:`sksurv.functions.StepFunction` instances will be returned.
563
564        Examples
565        --------
566        >>> import matplotlib.pyplot as plt
567        >>> from sksurv.datasets import load_whas500
568        >>> from sksurv.linear_model import CoxPHSurvivalAnalysis
569
570        Load the data.
571
572        >>> X, y = load_whas500()
573        >>> X = X.astype(float)
574
575        Fit the model.
576
577        >>> estimator = CoxPHSurvivalAnalysis().fit(X, y)
578
579        Estimate the cumulative hazard function for the first 10 samples.
580
581        >>> chf_funcs = estimator.predict_cumulative_hazard_function(X.iloc[:10])
582
583        Plot the estimated cumulative hazard functions.
584
585        >>> for fn in chf_funcs:
586        ...     plt.step(fn.x, fn(fn.x), where="post")
587        ...
588        >>> plt.ylim(0, 1)
589        >>> plt.show()
590        """
591        return self._predict_cumulative_hazard_function(
592            self._baseline_model, self.predict(X), return_array
593        )

596        """Predict survival function.
597
598        The survival function for an individual
599        with feature vector :math:`x` is defined as
600
601        .. math::
602
603            S(t \\mid x) = S_0(t)^{\\exp(x^\\top \\beta)} ,
604
605        where :math:`S_0(t)` is the baseline survival function,
606        estimated by Breslow's estimator.
607
608        Parameters
609        ----------
610        X : array-like, shape = (n_samples, n_features)
611            Data matrix.
612
613        return_array : boolean, default: False
614            If set, return an array with the probability
615            of survival for each `self.unique_times_`,
616            otherwise an array of :class:`sksurv.functions.StepFunction`.
617
618        Returns
619        -------
620        survival : ndarray
621            If `return_array` is set, an array with the probability of
622            survival for each `self.unique_times_`, otherwise an array of
623            length `n_samples` of :class:`sksurv.functions.StepFunction`
624            instances will be returned.
625
626        Examples
627        --------
628        >>> import matplotlib.pyplot as plt
629        >>> from sksurv.datasets import load_whas500
630        >>> from sksurv.linear_model import CoxPHSurvivalAnalysis
631
632        Load the data.
633
634        >>> X, y = load_whas500()
635        >>> X = X.astype(float)
636
637        Fit the model.
638
639        >>> estimator = CoxPHSurvivalAnalysis().fit(X, y)
640
641        Estimate the survival function for the first 10 samples.
642
643        >>> surv_funcs = estimator.predict_survival_function(X.iloc[:10])
644
645        Plot the estimated survival functions.
646
647        >>> for fn in surv_funcs:
648        ...     plt.step(fn.x, fn(fn.x), where="post")
649        ...
650        >>> plt.ylim(0, 1)
651        >>> plt.show()

654            self._baseline_model, self.predict(X), return_array
655        )

415        """Minimize negative partial log-likelihood for provided data.
416
417        Parameters
418        ----------
419        X : array-like, shape = (n_samples, n_features)
420            Data matrix
421
422        y : structured array, shape = (n_samples,)
423            A structured array containing the binary event indicator
424            as first field, and time of event or time of censoring as
425            second field.
426
427        Returns
428        -------
429        self

431        self._validate_params()
432
433        X = self._validate_data(X, ensure_min_samples=2, dtype=np.float64)
434        event, time = check_array_survival(X, y)
435
436        if isinstance(self.alpha, (numbers.Real, numbers.Integral)):
437            alphas = np.empty(X.shape[1], dtype=float)
438            alphas[:] = self.alpha
439        else:
440            alphas = self.alpha
441
442        alphas = check_array(
443            alphas,
444            ensure_2d=False,
445            ensure_min_samples=0,
446            estimator=self,
447            input_name="alpha",
448        )
449        if np.any(alphas < 0):
450            raise ValueError(f"alpha must be positive, but was {self.alpha!r}")
451        if alphas.shape[0] != X.shape[1]:
452            raise ValueError(
453                f"Length alphas ({alphas.shape[0]}) must match number of features ({
454                    X.shape[1]})."
455            )
456
457        optimizer = CoxPHOptimizer(X, event, time, alphas, self.ties)
458
459        verbose_reporter = VerboseReporter(self.verbose)
460        w = np.zeros(X.shape[1])
461        w_prev = w
462        i = 0
463        loss = float("inf")
464        while True:
465            if i >= self.n_iter:
466                verbose_reporter.end_max_iter(i)
467                warnings.warn(
468                    (
469                        "Optimization did not converge: Maximum number of iterations has been exceeded."
470                    ),
471                    stacklevel=2,
472                    category=ConvergenceWarning,
473                )
474                break
475
476            optimizer.update(w)
477            delta = solve(
478                optimizer.hessian,
479                optimizer.gradient,
480                overwrite_a=False,
481                overwrite_b=False,
482                check_finite=False,
483            )
484
485            if not np.all(np.isfinite(delta)):
486                raise ValueError(
487                    "search direction contains NaN or infinite values"
488                )
489
490            w_new = w - delta
491            loss_new = optimizer.nlog_likelihood(w_new)
492            verbose_reporter.update(i, delta, loss_new)
493            if loss_new > loss:
494                # perform step-halving if negative log-likelihood does not decrease
495                w = (w_prev + w) / 2
496                loss = optimizer.nlog_likelihood(w)
497                verbose_reporter.step_halving(i, loss)
498                i += 1
499                continue
500
501            w_prev = w
502            w = w_new
503
504            res = np.abs(1 - (loss_new / loss))
505            if res < self.tol:
506                verbose_reporter.end_converged(i)
507                break
508
509            loss = loss_new
510            i += 1
511
512        self.coef_ = w
513        self._baseline_model.fit(np.dot(X, self.coef_), event, time)
514        return self

Cox proportional hazards model.

There are two possible choices for handling tied event times. The default is Breslow's method, which considers each of the events at a given time as distinct. Efron's method is more accurate if there are a large number of ties. When the number of ties is small, the estimated coefficients by Breslow's and Efron's method are quite close. Uses Newton-Raphson optimization.

See 1, 2, 3 for further description.

Parameters

alpha : float, ndarray of shape (n_features,), optional, default: 0 Regularization parameter for ridge regression penalty. If a single float, the same penalty is used for all features. If an array, there must be one penalty for each feature. If you want to include a subset of features without penalization, set the corresponding entries to 0.

ties : {'breslow', 'efron'}, optional, default: 'breslow' The method to handle tied event times. If there are no tied event times all the methods are equivalent.

n_iter : int, optional, default: 100 Maximum number of iterations.

tol : float, optional, default: 1e-9 Convergence criteria. Convergence is based on the negative log-likelihood::

|1 - (new neg. log-likelihood / old neg. log-likelihood) | < tol

verbose : int, optional, default: 0 Specifies the amount of additional debug information during optimization.

Attributes

feature_names_in_ : ndarray of shape (n_features_in_,) Names of features seen during fit. Defined only when X has feature names that are all strings.

unique_times_ : array of shape = (n_unique_times,) Unique time points.
