132        self.unique_times_ = np.sort(np.unique(self.ss.times))

139            half_n = X_oo.shape[0] // 2
140            X_train_oo, X_calib_oo, y_train_oo, y_calib_oo = train_test_split(
141                X_oo, y_oo, test_size=half_n, random_state=self.random_state, stratify=y_oo
142            )
143            # Fit classifier
144            self.clf.fit(X_train_oo, y_train_oo, **kwargs)
145            # print("y_train_oo:", y_train_oo)

166    def _predict_survival_function_temp(self, X):
167        """
168        Predict survival function with normalized probabilities.
169        """
170        # print("X shape:", X.shape)
171        X_risk, _ = self.ss.transform(X)
172        # print("X_risk shape:", X_risk.shape)
173        oo_test_estimates = self.clf.predict_proba(X_risk)[:, 1]
174        # print("oo_test_estimates shape:", oo_test_estimates.shape)
175
176        if self.type_sim == "none":
177            # If no simulation, return the test estimates
178            return self.ss.predict_survival_function(oo_test_estimates)
179
180        # Apply the calibrated residuals
181        # print("Calibrated residuals shape:", self.calibrated_residuals_.shape)
182        # print("oo_test_estimates shape:", oo_test_estimates.shape)
183        # Simulate the residuals
184        simulations_oo_test_estimates = simulate_replications(
185            data=self.calibrated_residuals_,
186            num_replications=self.replications,
187            n_obs=oo_test_estimates.shape[0],
188            method=self.type_sim,
189        )
190        # print("simulations_oo_test_estimates shape:", simulations_oo_test_estimates.shape)
191        # Add the calibrated residuals to the test estimates
192        oo_test_estimates = np.tile(
193            oo_test_estimates, (self.replications, 1)).T + simulations_oo_test_estimates
194        # clip values to be between 0 and 1
195        oo_test_estimates = np.clip(oo_test_estimates, 0, 1)
196
197        # Normalize probabilities to sum to 1 for each replication
198        for i in range(self.replications):
199            probs = oo_test_estimates[:, i]
200            probs = np.maximum(probs, 0)  # Ensure probabilities are positive
201            probs = probs / np.sum(probs)  # Normalize to sum to 1
202            oo_test_estimates[:, i] = probs
203
204        return [self.ss.predict_survival_function(oo_test_estimates[:, i]) for i in range(self.replications)]

206    def predict(self, X, threshold=0.5):
207
208        surv = self._predict_survival_function_temp(X)
209
210        if self.type_sim == "none":
211            crossings = surv <= threshold  # Boolean array of threshold crossings
212            # For each sample, get the index of the first crossing (or -1 if none)
213            cross_indices = np.argmax(crossings, axis=1)
214            # Handle cases where survival never crosses the threshold:
215            # argmax returns 0 if no True found, so we need to check if the crossing is valid
216            valid_crossings = crossings[np.arange(
217                len(crossings)), cross_indices]
218            # Map to actual times
219            predicted_times = np.where(
220                valid_crossings,
221                self.unique_times_[cross_indices],
222                self.unique_times_[-1],  # use max time if no crossing
223            )
224            return predicted_times
225
226        crossings = [s <= threshold for s in surv]
227        cross_indices = [np.argmax(cross) for cross in crossings]
228        # Handle cases where survival never crosses the threshold:
229        # argmax returns 0 if no True found, so we need to check if the crossing is valid
230        valid_crossings = [cross[ci]
231                           for cross, ci in zip(crossings, cross_indices)]
232        # Map to actual times
233        predicted_times = [
234            # use max time if no crossing
235            np.where(valid, self.unique_times_[ci], self.unique_times_[-1])
236            for valid, ci in zip(valid_crossings, cross_indices)
237        ]
238        return np.array(predicted_times)

263    def predict_survival_function(self, X, return_array=False, level=95):
264        """
265        Predict survival function.
266        """
267        if hasattr(X, "to_numpy"):
268            X = X.to_numpy()
269
270        surv = self._predict_survival_function_temp(X)
271
272        # print("surv: ", surv)
273
274        if self.type_sim == "none":
275            if return_array:
276                return surv
277
278            funcs = []
279            surv = np.asarray(surv)
280            if surv.ndim == 1:
281                surv = surv.reshape(1, -1)
282
283            for i in range(surv.shape[0]):
284                if len(self.unique_times_) != len(surv[i]):
285                    # Crer une grille temporelle approprie
286                    x_old = np.linspace(0, 1, len(surv[i]))
287                    x_new = np.linspace(0, 1, len(self.unique_times_))
288                    surv_interp = np.interp(x_new, x_old, surv[i])
289                else:
290                    surv_interp = surv[i]
291                func = StepFunction(x=self.unique_times_, y=surv_interp)
292                funcs.append(func)
293            return np.array(funcs)
294        else:
295            SurvivalCurves = namedtuple(
296                "SurvivalCurves", ["mean", "lower", "upper"])
297            # `surv` contains `replications` number of survival functions,
298            # for each sample in the test set
299            n_obs = X.shape[0]
300            results = {}
301            for j in range(n_obs):
302                key = "obs" + str(j)
303                results[key] = []
304                for i in range(self.replications):
305                    results[key].append(surv[i][j])
306            # Calculate mean, lower and upper bounds
307            mean_surv = []
308            lower_surv = []
309            upper_surv = []
310            for key in results.keys():
311                mean_surv.append(np.mean(results[key], axis=0))
312                lower_surv.append(np.percentile(
313                    results[key], (100 - level) / 2, axis=0))
314                upper_surv.append(np.percentile(
315                    results[key], 100 - (100 - level) / 2, axis=0))
316            mean_surv = np.array(mean_surv)
317            lower_surv = np.array(lower_surv)
318            upper_surv = np.array(upper_surv)
319            if return_array:
320                return mean_surv, lower_surv, upper_surv
321            # Create StepFunction objects for mean, lower and upper bounds
322            mean_funcs = []
323            lower_funcs = []
324            upper_funcs = []
325            for i in range(mean_surv.shape[0]):
326                mean_func = StepFunction(x=self.unique_times_, y=mean_surv[i])
327                lower_func = StepFunction(
328                    x=self.unique_times_, y=lower_surv[i])
329                upper_func = StepFunction(
330                    x=self.unique_times_, y=upper_surv[i])
331                mean_funcs.append(mean_func)
332                lower_funcs.append(lower_func)
333                upper_funcs.append(upper_func)
334            return SurvivalCurves(mean=mean_funcs, lower=lower_funcs, upper=upper_funcs)
