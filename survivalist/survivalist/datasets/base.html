 31def _get_x_y_survival(dataset, col_event, col_time, val_outcome):
 32    if col_event is None or col_time is None:
 33        y = None
 34        x_frame = dataset
 35    else:
 36        y = np.empty(
 37            dtype=[(col_event, bool), (col_time, np.float64)],
 38            shape=dataset.shape[0],
 39        )
 40        y[col_event] = (dataset[col_event] == val_outcome).values
 41        y[col_time] = dataset[col_time].values
 42
 43        x_frame = dataset.drop([col_event, col_time], axis=1)
 44
 45    return x_frame, y

 59def _create_event_time_df(data):
 60    # Convert the list of tuples into a structured NumPy array
 61    structured_data = np.array(data, dtype=[("event", "?"), ("time", "f8")])
 62
 63    # Convert the structured array to a DataFrame
 64    df = pd.DataFrame(structured_data)
 65
 66    # Change 'event' column to integer64, mapping False to 0 and True to 1
 67    df["event"] = df["event"].astype("int64")
 68
 69    # Ensure 'time' column is float64
 70    df["time"] = df["time"].astype("float64")
 71
 72    return df

 75def _encode_categorical_columns(df, categorical_columns=None):
 76    """
 77    Automatically identifies categorical columns and applies one-hot encoding.
 78
 79    Parameters:
 80    - df (pd.DataFrame): The input DataFrame with mixed continuous and categorical variables.
 81    - categorical_columns (list): Optional list of column names to treat as categorical.
 82
 83    Returns:
 84    - pd.DataFrame: A new DataFrame with one-hot encoded categorical columns.
 85    """
 86    # Automatically identify categorical columns if not provided
 87    if categorical_columns is None:
 88        categorical_columns = df.select_dtypes(
 89            include=["object", "category"]).columns.tolist()
 90
 91    # Apply one-hot encoding to the identified categorical columns
 92    df_encoded = pd.get_dummies(df, columns=categorical_columns)
 93
 94    # Convert boolean columns to integer (0 and 1)
 95    bool_columns = df_encoded.select_dtypes(include=["bool"]).columns.tolist()
 96    df_encoded[bool_columns] = df_encoded[bool_columns].astype(int)
 97
 98    return df_encoded

101def get_x_y(data_frame, attr_labels, pos_label=None, survival=True):
102    """Split data frame into features and labels.
103
104    Parameters
105    ----------
106    data_frame : pandas.DataFrame, shape = (n_samples, n_columns)
107        A data frame.
108
109    attr_labels : sequence of str or None
110        A list of one or more columns that are considered the label.
111        If `survival` is `True`, then attr_labels has two elements:
112        1) the name of the column denoting the event indicator, and
113        2) the name of the column denoting the survival time.
114        If the sequence contains `None`, then labels are not retrieved
115        and only a data frame with features is returned.
116
117    pos_label : any, optional
118        Which value of the event indicator column denotes that a
119        patient experienced an event. This value is ignored if
120        `survival` is `False`.
121
122    survival : bool, optional, default: True
123        Whether to return `y` that can be used for survival analysis.
124
125    Returns
126    -------
127    X : pandas.DataFrame, shape = (n_samples, n_columns - len(attr_labels))
128        Data frame containing features.
129
130    y : None or pandas.DataFrame, shape = (n_samples, len(attr_labels))
131        Data frame containing columns with supervised information.
132        If `survival` was `True`, then the column denoting the event
133        indicator will be boolean and survival times will be float.
134        If `attr_labels` contains `None`, y is set to `None`.
135    """
136    if survival:
137        if len(attr_labels) != 2:
138            raise ValueError(
139                f"expected sequence of length two for attr_labels, but got {len(attr_labels)}")
140        if pos_label is None:
141            raise ValueError(
142                "pos_label needs to be specified if survival=True")
143        return _get_x_y_survival(data_frame, attr_labels[0], attr_labels[1], pos_label)
144
145    return _get_x_y_other(data_frame, attr_labels)

148def _loadarff_with_index(filename):
149    dataset = loadarff(filename)
150    if "index" in dataset.columns:
151        if isinstance(dataset["index"].dtype, CategoricalDtype):
152            # concatenating categorical index may raise TypeError
153            # see https://github.com/pandas-dev/pandas/issues/14586
154            dataset["index"] = dataset["index"].astype(object)
155        dataset.set_index("index", inplace=True)
156    return dataset

159def load_arff_files_standardized(
160    path_training,
161    attr_labels,
162    pos_label=None,
163    path_testing=None,
164    survival=True,
165    standardize_numeric=True,
166    to_numeric=True,
167):
168    """Load dataset in ARFF format.
169
170    Parameters
171    ----------
172    path_training : str
173        Path to ARFF file containing data.
174
175    attr_labels : sequence of str
176        Names of attributes denoting dependent variables.
177        If ``survival`` is set, it must be a sequence with two items:
178        the name of the event indicator and the name of the survival/censoring time.
179
180    pos_label : any type, optional
181        Value corresponding to an event in survival analysis.
182        Only considered if ``survival`` is ``True``.
183
184    path_testing : str, optional
185        Path to ARFF file containing hold-out data. Only columns that are available in both
186        training and testing are considered (excluding dependent variables).
187        If ``standardize_numeric`` is set, data is normalized by considering both training
188        and testing data.
189
190    survival : bool, optional, default: True
191        Whether the dependent variables denote event indicator and survival/censoring time.
192
193    standardize_numeric : bool, optional, default: True
194        Whether to standardize data to zero mean and unit variance.
195        See :func:`survivalist.column.standardize`.
196
197    to_numeric : boo, optional, default: True
198        Whether to convert categorical variables to numeric values.
199        See :func:`survivalist.column.categorical_to_numeric`.
200
201    Returns
202    -------
203    x_train : pandas.DataFrame, shape = (n_train, n_features)
204        Training data.
205
206    y_train : pandas.DataFrame, shape = (n_train, n_labels)
207        Dependent variables of training data.
208
209    x_test : None or pandas.DataFrame, shape = (n_train, n_features)
210        Testing data if `path_testing` was provided.
211
212    y_test : None or pandas.DataFrame, shape = (n_train, n_labels)
213        Dependent variables of testing data if `path_testing` was provided.
214    """
215    dataset = _loadarff_with_index(path_training)
216
217    x_train, y_train = get_x_y(dataset, attr_labels, pos_label, survival)
218
219    if path_testing is not None:
220        x_test, y_test = _load_arff_testing(
221            path_testing, attr_labels, pos_label, survival)
222
223        if len(x_train.columns.symmetric_difference(x_test.columns)) > 0:
224            warnings.warn(
225                "Restricting columns to intersection between training and testing data",
226                stacklevel=2,
227            )
228
229            cols = x_train.columns.intersection(x_test.columns)
230            if len(cols) == 0:
231                raise ValueError(
232                    "columns of training and test data do not intersect")
233
234            x_train = x_train.loc[:, cols]
235            x_test = x_test.loc[:, cols]
236
237        x = safe_concat((x_train, x_test), axis=0)
238        if standardize_numeric:
239            x = standardize(x)
240        if to_numeric:
241            x = categorical_to_numeric(x)
242
243        n_train = x_train.shape[0]
244        x_train = x.iloc[:n_train, :]
245        x_test = x.iloc[n_train:, :]
246    else:
247        if standardize_numeric:
248            x_train = standardize(x_train)
249        if to_numeric:
250            x_train = categorical_to_numeric(x_train)
251
252        x_test = None
253        y_test = None
254
255    return x_train, y_train, x_test, y_test

258def _load_arff_testing(path_testing, attr_labels, pos_label, survival):
259    test_dataset = _loadarff_with_index(path_testing)
260
261    has_labels = pd.Index(attr_labels).isin(test_dataset.columns).all()
262    if not has_labels:
263        if survival:
264            attr_labels = [None, None]
265        else:
266            attr_labels = None
267    return get_x_y(test_dataset, attr_labels, pos_label, survival)

270def load_whas500(*, return_X_y=True, as_frame=False):
271    """Load and return the Worcester Heart Attack Study dataset
272
273    The dataset has 500 samples and 14 features.
274    The endpoint is death, which occurred for 215 patients (43.0%).
275
276    See [1]_, [2]_ for further description.
277
278    Parameters
279    ----------
280    return_X_y : bool, default=False
281        If True, returns ``(data, target)`` instead of a Bunch object. See
282        below for more information about the `data` and `target` object.
283
284    as_frame : bool, default=False
285        If True, the data is a pandas DataFrame including columns with
286        appropriate dtypes (numeric). The target is
287        a pandas DataFrame or Series depending on the number of target columns.
288        If `return_X_y` is True, then (`data`, `target`) will be pandas
289        DataFrames or Series as described below.
290
291    Returns
292    -------
293    x : pandas.DataFrame
294        The measurements for each patient.
295
296    y : structured array with 2 fields
297        *fstat*: boolean indicating whether the endpoint has been reached
298        or the event time is right censored.
299
300        *lenfol*: total length of follow-up (days from hospital admission date
301        to date of last follow-up)
302
303    References
304    ----------
305    .. [1] https://web.archive.org/web/20170114043458/http://www.umass.edu/statdata/statdata/data/
306
307    .. [2] Hosmer, D., Lemeshow, S., May, S.:
308        "Applied Survival Analysis: Regression Modeling of Time to Event Data."
309        John Wiley & Sons, Inc. (2008)
310    """
311    fn = _get_data_path("whas500.arff")
312    X, y = get_x_y(loadarff(fn), attr_labels=[
313                   "fstat", "lenfol"], pos_label="1")
314    if as_frame:
315        return_X_y = False
316    if return_X_y:
317        return X, y
318    y = _create_event_time_df(y)
319    frame = pd.concat([_encode_categorical_columns(X), y], axis=1)
320    target_names = ["event", "time"]
321    fdescr = "The Worcester Heart Attack Study dataset"
322    feature_names = X.columns
323    return Bunch(
324        data=X,
325        target=y,
326        frame=frame,
327        target_names=target_names,
328        DESCR=fdescr,
329        feature_names=feature_names,
330        filename=fn,
331    )

463def load_aids(endpoint="aids", return_X_y=True, as_frame=False):
464    """Load and return the AIDS Clinical Trial dataset
465
466    The dataset has 1,151 samples and 11 features.
467    The dataset has 2 endpoints:
468
469    1. AIDS defining event, which occurred for 96 patients (8.3%)
470    2. Death, which occurred for 26 patients (2.3%)
471
472    See [1]_, [2]_ for further description.
473
474    Parameters
475    ----------
476    endpoint : aids|death
477        The endpoint
478
479    return_X_y : bool, default=False
480        If True, returns ``(data, target)`` instead of a Bunch object. See
481        below for more information about the `data` and `target` object.
482
483    as_frame : bool, default=False
484        If True, the data is a pandas DataFrame including columns with
485        appropriate dtypes (numeric). The target is
486        a pandas DataFrame or Series depending on the number of target columns.
487        If `return_X_y` is True, then (`data`, `target`) will be pandas
488        DataFrames or Series as described below.
489
490    Returns
491    -------
492    x : pandas.DataFrame
493        The measurements for each patient.
494
495    y : structured array with 2 fields
496        *censor*: boolean indicating whether the endpoint has been reached
497        or the event time is right censored.
498
499        *time*: total length of follow-up
500
501        If ``endpoint`` is death, the fields are named *censor_d* and *time_d*.
502
503    References
504    ----------
505    .. [1] https://web.archive.org/web/20170114043458/http://www.umass.edu/statdata/statdata/data/
506
507    .. [2] Hosmer, D., Lemeshow, S., May, S.:
508        "Applied Survival Analysis: Regression Modeling of Time to Event Data."

511    labels_aids = ["censor", "time"]
512    labels_death = ["censor_d", "time_d"]
513    if endpoint == "aids":
514        attr_labels = labels_aids
515        drop_columns = labels_death
516    elif endpoint == "death":
517        attr_labels = labels_death
518        drop_columns = labels_aids

520        raise ValueError("endpoint must be 'aids' or 'death'")

522    fn = _get_data_path("actg320.arff")
523    x, y = get_x_y(loadarff(fn), attr_labels=attr_labels, pos_label="1")
524    x.drop(drop_columns, axis=1, inplace=True)
525    if as_frame:
526        return_X_y = False
527    if return_X_y:
528        return x, y
529    y = _create_event_time_df(y)
530    frame = pd.concat([x, y], axis=1)
531    target_names = ["event", "time"]
532    fdescr = "The AIDS Clinical Trial dataset"
533    feature_names = x.columns
534    return Bunch(
535        data=x,
536        target=y,
537        frame=frame,
538        target_names=target_names,
539        DESCR=fdescr,
540        feature_names=feature_names,
541        filename=fn,
542    )

160def load_arff_files_standardized(
161    path_training,

170

187        training and testing are considered (excluding dependent variables).
188        If ``standardize_numeric`` is set, data is normalized by considering both training

209

211        Testing data if `path_testing` was provided.

214        Dependent variables of testing data if `path_testing` was provided.
215    """

217

219
220    if path_testing is not None:
221        x_test, y_test = _load_arff_testing(
222            path_testing, attr_labels, pos_label, survival)
223
224        if len(x_train.columns.symmetric_difference(x_test.columns)) > 0:
225            warnings.warn(
226                "Restricting columns to intersection between training and testing data",
227                stacklevel=2,
228            )
229
230            cols = x_train.columns.intersection(x_test.columns)
231            if len(cols) == 0:
232                raise ValueError(
233                    "columns of training and test data do not intersect")
234
235            x_train = x_train.loc[:, cols]
236            x_test = x_test.loc[:, cols]
237
238        x = safe_concat((x_train, x_test), axis=0)
239        if standardize_numeric:
240            x = standardize(x)
241        if to_numeric:
242            x = categorical_to_numeric(x)
243
244        n_train = x_train.shape[0]
245        x_train = x.iloc[:n_train, :]
246        x_test = x.iloc[n_train:, :]
247    else:
248        if standardize_numeric:
249            x_train = standardize(x_train)
250        if to_numeric:
251            x_train = categorical_to_numeric(x_train)
252
253        x_test = None
254        y_test = None
255
