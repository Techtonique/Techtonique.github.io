  38def _check_estimate_1d(estimate, test_time):
  39    estimate = check_array(estimate, ensure_2d=False, input_name="estimate")
  40    if estimate.ndim != 1:
  41        raise ValueError(
  42            f"Expected 1D array, got {
  43                estimate.ndim}D array instead:\narray={estimate}.\n"
  44        )
  45    check_consistent_length(test_time, estimate)
  46    return estimate

  49def _check_inputs(event_indicator, event_time, estimate):
  50    check_consistent_length(event_indicator, event_time, estimate)
  51    event_indicator = check_array(
  52        event_indicator, ensure_2d=False, input_name="event_indicator"
  53    )
  54    event_time = check_array(
  55        event_time, ensure_2d=False, input_name="event_time"
  56    )
  57    estimate = _check_estimate_1d(estimate, event_time)
  58
  59    if not np.issubdtype(event_indicator.dtype, np.bool_):
  60        raise ValueError(
  61            f"only boolean arrays are supported as class labels for survival analysis, got {
  62                event_indicator.dtype}"
  63        )
  64
  65    if len(event_time) < 2:
  66        raise ValueError("Need a minimum of two samples")
  67
  68    if not event_indicator.any():
  69        raise ValueError("All samples are censored")
  70
  71    return event_indicator, event_time, estimate

  74def _check_times(test_time, times):
  75    times = check_array(
  76        np.atleast_1d(times), ensure_2d=False, input_name="times"
  77    )
  78    times = np.unique(times)
  79
  80    if times.max() >= test_time.max() or times.min() < test_time.min():
  81        raise ValueError(
  82            f"all times must be within follow-up time of test data: [{test_time.min()}; {
  83                test_time.max()}["
  84        )
  85
  86    return times

  89def _check_estimate_2d(estimate, test_time, time_points, estimator):
  90    estimate = check_array(
  91        estimate,
  92        ensure_2d=False,
  93        allow_nd=False,
  94        input_name="estimate",
  95        estimator=estimator,

 100    if estimate.ndim == 2 and estimate.shape[1] != time_points.shape[0]:
 101        raise ValueError(

 103                time_points.shape[0]} columns, but got {estimate.shape[1]}"
 104        )
 105
 106    return estimate, time_points

 109def _iter_comparable(event_indicator, event_time, order):
 110    n_samples = len(event_time)
 111    tied_time = 0
 112    i = 0
 113    while i < n_samples - 1:
 114        time_i = event_time[order[i]]
 115        end = i + 1
 116        while end < n_samples and event_time[order[end]] == time_i:
 117            end += 1
 118
 119        # check for tied event times
 120        event_at_same_time = event_indicator[order[i:end]]
 121        censored_at_same_time = ~event_at_same_time
 122
 123        mask = np.zeros(n_samples, dtype=bool)
 124        mask[end:] = True
 125        # an event is comparable to censored samples at same time point
 126        mask[i:end] = censored_at_same_time
 127
 128        for j in range(i, end):
 129            if event_indicator[order[j]]:
 130                tied_time += censored_at_same_time.sum()
 131                yield (j, mask, tied_time)
 132        i = end

 135def _estimate_concordance_index(
 136    event_indicator, event_time, estimate, weights, tied_tol=1e-8
 137):
 138    order = np.argsort(event_time)
 139
 140    tied_time = None
 141
 142    concordant = 0
 143    discordant = 0
 144    tied_risk = 0
 145    numerator = 0.0
 146    denominator = 0.0
 147    for ind, mask, tied_time in _iter_comparable(
 148        event_indicator, event_time, order
 149    ):
 150        est_i = estimate[order[ind]]
 151        event_i = event_indicator[order[ind]]
 152        w_i = weights[order[ind]]
 153
 154        est = estimate[order[mask]]
 155
 156        assert (
 157            event_i
 158        ), f"got censored sample at index {order[ind]}, but expected uncensored"
 159
 160        ties = np.absolute(est - est_i) <= tied_tol
 161        n_ties = ties.sum()
 162        # an event should have a higher score
 163        con = est < est_i
 164        n_con = con[~ties].sum()
 165
 166        numerator += w_i * n_con + 0.5 * w_i * n_ties
 167        denominator += w_i * mask.sum()
 168
 169        tied_risk += n_ties
 170        concordant += n_con
 171        discordant += est.size - n_con - n_ties
 172
 173    if tied_time is None:
 174        raise NoComparablePairException(
 175            "Data has no comparable pairs, cannot estimate concordance index."
 176        )
 177
 178    cindex = numerator / denominator
 179    return cindex, concordant, discordant, tied_risk, tied_time

 182def concordance_index_censored(
 183    event_indicator, event_time, estimate, tied_tol=1e-8
 184):
 185    """Concordance index for right-censored data
 186
 187    The concordance index is defined as the proportion of all comparable pairs
 188    in which the predictions and outcomes are concordant.
 189
 190    Two samples are comparable if (i) both of them experienced an event (at different times),
 191    or (ii) the one with a shorter observed survival time experienced an event, in which case
 192    the event-free subject "outlived" the other. A pair is not comparable if they experienced
 193    events at the same time.
 194
 195    Concordance intuitively means that two samples were ordered correctly by the model.
 196    More specifically, two samples are concordant, if the one with a higher estimated
 197    risk score has a shorter actual survival time.
 198    When predicted risks are identical for a pair, 0.5 rather than 1 is added to the count
 199    of concordant pairs.
 200
 201    See the :ref:`User Guide </user_guide/evaluating-survival-models.ipynb>`
 202    and [1]_ for further description.
 203
 204    Parameters
 205    ----------
 206    event_indicator : array-like, shape = (n_samples,)
 207        Boolean array denotes whether an event occurred
 208
 209    event_time : array-like, shape = (n_samples,)
 210        Array containing the time of an event or time of censoring
 211
 212    estimate : array-like, shape = (n_samples,)
 213        Estimated risk of experiencing an event
 214
 215    tied_tol : float, optional, default: 1e-8
 216        The tolerance value for considering ties.
 217        If the absolute difference between risk scores is smaller
 218        or equal than `tied_tol`, risk scores are considered tied.
 219
 220    Returns
 221    -------
 222    cindex : float
 223        Concordance index
 224
 225    concordant : int
 226        Number of concordant pairs
 227
 228    discordant : int
 229        Number of discordant pairs
 230
 231    tied_risk : int
 232        Number of pairs having tied estimated risks
 233
 234    tied_time : int
 235        Number of comparable pairs sharing the same time
 236
 237    See also
 238    --------
 239    concordance_index_ipcw
 240        Alternative estimator of the concordance index with less bias.
 241
 242    References
 243    ----------
 244    .. [1] Harrell, F.E., Califf, R.M., Pryor, D.B., Lee, K.L., Rosati, R.A,
 245           "Multivariable prognostic models: issues in developing models,
 246           evaluating assumptions and adequacy, and measuring and reducing errors",
 247           Statistics in Medicine, 15(4), 361-87, 1996.
 248    """
 249    event_indicator, event_time, estimate = _check_inputs(
 250        event_indicator, event_time, estimate
 251    )
 252
 253    w = np.ones_like(estimate)
 254
 255    return _estimate_concordance_index(
 256        event_indicator, event_time, estimate, w, tied_tol
 257    )

 260def concordance_index_ipcw(
 261    survival_train, survival_test, estimate, tau=None, tied_tol=1e-8
 262):
 263    """Concordance index for right-censored data based on inverse probability of censoring weights.
 264
 265    This is an alternative to the estimator in :func:`concordance_index_censored`
 266    that does not depend on the distribution of censoring times in the test data.
 267    Therefore, the estimate is unbiased and consistent for a population concordance
 268    measure that is free of censoring.
 269
 270    It is based on inverse probability of censoring weights, thus requires
 271    access to survival times from the training data to estimate the censoring
 272    distribution. Note that this requires that survival times `survival_test`
 273    lie within the range of survival times `survival_train`. This can be
 274    achieved by specifying the truncation time `tau`.
 275    The resulting `cindex` tells how well the given prediction model works in
 276    predicting events that occur in the time range from 0 to `tau`.
 277
 278    The estimator uses the Kaplan-Meier estimator to estimate the
 279    censoring survivor function. Therefore, it is restricted to
 280    situations where the random censoring assumption holds and
 281    censoring is independent of the features.
 282
 283    See the :ref:`User Guide </user_guide/evaluating-survival-models.ipynb>`
 284    and [1]_ for further description.
 285
 286    Parameters
 287    ----------
 288    survival_train : structured array, shape = (n_train_samples,)
 289        Survival times for training data to estimate the censoring
 290        distribution from.
 291        A structured array containing the binary event indicator
 292        as first field, and time of event or time of censoring as
 293        second field.
 294
 295    survival_test : structured array, shape = (n_samples,)
 296        Survival times of test data.
 297        A structured array containing the binary event indicator
 298        as first field, and time of event or time of censoring as
 299        second field.
 300
 301    estimate : array-like, shape = (n_samples,)
 302        Estimated risk of experiencing an event of test data.
 303
 304    tau : float, optional
 305        Truncation time. The survival function for the underlying
 306        censoring time distribution :math:`D` needs to be positive
 307        at `tau`, i.e., `tau` should be chosen such that the
 308        probability of being censored after time `tau` is non-zero:
 309        :math:`P(D > \\tau) > 0`. If `None`, no truncation is performed.
 310
 311    tied_tol : float, optional, default: 1e-8
 312        The tolerance value for considering ties.
 313        If the absolute difference between risk scores is smaller
 314        or equal than `tied_tol`, risk scores are considered tied.
 315
 316    Returns
 317    -------
 318    cindex : float
 319        Concordance index
 320
 321    concordant : int
 322        Number of concordant pairs
 323
 324    discordant : int
 325        Number of discordant pairs
 326
 327    tied_risk : int
 328        Number of pairs having tied estimated risks
 329
 330    tied_time : int
 331        Number of comparable pairs sharing the same time
 332
 333    See also
 334    --------
 335    concordance_index_censored
 336        Simpler estimator of the concordance index.
 337
 338    as_concordance_index_ipcw_scorer
 339        Wrapper class that uses :func:`concordance_index_ipcw`
 340        in its ``score`` method instead of the default
 341        :func:`concordance_index_censored`.
 342
 343    References
 344    ----------
 345    .. [1] Uno, H., Cai, T., Pencina, M. J., DAgostino, R. B., & Wei, L. J. (2011).
 346           "On the C-statistics for evaluating overall adequacy of risk prediction
 347           procedures with censored survival data".
 348           Statistics in Medicine, 30(10), 11051117.
 349    """
 350    test_event, test_time = check_y_survival(survival_test)
 351
 352    if tau is not None:
 353        mask = test_time < tau
 354        survival_test = survival_test[mask]
 355
 356    estimate = _check_estimate_1d(estimate, test_time)
 357
 358    cens = CensoringDistributionEstimator()
 359    cens.fit(survival_train)
 360    ipcw_test = cens.predict_ipcw(survival_test)
 361    if tau is None:
 362        ipcw = ipcw_test
 363    else:
 364        ipcw = np.empty(estimate.shape[0], dtype=ipcw_test.dtype)
 365        ipcw[mask] = ipcw_test
 366        ipcw[~mask] = 0
 367
 368    w = np.square(ipcw)
 369
 370    return _estimate_concordance_index(
 371        test_event, test_time, estimate, w, tied_tol
 372    )

 375def cumulative_dynamic_auc(
 376    survival_train, survival_test, estimate, times, tied_tol=1e-8
 377):
 378    """Estimator of cumulative/dynamic AUC for right-censored time-to-event data.
 379
 380    The receiver operating characteristic (ROC) curve and the area under the
 381    ROC curve (AUC) can be extended to survival data by defining
 382    sensitivity (true positive rate) and specificity (true negative rate)
 383    as time-dependent measures. *Cumulative cases* are all individuals that
 384    experienced an event prior to or at time :math:`t` (:math:`t_i \\leq t`),
 385    whereas *dynamic controls* are those with :math:`t_i > t`.
 386    The associated cumulative/dynamic AUC quantifies how well a model can
 387    distinguish subjects who fail by a given time (:math:`t_i \\leq t`) from
 388    subjects who fail after this time (:math:`t_i > t`).
 389
 390    Given an estimator of the :math:`i`-th individual's risk score
 391    :math:`\\hat{f}(\\mathbf{x}_i)`, the cumulative/dynamic AUC at time
 392    :math:`t` is defined as
 393
 394    .. math::
 395
 396        \\widehat{\\mathrm{AUC}}(t) =
 397        \\frac{\\sum_{i=1}^n \\sum_{j=1}^n I(y_j > t) I(y_i \\leq t) \\omega_i
 398        I(\\hat{f}(\\mathbf{x}_j) \\leq \\hat{f}(\\mathbf{x}_i))}
 399        {(\\sum_{i=1}^n I(y_i > t)) (\\sum_{i=1}^n I(y_i \\leq t) \\omega_i)}
 400
 401    where :math:`\\omega_i` are inverse probability of censoring weights (IPCW).
 402
 403    To estimate IPCW, access to survival times from the training data is required
 404    to estimate the censoring distribution. Note that this requires that survival
 405    times `survival_test` lie within the range of survival times `survival_train`.
 406    This can be achieved by specifying `times` accordingly, e.g. by setting
 407    `times[-1]` slightly below the maximum expected follow-up time.
 408    IPCW are computed using the Kaplan-Meier estimator, which is
 409    restricted to situations where the random censoring assumption holds and
 410    censoring is independent of the features.
 411
 412    This function can also be used to evaluate models with time-dependent predictions
 413    :math:`\\hat{f}(\\mathbf{x}_i, t)`, such as :class:`survivalist.ensemble.RandomSurvivalForest`
 414    (see :ref:`User Guide </user_guide/evaluating-survival-models.ipynb#Using-Time-dependent-Risk-Scores>`).
 415    In this case, `estimate` must be a 2-d array where ``estimate[i, j]`` is the
 416    predicted risk score for the i-th instance at time point ``times[j]``.
 417
 418    Finally, the function also provides a single summary measure that refers to the mean
 419    of the :math:`\\mathrm{AUC}(t)` over the time range :math:`(\\tau_1, \\tau_2)`.
 420
 421    .. math::
 422
 423        \\overline{\\mathrm{AUC}}(\\tau_1, \\tau_2) =
 424        \\frac{1}{\\hat{S}(\\tau_1) - \\hat{S}(\\tau_2)}
 425        \\int_{\\tau_1}^{\\tau_2} \\widehat{\\mathrm{AUC}}(t)\\,d \\hat{S}(t)
 426
 427    where :math:`\\hat{S}(t)` is the KaplanMeier estimator of the survival function.
 428
 429    See the :ref:`User Guide </user_guide/evaluating-survival-models.ipynb#Time-dependent-Area-under-the-ROC>`,
 430    [1]_, [2]_, [3]_ for further description.
 431
 432    Parameters
 433    ----------
 434    survival_train : structured array, shape = (n_train_samples,)
 435        Survival times for training data to estimate the censoring
 436        distribution from.
 437        A structured array containing the binary event indicator
 438        as first field, and time of event or time of censoring as
 439        second field.
 440
 441    survival_test : structured array, shape = (n_samples,)
 442        Survival times of test data.
 443        A structured array containing the binary event indicator
 444        as first field, and time of event or time of censoring as
 445        second field.
 446
 447    estimate : array-like, shape = (n_samples,) or (n_samples, n_times)
 448        Estimated risk of experiencing an event of test data.
 449        If `estimate` is a 1-d array, the same risk score across all time
 450        points is used. If `estimate` is a 2-d array, the risk scores in the
 451        j-th column are used to evaluate the j-th time point.
 452
 453    times : array-like, shape = (n_times,)
 454        The time points for which the area under the
 455        time-dependent ROC curve is computed. Values must be
 456        within the range of follow-up times of the test data
 457        `survival_test`.
 458
 459    tied_tol : float, optional, default: 1e-8
 460        The tolerance value for considering ties.
 461        If the absolute difference between risk scores is smaller
 462        or equal than `tied_tol`, risk scores are considered tied.
 463
 464    Returns
 465    -------
 466    auc : array, shape = (n_times,)
 467        The cumulative/dynamic AUC estimates (evaluated at `times`).
 468    mean_auc : float
 469        Summary measure referring to the mean cumulative/dynamic AUC
 470        over the specified time range `(times[0], times[-1])`.
 471
 472    See also
 473    --------
 474    as_cumulative_dynamic_auc_scorer
 475        Wrapper class that uses :func:`cumulative_dynamic_auc`
 476        in its ``score`` method instead of the default
 477        :func:`concordance_index_censored`.
 478
 479    References
 480    ----------
 481    .. [1] H. Uno, T. Cai, L. Tian, and L. J. Wei,
 482           "Evaluating prediction rules for t-year survivors with censored regression models,"
 483           Journal of the American Statistical Association, vol. 102, pp. 527537, 2007.
 484    .. [2] H. Hung and C. T. Chiang,
 485           "Estimation methods for time-dependent AUC models with survival data,"
 486           Canadian Journal of Statistics, vol. 38, no. 1, pp. 826, 2010.
 487    .. [3] J. Lambert and S. Chevret,
 488           "Summary measure of discrimination in survival models based on cumulative/dynamic time-dependent ROC curves,"
 489           Statistical Methods in Medical Research, 2014.
 490    """
 491    test_event, test_time = check_y_survival(survival_test)
 492    estimate, times = _check_estimate_2d(
 493        estimate, test_time, times, estimator="cumulative_dynamic_auc"
 494    )
 495
 496    n_samples = estimate.shape[0]
 497    n_times = times.shape[0]
 498    if estimate.ndim == 1:
 499        estimate = np.broadcast_to(
 500            estimate[:, np.newaxis], (n_samples, n_times)
 501        )
 502
 503    # fit and transform IPCW
 504    cens = CensoringDistributionEstimator()
 505    cens.fit(survival_train)
 506    ipcw = cens.predict_ipcw(survival_test)
 507
 508    # expand arrays to (n_samples, n_times) shape
 509    test_time = np.broadcast_to(test_time[:, np.newaxis], (n_samples, n_times))
 510    test_event = np.broadcast_to(
 511        test_event[:, np.newaxis], (n_samples, n_times)
 512    )
 513    times_2d = np.broadcast_to(times, (n_samples, n_times))
 514    ipcw = np.broadcast_to(ipcw[:, np.newaxis], (n_samples, n_times))
 515
 516    # sort each time point (columns) by risk score (descending)
 517    o = np.argsort(-estimate, axis=0)
 518    test_time = np.take_along_axis(test_time, o, axis=0)
 519    test_event = np.take_along_axis(test_event, o, axis=0)
 520    estimate = np.take_along_axis(estimate, o, axis=0)
 521    ipcw = np.take_along_axis(ipcw, o, axis=0)
 522
 523    is_case = (test_time <= times_2d) & test_event
 524    is_control = test_time > times_2d
 525    n_controls = is_control.sum(axis=0)
 526
 527    # prepend row of infinity values
 528    estimate_diff = np.concatenate(
 529        (np.broadcast_to(np.inf, (1, n_times)), estimate)
 530    )
 531    is_tied = np.absolute(np.diff(estimate_diff, axis=0)) <= tied_tol
 532
 533    cumsum_tp = np.cumsum(is_case * ipcw, axis=0)
 534    cumsum_fp = np.cumsum(is_control, axis=0)
 535    true_pos = cumsum_tp / cumsum_tp[-1]
 536    false_pos = cumsum_fp / n_controls
 537
 538    scores = np.empty(n_times, dtype=float)
 539    it = np.nditer(
 540        (true_pos, false_pos, is_tied), order="F", flags=["external_loop"]
 541    )
 542    with it:
 543        for i, (tp, fp, mask) in enumerate(it):
 544            idx = np.flatnonzero(mask) - 1
 545            # only keep the last estimate for tied risk scores
 546            tp_no_ties = np.delete(tp, idx)
 547            fp_no_ties = np.delete(fp, idx)
 548            # Add an extra threshold position
 549            # to make sure that the curve starts at (0, 0)
 550            tp_no_ties = np.r_[0, tp_no_ties]
 551            fp_no_ties = np.r_[0, fp_no_ties]
 552            scores[i] = np.trapz(tp_no_ties, fp_no_ties)
 553
 554    if n_times == 1:
 555        mean_auc = scores[0]
 556    else:
 557        surv = SurvivalFunctionEstimator()
 558        surv.fit(survival_test)
 559        s_times = surv.predict_proba(times)
 560        # compute integral of AUC over survival function
 561        d = -np.diff(np.r_[1.0, s_times])
 562        integral = (scores * d).sum()
 563        mean_auc = integral / (1.0 - s_times[-1])
 564
 565    return scores, mean_auc

 568def brier_score(survival_train, survival_test, estimate, times):
 569    """Estimate the time-dependent Brier score for right censored data.
 570
 571    The time-dependent Brier score is the mean squared error at time point :math:`t`:
 572
 573    .. math::
 574
 575        \\mathrm{BS}^c(t) = \\frac{1}{n} \\sum_{i=1}^n I(y_i \\leq t \\land \\delta_i = 1)
 576        \\frac{(0 - \\hat{\\pi}(t | \\mathbf{x}_i))^2}{\\hat{G}(y_i)} + I(y_i > t)
 577        \\frac{(1 - \\hat{\\pi}(t | \\mathbf{x}_i))^2}{\\hat{G}(t)} ,
 578
 579    where :math:`\\hat{\\pi}(t | \\mathbf{x})` is the predicted probability of
 580    remaining event-free up to time point :math:`t` for a feature vector :math:`\\mathbf{x}`,
 581    and :math:`1/\\hat{G}(t)` is a inverse probability of censoring weight, estimated by
 582    the Kaplan-Meier estimator.
 583
 584    See the :ref:`User Guide </user_guide/evaluating-survival-models.ipynb#Time-dependent-Brier-Score>`
 585    and [1]_ for details.
 586
 587    Parameters
 588    ----------
 589    survival_train : structured array, shape = (n_train_samples,)
 590        Survival times for training data to estimate the censoring
 591        distribution from.
 592        A structured array containing the binary event indicator
 593        as first field, and time of event or time of censoring as
 594        second field.
 595
 596    survival_test : structured array, shape = (n_samples,)
 597        Survival times of test data.
 598        A structured array containing the binary event indicator
 599        as first field, and time of event or time of censoring as
 600        second field.
 601
 602    estimate : array-like, shape = (n_samples, n_times)
 603        Estimated probability of remaining event-free at time points
 604        specified by `times`. The value of ``estimate[i]`` must correspond to
 605        the estimated probability of remaining event-free up to the time point
 606        ``times[i]``. Typically, estimated probabilities are obtained via the
 607        survival function returned by an estimator's
 608        ``predict_survival_function`` method.
 609
 610    times : array-like, shape = (n_times,)
 611        The time points for which to estimate the Brier score.
 612        Values must be within the range of follow-up times of
 613        the test data `survival_test`.
 614
 615    Returns
 616    -------
 617    times : array, shape = (n_times,)
 618        Unique time points at which the brier scores was estimated.
 619
 620    brier_scores : array , shape = (n_times,)
 621        Values of the brier score.
 622
 623    Examples
 624    --------
 625    >>> from survivalist.datasets import load_gbsg2
 626    >>> from survivalist.linear_model import CoxPHSurvivalAnalysis
 627    >>> from survivalist.metrics import brier_score
 628    >>> from survivalist.preprocessing import OneHotEncoder
 629
 630    Load and prepare data.
 631
 632    >>> X, y = load_gbsg2()
 633    >>> X.loc[:, "tgrade"] = X.loc[:, "tgrade"].map(len).astype(int)
 634    >>> Xt = OneHotEncoder().fit_transform(X)
 635
 636    Fit a Cox model.
 637
 638    >>> est = CoxPHSurvivalAnalysis(ties="efron").fit(Xt, y)
 639
 640    Retrieve individual survival functions and get probability
 641    of remaining event free up to 5 years (=1825 days).
 642
 643    >>> survs = est.predict_survival_function(Xt)
 644    >>> preds = [fn(1825) for fn in survs]
 645
 646    Compute the Brier score at 5 years.
 647
 648    >>> times, score = brier_score(y, y, preds, 1825)
 649    >>> print(score)
 650    [0.20881843]
 651
 652    See also
 653    --------
 654    integrated_brier_score
 655        Computes the average Brier score over all time points.
 656
 657    References
 658    ----------
 659    .. [1] E. Graf, C. Schmoor, W. Sauerbrei, and M. Schumacher,
 660           "Assessment and comparison of prognostic classification schemes for survival data,"
 661           Statistics in Medicine, vol. 18, no. 17-18, pp. 25292545, 1999.
 662    """
 663    test_event, test_time = check_y_survival(survival_test)
 664    estimate, times = _check_estimate_2d(
 665        estimate, test_time, times, estimator="brier_score"
 666    )
 667    if estimate.ndim == 1 and times.shape[0] == 1:
 668        estimate = estimate.reshape(-1, 1)
 669
 670    # fit IPCW estimator
 671    cens = CensoringDistributionEstimator().fit(survival_train)
 672    # calculate inverse probability of censoring weight at current time point t.
 673    prob_cens_t = cens.predict_proba(times)
 674    prob_cens_t[prob_cens_t == 0] = np.inf
 675    # calculate inverse probability of censoring weights at observed time point
 676    prob_cens_y = cens.predict_proba(test_time)
 677    prob_cens_y[prob_cens_y == 0] = np.inf
 678
 679    # Calculating the brier scores at each time point
 680    brier_scores = np.empty(times.shape[0], dtype=float)
 681    for i, t in enumerate(times):
 682        est = estimate[:, i]
 683        is_case = (test_time <= t) & test_event
 684        is_control = test_time > t
 685
 686        brier_scores[i] = np.mean(
 687            np.square(est) * is_case.astype(int) / prob_cens_y
 688            + np.square(1.0 - est) * is_control.astype(int) / prob_cens_t[i]
 689        )
 690
 691    return times, brier_scores

 694def integrated_brier_score(survival_train, survival_test, estimate, times):
 695    """The Integrated Brier Score (IBS) provides an overall calculation of
 696    the model performance at all available times :math:`t_1 \\leq t \\leq t_\\text{max}`.
 697
 698    The integrated time-dependent Brier score over the interval
 699    :math:`[t_1; t_\\text{max}]` is defined as
 700
 701    .. math::
 702
 703        \\mathrm{IBS} = \\int_{t_1}^{t_\\text{max}} \\mathrm{BS}^c(t) d w(t)
 704
 705    where the weighting function is :math:`w(t) = t / t_\\text{max}`.
 706    The integral is estimated via the trapezoidal rule.
 707
 708    See the :ref:`User Guide </user_guide/evaluating-survival-models.ipynb#Time-dependent-Brier-Score>`
 709    and [1]_ for further details.
 710
 711    Parameters
 712    ----------
 713    survival_train : structured array, shape = (n_train_samples,)
 714        Survival times for training data to estimate the censoring
 715        distribution from.
 716        A structured array containing the binary event indicator
 717        as first field, and time of event or time of censoring as
 718        second field.
 719
 720    survival_test : structured array, shape = (n_samples,)
 721        Survival times of test data.
 722        A structured array containing the binary event indicator
 723        as first field, and time of event or time of censoring as
 724        second field.
 725
 726    estimate : array-like, shape = (n_samples, n_times)
 727        Estimated probability of remaining event-free at time points
 728        specified by `times`. The value of ``estimate[i]`` must correspond to
 729        the estimated probability of remaining event-free up to the time point
 730        ``times[i]``. Typically, estimated probabilities are obtained via the
 731        survival function returned by an estimator's
 732        ``predict_survival_function`` method.
 733
 734    times : array-like, shape = (n_times,)
 735        The time points for which to estimate the Brier score.
 736        Values must be within the range of follow-up times of
 737        the test data `survival_test`.
 738
 739    Returns
 740    -------
 741    ibs : float
 742        The integrated Brier score.
 743
 744    Examples
 745    --------
 746    >>> import numpy as np
 747    >>> from survivalist.datasets import load_gbsg2
 748    >>> from survivalist.linear_model import CoxPHSurvivalAnalysis
 749    >>> from survivalist.metrics import integrated_brier_score
 750    >>> from survivalist.preprocessing import OneHotEncoder
 751
 752    Load and prepare data.
 753
 754    >>> X, y = load_gbsg2()
 755    >>> X.loc[:, "tgrade"] = X.loc[:, "tgrade"].map(len).astype(int)
 756    >>> Xt = OneHotEncoder().fit_transform(X)
 757
 758    Fit a Cox model.
 759
 760    >>> est = CoxPHSurvivalAnalysis(ties="efron").fit(Xt, y)
 761
 762    Retrieve individual survival functions and get probability
 763    of remaining event free from 1 year to 5 years (=1825 days).
 764
 765    >>> survs = est.predict_survival_function(Xt)
 766    >>> times = np.arange(365, 1826)
 767    >>> preds = np.asarray([[fn(t) for t in times] for fn in survs])
 768
 769    Compute the integrated Brier score from 1 to 5 years.
 770
 771    >>> score = integrated_brier_score(y, y, preds, times)
 772    >>> print(score)
 773    0.1815853064627424
 774
 775    See also
 776    --------
 777    brier_score
 778        Computes the Brier score at specified time points.
 779
 780    as_integrated_brier_score_scorer
 781        Wrapper class that uses :func:`integrated_brier_score`
 782        in its ``score`` method instead of the default
 783        :func:`concordance_index_censored`.
 784
 785    References
 786    ----------
 787    .. [1] E. Graf, C. Schmoor, W. Sauerbrei, and M. Schumacher,
 788           "Assessment and comparison of prognostic classification schemes for survival data,"
 789           Statistics in Medicine, vol. 18, no. 17-18, pp. 25292545, 1999.
 790    """
 791    # Computing the brier scores
 792    times, brier_scores = brier_score(
 793        survival_train, survival_test, estimate, times
 794    )
 795
 796    if times.shape[0] < 2:
 797        raise ValueError("At least two time points must be given")
 798
 799    # Computing the IBS
 800    ibs_value = np.trapz(brier_scores, times) / (times[-1] - times[0])
 801
 802    return ibs_value

380
381    The receiver operating characteristic (ROC) curve and the area under the
382    ROC curve (AUC) can be extended to survival data by defining
383    sensitivity (true positive rate) and specificity (true negative rate)
384    as time-dependent measures. *Cumulative cases* are all individuals that
385    experienced an event prior to or at time :math:`t` (:math:`t_i \\leq t`),
386    whereas *dynamic controls* are those with :math:`t_i > t`.
387    The associated cumulative/dynamic AUC quantifies how well a model can
388    distinguish subjects who fail by a given time (:math:`t_i \\leq t`) from
389    subjects who fail after this time (:math:`t_i > t`).
390
391    Given an estimator of the :math:`i`-th individual's risk score
392    :math:`\\hat{f}(\\mathbf{x}_i)`, the cumulative/dynamic AUC at time
393    :math:`t` is defined as
394
395    .. math::
396
397        \\widehat{\\mathrm{AUC}}(t) =
398        \\frac{\\sum_{i=1}^n \\sum_{j=1}^n I(y_j > t) I(y_i \\leq t) \\omega_i

509    # expand arrays to (n_samples, n_times) shape
510    test_time = np.broadcast_to(test_time[:, np.newaxis], (n_samples, n_times))
511    test_event = np.broadcast_to(
512        test_event[:, np.newaxis], (n_samples, n_times)
513    )
514    times_2d = np.broadcast_to(times, (n_samples, n_times))
515    ipcw = np.broadcast_to(ipcw[:, np.newaxis], (n_samples, n_times))
516
517    # sort each time point (columns) by risk score (descending)
518    o = np.argsort(-estimate, axis=0)
519    test_time = np.take_along_axis(test_time, o, axis=0)
520    test_event = np.take_along_axis(test_event, o, axis=0)
521    estimate = np.take_along_axis(estimate, o, axis=0)
522    ipcw = np.take_along_axis(ipcw, o, axis=0)
523
524    is_case = (test_time <= times_2d) & test_event
525    is_control = test_time > times_2d
526    n_controls = is_control.sum(axis=0)
527
528    # prepend row of infinity values
529    estimate_diff = np.concatenate(
530        (np.broadcast_to(np.inf, (1, n_times)), estimate)
531    )
532    is_tied = np.absolute(np.diff(estimate_diff, axis=0)) <= tied_tol
533
534    cumsum_tp = np.cumsum(is_case * ipcw, axis=0)
535    cumsum_fp = np.cumsum(is_control, axis=0)
536    true_pos = cumsum_tp / cumsum_tp[-1]
537    false_pos = cumsum_fp / n_controls
538
539    scores = np.empty(n_times, dtype=float)
540    it = np.nditer(
541        (true_pos, false_pos, is_tied), order="F", flags=["external_loop"]
542    )
543    with it:
544        for i, (tp, fp, mask) in enumerate(it):
545            idx = np.flatnonzero(mask) - 1
546            # only keep the last estimate for tied risk scores
547            tp_no_ties = np.delete(tp, idx)
548            fp_no_ties = np.delete(fp, idx)
549            # Add an extra threshold position
550            # to make sure that the curve starts at (0, 0)
551            tp_no_ties = np.r_[0, tp_no_ties]
552            fp_no_ties = np.r_[0, fp_no_ties]
553            scores[i] = np.trapz(tp_no_ties, fp_no_ties)
554
555    if n_times == 1:
556        mean_auc = scores[0]
557    else:
558        surv = SurvivalFunctionEstimator()
559        surv.fit(survival_test)
560        s_times = surv.predict_proba(times)
561        # compute integral of AUC over survival function
562        d = -np.diff(np.r_[1.0, s_times])
563        integral = (scores * d).sum()
564        mean_auc = integral / (1.0 - s_times[-1])
565
566    return scores, mean_auc

Estimator of cumulative/dynamic AUC for right-censored time-to-event data.

The receiver operating characteristic (ROC) curve and the area under the ROC curve (AUC) can be extended to survival data by defining sensitivity (true positive rate) and specificity (true negative rate) as time-dependent measures. Cumulative cases are all individuals that experienced an event prior to or at time \( t \) (\( t_i \leq t \)), whereas *dynamic controls* are those with \( t_i > t \). The associated cumulative/dynamic AUC quantifies how well a model can distinguish subjects who fail by a given time (\( t_i \leq t \)) from subjects who fail after this time (\( t_i > t \)).

Given an estimator of the \( i \)-th individual's risk score \( \hat{f}(\mathbf{x}_i) \), the cumulative/dynamic AUC at time \( t \) is defined as

$$\widehat{\mathrm{AUC}}(t) = \frac{\sum_{i=1}^n \sum_{j=1}^n I(y_j > t) I(y_i \leq t) \omega_i I(\hat{f}(\mathbf{x}_j) \leq \hat{f}(\mathbf{x}_i))} {(\sum_{i=1}^n I(y_i > t)) (\sum_{i=1}^n I(y_i \leq t) \omega_i)}$$

where \( \omega_i \) are inverse probability of censoring weights (IPCW).

To estimate IPCW, access to survival times from the training data is required to estimate the censoring distribution. Note that this requires that survival times survival_test lie within the range of survival times survival_train. This can be achieved by specifying times accordingly, e.g. by setting times[-1] slightly below the maximum expected follow-up time. IPCW are computed using the Kaplan-Meier estimator, which is restricted to situations where the random censoring assumption holds and censoring is independent of the features.

This function can also be used to evaluate models with time-dependent predictions \( \hat{f}(\mathbf{x}_i, t) \), such as survivalist.ensemble.RandomSurvivalForest (see :ref:User Guide </user_guide/evaluating-survival-models.ipynb#Using-Time-dependent-Risk-Scores>). In this case, estimate must be a 2-d array where estimate[i, j] is the predicted risk score for the i-th instance at time point times[j].

Finally, the function also provides a single summary measure that refers to the mean of the \( \mathrm{AUC}(t) \) over the time range \( (\tau_1, \tau_2) \).

$$\overline{\mathrm{AUC}}(\tau_1, \tau_2) = \frac{1}{\hat{S}(\tau_1) - \hat{S}(\tau_2)} \int_{\tau_1}^{\tau_2} \widehat{\mathrm{AUC}}(t)\,d \hat{S}(t)$$

where \( \hat{S}(t) \) is the KaplanMeier estimator of the survival function.

See the :ref:User Guide </user_guide/evaluating-survival-models.ipynb#Time-dependent-Area-under-the-ROC>, 1, 2, 3 for further description.

Parameters

survival_test : structured array, shape = (n_samples,) Survival times of test data. A structured array containing the binary event indicator as first field, and time of event or time of censoring as second field.

estimate : array-like, shape = (n_samples,) or (n_samples, n_times) Estimated risk of experiencing an event of test data. If estimate is a 1-d array, the same risk score across all time points is used. If estimate is a 2-d array, the risk scores in the j-th column are used to evaluate the j-th time point.

times : array-like, shape = (n_times,) The time points for which the area under the time-dependent ROC curve is computed. Values must be within the range of follow-up times of the test data survival_test.

tied_tol : float, optional, default: 1e-8 The tolerance value for considering ties. If the absolute difference between risk scores is smaller or equal than tied_tol, risk scores are considered tied.

Returns
