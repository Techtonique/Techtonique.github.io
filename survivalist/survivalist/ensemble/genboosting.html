 29def _sample_binomial_plus_one(p, size, random_state):
 30    drop_model = random_state.binomial(1, p=p, size=size)
 31    n_dropped = np.sum(drop_model)
 32    if n_dropped == 0:
 33        idx = random_state.randint(0, size)
 34        drop_model[idx] = 1
 35        n_dropped = 1
 36    return drop_model, n_dropped

 39class _ComponentwiseBaseLearner(BaseEstimator):

 55def _fit_stage_componentwise(
 56    X, residuals, sample_weight, regr, **fit_params
 57):  # pylint: disable=unused-argument
 58    """Fit component-wise weighted least squares model"""
 59    n_features = X.shape[1]
 60
 61    base_learners = []
 62    error = np.empty(n_features)
 63    for component in range(n_features):
 64        learner = _ComponentwiseBaseLearner(component, regr=regr).fit(
 65            X, residuals, sample_weight
 66        )
 67        l_pred = learner.predict(X)
 68        error[component] = squared_norm(residuals - l_pred)
 69        base_learners.append(learner)
 70
 71    # TODO: could use bottleneck.nanargmin for speed
 72    return base_learners[np.nanargmin(error)]

 75class ComponentwiseGenGradientBoostingSurvivalAnalysis(
 76    BaseEnsemble, SurvivalAnalysisMixin

 78    """Generic Gradient boosting with any base learner.
 79
 80    Parameters
 81    ----------
 82    regr : object, optional, default: LinearRegression()
 83        The base learner to be used for fitting the residuals. The base learner
 84        must have a `fit` and `predict` method. The default is `LinearRegression`.
 85
 86    loss : {'coxph', 'squared', 'ipcwls'}, optional, default: 'coxph'
 87        loss function to be optimized. 'coxph' refers to partial likelihood loss
 88        of Cox's proportional hazards model. The loss 'squared' minimizes a
 89        squared regression loss that ignores predictions beyond the time of censoring,
 90        and 'ipcwls' refers to inverse-probability of censoring weighted least squares error.
 91
 92    learning_rate : float, optional, default: 0.1
 93        learning rate shrinks the contribution of each base learner by `learning_rate`.
 94        There is a trade-off between `learning_rate` and `n_estimators`.
 95        Values must be in the range `[0.0, inf)`.
 96
 97    n_estimators : int, default: 100
 98        The number of boosting stages to perform. Gradient boosting
 99        is fairly robust to over-fitting so a large number usually
100        results in better performance.
101        Values must be in the range `[1, inf)`.
102
103    subsample : float, optional, default: 1.0
104        The fraction of samples to be used for fitting the individual base
105        learners. If smaller than 1.0 this results in Stochastic Gradient
106        Boosting. `subsample` interacts with the parameter `n_estimators`.
107        Choosing `subsample < 1.0` leads to a reduction of variance
108        and an increase in bias.
109        Values must be in the range `(0.0, 1.0]`.
110
111    warm_start : bool, default: False
112        When set to ``True``, reuse the solution of the previous call to fit
113        and add more estimators to the ensemble, otherwise, just erase the
114        previous solution.
115
116    dropout_rate : float, optional, default: 0.0
117        If larger than zero, the residuals at each iteration are only computed
118        from a random subset of base learners. The value corresponds to the
119        percentage of base learners that are dropped. In each iteration,
120        at least one base learner is dropped. This is an alternative regularization
121        to shrinkage, i.e., setting `learning_rate < 1.0`.
122        Values must be in the range `[0.0, 1.0)`.
123
124    random_state : int seed, RandomState instance, or None, default: None
125        The seed of the pseudo random number generator to use when
126        shuffling the data.
127
128    verbose : int, default: 0
129        Enable verbose output. If 1 then it prints progress and performance
130        once in a while.
131        Values must be in the range `[0, inf)`.
132
133    show_progress : bool, default: True
134        If set, show a progress bar for the fitting process.
135
136    Attributes
137    ----------
138    estimators_ : list of base learners
139        The collection of fitted sub-estimators.
140
141    train_score_ : ndarray, shape = (n_estimators,)
142        The i-th score ``train_score_[i]`` is the loss of the
143        model at iteration ``i`` on the in-bag sample.
144        If ``subsample == 1`` this is the loss on the training data.
145
146    oob_improvement_ : ndarray, shape = (n_estimators,)
147        The improvement in loss on the out-of-bag samples
148        relative to the previous iteration.
149        ``oob_improvement_[0]`` is the improvement in
150        loss of the first stage over the ``init`` estimator.
151        Only available if ``subsample < 1.0``.
152
153    oob_scores_ : ndarray of shape (n_estimators,)
154        The full history of the loss values on the out-of-bag
155        samples. Only available if ``subsample < 1.0``.
156
157    oob_score_ : float
158        The last value of the loss on the out-of-bag samples. It is
159        the same as ``oob_scores_[-1]``. Only available if ``subsample < 1.0``.
160
161    n_features_in_ : int
162        Number of features seen during ``fit``.
163
164    feature_names_in_ : ndarray of shape (`n_features_in_`,)
165        Names of features seen during ``fit``. Defined only when `X`
166        has feature names that are all strings.
167
168    unique_times_ : array of shape = (n_unique_times,)
169        Unique time points.
170
171    References
172    ----------
173    .. [1] Hothorn, T., Bühlmann, P., Dudoit, S., Molinaro, A., van der Laan, M. J.,
174           "Survival ensembles", Biostatistics, 7(3), 355-73, 2006
175    """

221    def _init_state(self):
222        self.estimators_ = np.empty(self.n_estimators, dtype=object)
223
224        self.train_score_ = np.zeros(self.n_estimators, dtype=np.float64)

236    def _resize_state(self):
237        """Add additional ``n_estimators`` entries to all attributes."""
238        # self.n_estimators is the number of additional est to fit
239        total_n_estimators = self.n_estimators
240
241        self.estimators_ = np.resize(self.estimators_, total_n_estimators)
242        self.train_score_ = np.resize(self.train_score_, total_n_estimators)
243        if self.subsample < 1 or hasattr(self, "oob_improvement_"):
244            # if do oob resize arrays or create new if not available
245            if hasattr(self, "oob_improvement_"):
246                self.oob_improvement_ = np.resize(
247                    self.oob_improvement_, total_n_estimators
248                )
249                self.oob_scores_ = np.resize(
250                    self.oob_scores_, total_n_estimators
251                )
252                self.oob_score_ = np.nan
253            else:
254                self.oob_improvement_ = np.zeros(
255                    total_n_estimators, dtype=np.float64
256                )
257                self.oob_scores_ = np.zeros(
258                    (total_n_estimators,), dtype=np.float64
259                )
260                self.oob_score_ = np.nan
261
262        if self.dropout_rate > 0:
263            if not hasattr(self, "_scale"):
264                raise ValueError(
265                    "fitting with warm_start=True and dropout_rate > 0 is only "
266                    "supported if the previous fit used dropout_rate > 0 too"
267                )
268
269            self._scale = np.resize(self._scale, total_n_estimators)
270            self._scale[self.n_estimators_:] = 1

289    def _update_with_dropout(self, i, X, raw_predictions, scale, random_state):
290        # select base learners to be dropped for next iteration
291        drop_model, n_dropped = _sample_binomial_plus_one(
292            self.dropout_rate, i + 1, random_state
293        )
294
295        # adjust scaling factor of tree that is going to be trained in next iteration
296        scale[i + 1] = 1.0 / (n_dropped + 1.0)
297
298        raw_predictions[:] = 0
299        for m in range(i + 1):
300            if drop_model[m] == 1:
301                # adjust scaling factor of dropped trees
302                scale[m] *= n_dropped / (n_dropped + 1.0)
303            else:
304                # pseudoresponse of next iteration (without contribution of dropped trees)
305                raw_predictions += (
306                    self.learning_rate
307                    * scale[m]
308                    * self.estimators_[m].predict(X)
309                )

311    def _fit(
312        self,
313        X,
314        event,
315        time,
316        y_pred,
317        sample_weight,
318        random_state,
319        begin_at_stage=0,
320    ):  # noqa: C901
321        n_samples = X.shape[0]
322        # account for intercept
323        y = np.fromiter(
324            zip(event, time), dtype=[("event", bool), ("time", np.float64)]
325        )
326
327        do_oob = self.subsample < 1.0
328        if do_oob:
329            n_inbag = max(1, int(self.subsample * n_samples))
330
331        do_dropout = self.dropout_rate > 0
332        if do_dropout:
333            scale = self._scale
334
335        if self.verbose:
336            verbose_reporter = VerboseReporter(verbose=self.verbose)
337            verbose_reporter.init(self, 0)
338
339        # perform boosting iterations
340        i = begin_at_stage
341        if self.show_progress:
342            iterator = tqdm(range(begin_at_stage, int(self.n_estimators)))
343        else:
344            iterator = range(begin_at_stage, int(self.n_estimators))
345        for i in iterator:
346            # subsampling
347            if do_oob:
348                sample_mask = _random_sample_mask(
349                    n_samples, n_inbag, random_state
350                )
351                subsample_weight = sample_weight * sample_mask.astype(
352                    np.float64
353                )
354
355                # OOB score before adding this stage
356                y_oob_masked = y[~sample_mask]
357                sample_weight_oob_masked = sample_weight[~sample_mask]
358                if i == 0:  # store the initial loss to compute the OOB score
359                    initial_loss = self.loss_obj(
360                        y_true=y_oob_masked,
361                        raw_prediction=y_pred[~sample_mask],
362                        sample_weight=sample_weight_oob_masked,
363                    )
364            else:
365                subsample_weight = sample_weight
366
367            residuals = self.loss_obj.gradient(
368                y, y_pred, sample_weight=sample_weight
369            )
370
371            best_learner = _fit_stage_componentwise(
372                X, residuals, subsample_weight, regr=self.regr
373            )
374            self.estimators_[i] = best_learner
375
376            if do_dropout and i < len(scale) - 1:
377                self._update_with_dropout(i, X, y_pred, scale, random_state)
378            else:
379                y_pred += self.learning_rate * best_learner.predict(X)
380
381            # track loss
382            if do_oob:
383                self.train_score_[i] = self.loss_obj(
384                    y_true=y[sample_mask],
385                    raw_prediction=y_pred[sample_mask],
386                    sample_weight=sample_weight[sample_mask],
387                )
388                self.oob_scores_[i] = self.loss_obj(
389                    y_true=y_oob_masked,
390                    raw_prediction=y_pred[~sample_mask],
391                    sample_weight=sample_weight_oob_masked,
392                )
393                previous_loss = (
394                    initial_loss if i == 0 else self.oob_scores_[i - 1]
395                )
396                self.oob_improvement_[i] = previous_loss - self.oob_scores_[i]
397                self.oob_score_ = self.oob_scores_[-1]
398            else:
399                # no need to fancy index w/ no subsampling
400                self.train_score_[i] = self.loss_obj(
401                    y_true=y, raw_prediction=y_pred, sample_weight=sample_weight
402                )
403
404            if self.verbose > 0:
405                verbose_reporter.update(i, self)
406
407        return i + 1

409    def fit(self, X, y, sample_weight=None):
410        """Fit estimator.
411
412        Parameters
413        ----------
414        X : array-like, shape = (n_samples, n_features)
415            Data matrix
416
417        y : structured array, shape = (n_samples,)
418            A structured array containing the binary event indicator
419            as first field, and time of event or time of censoring as
420            second field.
421
422        sample_weight : array-like, shape = (n_samples,), optional
423            Weights given to each sample. If omitted, all samples have weight 1.
424
425        Returns
426        -------
427        self
428        """
429        self._validate_params()
430
431        if not self.warm_start:
432            self._clear_state()
433
434        X = self._validate_data(X, ensure_min_samples=2)
435        event, time = check_array_survival(X, y)
436
437        sample_weight = _check_sample_weight(sample_weight, X)
438
439        n_samples = X.shape[0]
440        Xi = np.column_stack((np.ones(n_samples), X))
441
442        self.loss_obj = LOSS_FUNCTIONS[self.loss]()
443        if isinstance(
444            self.loss_obj, (CensoredSquaredLoss, IPCWLeastSquaresError)
445        ):
446            time = np.log(time)
447
448        if not self._is_fitted():
449            self._init_state()
450
451            y_pred = np.zeros(n_samples, dtype=np.float64)
452
453            begin_at_stage = 0
454
455            self._rng = check_random_state(self.random_state)
456        else:
457            # add more estimators to fitted model
458            # invariant: warm_start = True
459            if self.n_estimators < self.estimators_.shape[0]:
460                raise ValueError(
461                    "n_estimators=%d must be larger or equal to "
462                    "estimators_.shape[0]=%d when "
463                    "warm_start==True"
464                    % (self.n_estimators, self.estimators_.shape[0])
465                )
466            begin_at_stage = self.estimators_.shape[0]
467            y_pred = self._raw_predict(Xi)
468            self._resize_state()
469
470            # apply dropout to last stage of previous fit
471            if hasattr(self, "_scale") and self.dropout_rate > 0:
472                # pylint: disable-next=access-member-before-definition
473                self._update_with_dropout(
474                    self.n_estimators_ - 1, Xi, y_pred, self._scale, self._rng
475                )
476
477        self.n_estimators_ = self._fit(
478            Xi, event, time, y_pred, sample_weight, self._rng, begin_at_stage
479        )
480
481        self.set_baseline_model(X, event, time)
482        try:
483            self.X_ = self.regr.X_
484        except AttributeError:
485            pass
486        return self

488    def set_baseline_model(self, X, event, time):

497    def _raw_predict(self, X, **kwargs):
498        if ("return_pi" in kwargs) or ("return_std" in kwargs):
499            pred = {
500                "mean": np.zeros(
501                    X.shape[0], dtype=float
502                ),  # 4 is mean, std, lower, upper
503                "lower": np.zeros(X.shape[0], dtype=float),
504                "upper": np.zeros(X.shape[0], dtype=float),
505            }
506            for estimator in self.estimators_:
507                preds = estimator.predict(X, **kwargs)
508                if "return_std" in kwargs:
509                    pred["mean"] += self.learning_rate * preds.mean
510                    pred["lower"] += self.learning_rate * preds.upper
511                    pred["upper"] += self.learning_rate * preds.lower
512            DescribeResult = namedtuple(
513                "DescribeResult", ["mean", "lower", "upper"]
514            )
515            return DescribeResult(pred["mean"], pred["lower"], pred["upper"])
516        pred = np.zeros(X.shape[0], dtype=float)
517        for estimator in self.estimators_:
518            pred += self.learning_rate * estimator.predict(X, **kwargs)
519        return pred
520
521    def _predict(self, X, **kwargs):
522        # account for intercept
523        Xi = np.column_stack((np.ones(X.shape[0]), X))
524        pred = self._raw_predict(Xi, **kwargs)
525        if len(np.asarray(pred).shape) == 1:
526            return self.loss_obj._scale_raw_prediction(pred)
527        if ("return_std" in kwargs) or ("return_pi" in kwargs):
528            DescribeResult = namedtuple(
529                "DescribeResult", ["mean", "lower", "upper"]
530            )
531            res = [
532                self.loss_obj._scale_raw_prediction(p)
533                for p in (pred[0], pred[1], pred[2])
534            ]
535            return DescribeResult(res[0], res[1], res[2])

537    def predict(self, X, **kwargs):
538        """Predict risk scores.
539
540        If `loss='coxph'`, predictions can be interpreted as log hazard ratio
541        corresponding to the linear predictor of a Cox proportional hazards
542        model. If `loss='squared'` or `loss='ipcwls'`, predictions are the
543        time to event.
544
545        Parameters
546        ----------
547        X : array-like, shape = (n_samples, n_features)
548            Data matrix.
549
550        Returns
551        -------
552        risk_score : array, shape = (n_samples,)
553            Predicted risk scores.
554        """
555        check_is_fitted(self, "estimators_")
556        X = self._validate_data(X, reset=False)
557        return self._predict(X, **kwargs)

566    def predict_cumulative_hazard_function(
567        self, X, return_array=False, **kwargs
568    ):
569        """Predict cumulative hazard function.
570
571        Only available if :meth:`fit` has been called with `loss = "coxph"`.
572
573        The cumulative hazard function for an individual
574        with feature vector :math:`x` is defined as
575
576        .. math::
577
578            H(t \\mid x) = \\exp(f(x)) H_0(t) ,
579
580        where :math:`f(\\cdot)` is the additive ensemble of base learners,
581        and :math:`H_0(t)` is the baseline hazard function,
582        estimated by Breslow's estimator.
583
584        Parameters
585        ----------
586        X : array-like, shape = (n_samples, n_features)
587            Data matrix.
588
589        return_array : boolean, default: False
590            If set, return an array with the cumulative hazard rate
591            for each `self.unique_times_`, otherwise an array of
592            :class:`survivalist.functions.StepFunction`.
593
594        Returns
595        -------
596        cum_hazard : ndarray
597            If `return_array` is set, an array with the cumulative hazard rate
598            for each `self.unique_times_`, otherwise an array of length `n_samples`
599            of :class:`survivalist.functions.StepFunction` instances will be returned.
600
601        Examples
602        --------
603        >>> import matplotlib.pyplot as plt
604        >>> from survivalist.datasets import load_whas500
605        >>> from survivalist.ensemble import ComponentwiseGradientBoostingSurvivalAnalysis
606
607        Load the data.
608
609        >>> X, y = load_whas500()
610        >>> X = X.astype(float)
611
612        Fit the model.
613
614        >>> estimator = ComponentwiseGradientBoostingSurvivalAnalysis(loss="coxph").fit(X, y)
615
616        Estimate the cumulative hazard function for the first 10 samples.
617
618        >>> chf_funcs = estimator.predict_cumulative_hazard_function(X.iloc[:10])
619
620        Plot the estimated cumulative hazard functions.
621
622        >>> for fn in chf_funcs:
623        ...     plt.step(fn.x, fn(fn.x), where="post")
624        ...
625        >>> plt.ylim(0, 1)
626        >>> plt.show()
627        """
628        if self.quantiles_ is not None:
629            preds = self.predict(X, **kwargs)
630            DescribeResult = namedtuple(
631                "DescribeResult", ["mean", "lower", "upper"]
632            )
633            res = (
634                self._predict_cumulative_hazard_function(
635                    self.get_baseline_model(), preds.mean, return_array
636                ),
637                self._predict_cumulative_hazard_function(
638                    self.get_baseline_model(), preds.lower, return_array
639                ),
640                self._predict_cumulative_hazard_function(
641                    self.get_baseline_model(), preds.upper, return_array
642                ),
643            )
644            return DescribeResult(res[0], res[1], res[2])
645        return self._predict_cumulative_hazard_function(

649    def predict_survival_function(self, X, return_array=False, **kwargs):
650        """Predict survival function.
651
652        Only available if :meth:`fit` has been called with `loss = "coxph"`.
653
654        The survival function for an individual
655        with feature vector :math:`x` is defined as
656
657        .. math::
658
659            S(t \\mid x) = S_0(t)^{\\exp(f(x)} ,
660
661        where :math:`f(\\cdot)` is the additive ensemble of base learners,
662        and :math:`S_0(t)` is the baseline survival function,
663        estimated by Breslow's estimator.
664
665        Parameters
666        ----------
667        X : array-like, shape = (n_samples, n_features)
668            Data matrix.
669
670        return_array : boolean, default: False
671            If set, return an array with the probability
672            of survival for each `self.unique_times_`,
673            otherwise an array of :class:`survivalist.functions.StepFunction`.
674
675        Returns
676        -------
677        survival : ndarray
678            If `return_array` is set, an array with the probability of
679            survival for each `self.unique_times_`, otherwise an array of
680            length `n_samples` of :class:`survivalist.functions.StepFunction`
681            instances will be returned.
682
683        Examples
684        --------
685        >>> import matplotlib.pyplot as plt
686        >>> from survivalist.datasets import load_whas500
687        >>> from survivalist.ensemble import ComponentwiseGradientBoostingSurvivalAnalysis
688
689        Load the data.
690
691        >>> X, y = load_whas500()
692        >>> X = X.astype(float)
693
694        Fit the model.
695
696        >>> estimator = ComponentwiseGradientBoostingSurvivalAnalysis(loss="coxph").fit(X, y)
697
698        Estimate the survival function for the first 10 samples.
699
700        >>> surv_funcs = estimator.predict_survival_function(X.iloc[:10])
701
702        Plot the estimated survival functions.
703
704        >>> for fn in surv_funcs:
705        ...     plt.step(fn.x, fn(fn.x), where="post")
706        ...
707        >>> plt.ylim(0, 1)
708        >>> plt.show()
709        """
710        if ("return_pi" in kwargs) or ("return_std" in kwargs):
711            preds = self.predict(X, **kwargs)
712            DescribeResult = namedtuple(
713                "DescribeResult", ["mean", "lower", "upper"]
714            )
715            res = (
716                self._predict_survival_function(
717                    self.get_baseline_model(),
718                    preds.mean,
719                    return_array,
720                    **kwargs,
721                ),
722                self._predict_survival_function(
723                    self.get_baseline_model(),
724                    preds.lower,
725                    return_array,
726                    **kwargs,
727                ),
728                self._predict_survival_function(
729                    self.get_baseline_model(),
730                    preds.upper,
731                    return_array,
732                    **kwargs,
733                ),
734            )
735            return DescribeResult(res[0], res[1], res[2])
736        return self._predict_survival_function(
737            self.get_baseline_model(),
738            self.predict(X, **kwargs),
739            return_array,
740            **kwargs,
741        )

 95        There is a trade-off between `learning_rate` and `n_estimators`.
 96        Values must be in the range `[0.0, inf)`.
 97

101        results in better performance.

106        learners. If smaller than 1.0 this results in Stochastic Gradient
107        Boosting. `subsample` interacts with the parameter `n_estimators`.
108        Choosing `subsample < 1.0` leads to a reduction of variance
109        and an increase in bias.

122        to shrinkage, i.e., setting `learning_rate < 1.0`.
123        Values must be in the range `[0.0, 1.0)`.
124

126        The seed of the pseudo random number generator to use when
127        shuffling the data.

131        once in a while.

290    def _update_with_dropout(self, i, X, raw_predictions, scale, random_state):
291        # select base learners to be dropped for next iteration
292        drop_model, n_dropped = _sample_binomial_plus_one(
293            self.dropout_rate, i + 1, random_state
294        )
295
296        # adjust scaling factor of tree that is going to be trained in next iteration
297        scale[i + 1] = 1.0 / (n_dropped + 1.0)
298
299        raw_predictions[:] = 0
300        for m in range(i + 1):
301            if drop_model[m] == 1:
302                # adjust scaling factor of dropped trees
303                scale[m] *= n_dropped / (n_dropped + 1.0)
304            else:
305                # pseudoresponse of next iteration (without contribution of dropped trees)
306                raw_predictions += (
307                    self.learning_rate
308                    * scale[m]
309                    * self.estimators_[m].predict(X)
310                )

312    def _fit(
313        self,
314        X,
315        event,
316        time,
317        y_pred,
318        sample_weight,
319        random_state,
320        begin_at_stage=0,
321    ):  # noqa: C901
322        n_samples = X.shape[0]
323        # account for intercept
324        y = np.fromiter(
325            zip(event, time), dtype=[("event", bool), ("time", np.float64)]
326        )
327
328        do_oob = self.subsample < 1.0
329        if do_oob:
330            n_inbag = max(1, int(self.subsample * n_samples))
331
332        do_dropout = self.dropout_rate > 0
333        if do_dropout:
334            scale = self._scale
335
336        if self.verbose:
337            verbose_reporter = VerboseReporter(verbose=self.verbose)
338            verbose_reporter.init(self, 0)
339
340        # perform boosting iterations
341        i = begin_at_stage
342        if self.show_progress:
343            iterator = tqdm(range(begin_at_stage, int(self.n_estimators)))
344        else:
345            iterator = range(begin_at_stage, int(self.n_estimators))
346        for i in iterator:
347            # subsampling
348            if do_oob:
349                sample_mask = _random_sample_mask(
350                    n_samples, n_inbag, random_state
351                )
352                subsample_weight = sample_weight * sample_mask.astype(
353                    np.float64
354                )
355
356                # OOB score before adding this stage
357                y_oob_masked = y[~sample_mask]
358                sample_weight_oob_masked = sample_weight[~sample_mask]
359                if i == 0:  # store the initial loss to compute the OOB score
360                    initial_loss = self.loss_obj(
361                        y_true=y_oob_masked,
362                        raw_prediction=y_pred[~sample_mask],
363                        sample_weight=sample_weight_oob_masked,
364                    )
365            else:
366                subsample_weight = sample_weight
367
368            residuals = self.loss_obj.gradient(
369                y, y_pred, sample_weight=sample_weight
370            )
371
372            best_learner = _fit_stage_componentwise(
373                X, residuals, subsample_weight, regr=self.regr
374            )
375            self.estimators_[i] = best_learner
376
377            if do_dropout and i < len(scale) - 1:
378                self._update_with_dropout(i, X, y_pred, scale, random_state)
379            else:
380                y_pred += self.learning_rate * best_learner.predict(X)
381
382            # track loss
383            if do_oob:
384                self.train_score_[i] = self.loss_obj(
385                    y_true=y[sample_mask],
386                    raw_prediction=y_pred[sample_mask],
387                    sample_weight=sample_weight[sample_mask],
388                )
389                self.oob_scores_[i] = self.loss_obj(
390                    y_true=y_oob_masked,
391                    raw_prediction=y_pred[~sample_mask],
392                    sample_weight=sample_weight_oob_masked,
393                )
394                previous_loss = (
395                    initial_loss if i == 0 else self.oob_scores_[i - 1]
396                )
397                self.oob_improvement_[i] = previous_loss - self.oob_scores_[i]
398                self.oob_score_ = self.oob_scores_[-1]
399            else:
400                # no need to fancy index w/ no subsampling
401                self.train_score_[i] = self.loss_obj(
402                    y_true=y, raw_prediction=y_pred, sample_weight=sample_weight
403                )
404
405            if self.verbose > 0:
406                verbose_reporter.update(i, self)
407
408        return i + 1

410    def fit(self, X, y, sample_weight=None):
411        """Fit estimator.
412
413        Parameters
414        ----------
415        X : array-like, shape = (n_samples, n_features)
416            Data matrix
417
418        y : structured array, shape = (n_samples,)
419            A structured array containing the binary event indicator
420            as first field, and time of event or time of censoring as
421            second field.
422
423        sample_weight : array-like, shape = (n_samples,), optional
424            Weights given to each sample. If omitted, all samples have weight 1.
425
426        Returns
427        -------
428        self
429        """
430        self._validate_params()
431
432        if not self.warm_start:
433            self._clear_state()
434
435        X = self._validate_data(X, ensure_min_samples=2)
436        event, time = check_array_survival(X, y)
437
438        sample_weight = _check_sample_weight(sample_weight, X)
439
440        n_samples = X.shape[0]
441        Xi = np.column_stack((np.ones(n_samples), X))
442
443        self.loss_obj = LOSS_FUNCTIONS[self.loss]()
444        if isinstance(
445            self.loss_obj, (CensoredSquaredLoss, IPCWLeastSquaresError)
446        ):
447            time = np.log(time)
448
449        if not self._is_fitted():
450            self._init_state()
451
452            y_pred = np.zeros(n_samples, dtype=np.float64)
453
454            begin_at_stage = 0
455
456            self._rng = check_random_state(self.random_state)
457        else:
458            # add more estimators to fitted model
459            # invariant: warm_start = True
460            if self.n_estimators < self.estimators_.shape[0]:
461                raise ValueError(
462                    "n_estimators=%d must be larger or equal to "
463                    "estimators_.shape[0]=%d when "
464                    "warm_start==True"
465                    % (self.n_estimators, self.estimators_.shape[0])
466                )
467            begin_at_stage = self.estimators_.shape[0]
468            y_pred = self._raw_predict(Xi)
469            self._resize_state()
470
471            # apply dropout to last stage of previous fit
472            if hasattr(self, "_scale") and self.dropout_rate > 0:
473                # pylint: disable-next=access-member-before-definition
474                self._update_with_dropout(
475                    self.n_estimators_ - 1, Xi, y_pred, self._scale, self._rng
476                )
477
478        self.n_estimators_ = self._fit(
479            Xi, event, time, y_pred, sample_weight, self._rng, begin_at_stage
480        )
481
482        self.set_baseline_model(X, event, time)
483        try:
484            self.X_ = self.regr.X_
485        except AttributeError:
486            pass
487        return self

489    def set_baseline_model(self, X, event, time):
490        if isinstance(self.loss_obj, CoxPH):
491            risk_scores = self._predict(X)
492            self.baseline_model = BreslowEstimator().fit(
493                risk_scores, event, time
494            )
495        else:
496            self.baseline_model = None

498    def _raw_predict(self, X, **kwargs):
499        if ("return_pi" in kwargs) or ("return_std" in kwargs):
500            pred = {
501                "mean": np.zeros(
502                    X.shape[0], dtype=float
503                ),  # 4 is mean, std, lower, upper
504                "lower": np.zeros(X.shape[0], dtype=float),
505                "upper": np.zeros(X.shape[0], dtype=float),
506            }
507            for estimator in self.estimators_:
508                preds = estimator.predict(X, **kwargs)
509                if "return_std" in kwargs:
510                    pred["mean"] += self.learning_rate * preds.mean
511                    pred["lower"] += self.learning_rate * preds.upper
512                    pred["upper"] += self.learning_rate * preds.lower
513            DescribeResult = namedtuple(
514                "DescribeResult", ["mean", "lower", "upper"]
515            )
516            return DescribeResult(pred["mean"], pred["lower"], pred["upper"])
517        pred = np.zeros(X.shape[0], dtype=float)
518        for estimator in self.estimators_:
519            pred += self.learning_rate * estimator.predict(X, **kwargs)
520        return pred
521
522    def _predict(self, X, **kwargs):
523        # account for intercept
524        Xi = np.column_stack((np.ones(X.shape[0]), X))
525        pred = self._raw_predict(Xi, **kwargs)
526        if len(np.asarray(pred).shape) == 1:
527            return self.loss_obj._scale_raw_prediction(pred)
528        if ("return_std" in kwargs) or ("return_pi" in kwargs):
529            DescribeResult = namedtuple(
530                "DescribeResult", ["mean", "lower", "upper"]
531            )
532            res = [
533                self.loss_obj._scale_raw_prediction(p)
534                for p in (pred[0], pred[1], pred[2])
535            ]
536            return DescribeResult(res[0], res[1], res[2])

567    def predict_cumulative_hazard_function(
568        self, X, return_array=False, **kwargs
569    ):
570        """Predict cumulative hazard function.
571
572        Only available if :meth:`fit` has been called with `loss = "coxph"`.
573
574        The cumulative hazard function for an individual
575        with feature vector :math:`x` is defined as
576
577        .. math::
578
579            H(t \\mid x) = \\exp(f(x)) H_0(t) ,
580
581        where :math:`f(\\cdot)` is the additive ensemble of base learners,
582        and :math:`H_0(t)` is the baseline hazard function,
583        estimated by Breslow's estimator.
584
585        Parameters
586        ----------
587        X : array-like, shape = (n_samples, n_features)
588            Data matrix.
589
590        return_array : boolean, default: False
591            If set, return an array with the cumulative hazard rate
592            for each `self.unique_times_`, otherwise an array of
593            :class:`survivalist.functions.StepFunction`.
594
595        Returns
596        -------
597        cum_hazard : ndarray
598            If `return_array` is set, an array with the cumulative hazard rate
599            for each `self.unique_times_`, otherwise an array of length `n_samples`
600            of :class:`survivalist.functions.StepFunction` instances will be returned.
601
602        Examples
603        --------
604        >>> import matplotlib.pyplot as plt
605        >>> from survivalist.datasets import load_whas500
606        >>> from survivalist.ensemble import ComponentwiseGradientBoostingSurvivalAnalysis

Generic Gradient boosting with any base learner.

Parameters

regr : object, optional, default: LinearRegression() The base learner to be used for fitting the residuals. The base learner must have a fit and predict method. The default is LinearRegression.

loss : {'coxph', 'squared', 'ipcwls'}, optional, default: 'coxph' loss function to be optimized. 'coxph' refers to partial likelihood loss of Cox's proportional hazards model. The loss 'squared' minimizes a squared regression loss that ignores predictions beyond the time of censoring, and 'ipcwls' refers to inverse-probability of censoring weighted least squares error.

learning_rate : float, optional, default: 0.1 learning rate shrinks the contribution of each base learner by learning_rate. There is a trade-off between learning_rate and n_estimators. Values must be in the range [0.0, inf).

n_estimators : int, default: 100 The number of boosting stages to perform. Gradient boosting is fairly robust to over-fitting so a large number usually results in better performance. Values must be in the range [1, inf).

subsample : float, optional, default: 1.0 The fraction of samples to be used for fitting the individual base learners. If smaller than 1.0 this results in Stochastic Gradient Boosting. subsample interacts with the parameter n_estimators. Choosing subsample < 1.0 leads to a reduction of variance and an increase in bias. Values must be in the range (0.0, 1.0].

warm_start : bool, default: False When set to True, reuse the solution of the previous call to fit and add more estimators to the ensemble, otherwise, just erase the previous solution.

dropout_rate : float, optional, default: 0.0 If larger than zero, the residuals at each iteration are only computed from a random subset of base learners. The value corresponds to the percentage of base learners that are dropped. In each iteration, at least one base learner is dropped. This is an alternative regularization to shrinkage, i.e., setting learning_rate < 1.0. Values must be in the range [0.0, 1.0).

random_state : int seed, RandomState instance, or None, default: None The seed of the pseudo random number generator to use when shuffling the data.

verbose : int, default: 0 Enable verbose output. If 1 then it prints progress and performance once in a while. Values must be in the range [0, inf).

show_progress : bool, default: True If set, show a progress bar for the fitting process.

Attributes

estimators_ : list of base learners The collection of fitted sub-estimators.

train_score_ : ndarray, shape = (n_estimators,) The i-th score train_score_[i] is the loss of the model at iteration i on the in-bag sample. If subsample == 1 this is the loss on the training data.

oob_improvement_ : ndarray, shape = (n_estimators,) The improvement in loss on the out-of-bag samples relative to the previous iteration. oob_improvement_[0] is the improvement in loss of the first stage over the init estimator. Only available if subsample < 1.0.

oob_scores_ : ndarray of shape (n_estimators,) The full history of the loss values on the out-of-bag samples. Only available if subsample < 1.0.

oob_score_ : float The last value of the loss on the out-of-bag samples. It is the same as oob_scores_[-1]. Only available if subsample < 1.0.

n_features_in_ : int Number of features seen during fit.

feature_names_in_ : ndarray of shape (n_features_in_,) Names of features seen during fit. Defined only when X has feature names that are all strings.

unique_times_ : array of shape = (n_unique_times,) Unique time points.
