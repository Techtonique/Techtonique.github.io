 130    def _setup_jax_backend(self):
 131        """Setup JAX backend for GPU/TPU acceleration."""
 132        if not JAX_AVAILABLE:

 135        # JIT compiled distance functions
 136        @jit
 137        def pairwise_sq_dists_jax(X1, X2):
 138            X1_sq = jnp.sum(X1**2, axis=1)[:, jnp.newaxis]
 139            X2_sq = jnp.sum(X2**2, axis=1)[jnp.newaxis, :]
 140            return X1_sq + X2_sq - 2 * X1 @ X2.T

 264    def _pairwise_sq_dists(self, X1, X2):
 265        """Compute pairwise squared Euclidean distances."""
 266        if self.backend in ["gpu", "tpu"] and JAX_AVAILABLE:
 267            X1_jax = jnp.array(X1)
 268            X2_jax = jnp.array(X2)
 269            result = self._pairwise_sq_dists_jax(X1_jax, X2_jax)
 270            return np.array(result)
 271        else:
 272            X1 = np.atleast_2d(X1)
 273            X2 = np.atleast_2d(X2)
 274            return (
 275                np.sum(X1**2, axis=1)[:, np.newaxis]
 276                + np.sum(X2**2, axis=1)[np.newaxis, :]
 277                - 2 * X1 @ X2.T
 278            )

 309    def _mmd(self, u, v, kernel_sigma=1):
 310        """Maximum Mean Discrepancy between two distributions."""
 311        if u.ndim == 1:
 312            u = u.reshape(-1, 1)
 313        if v.ndim == 1:
 314            v = v.reshape(-1, 1)
 315
 316        def kmat(A, B):
 317            return np.exp(
 318                -self._pairwise_sq_dists(A, B) / (2 * kernel_sigma**2)
 319            )
 320
 321        return (
 322            np.mean(kmat(u, u)) + np.mean(kmat(v, v)) - 2 * np.mean(kmat(u, v))
 323        )

 325    def _custom_energy_distance(self, u, v):
 326        """Energy distance between two distributions."""
 327        if u.ndim == 1:
 328            u = u.reshape(-1, 1)
 329        if v.ndim == 1:
 330            v = v.reshape(-1, 1)
 331
 332        n, d = u.shape
 333        m = v.shape[0]
 334
 335        if self.backend in ["gpu", "tpu"] and JAX_AVAILABLE:
 336            # JAX implementation
 337            u_jax = jnp.array(u)
 338            v_jax = jnp.array(v)
 339            dist_xx = self._cdist_jax(u_jax, u_jax)
 340            dist_yy = self._cdist_jax(v_jax, v_jax)
 341            dist_xy = self._cdist_jax(u_jax, v_jax)
 342            term1 = 2 * jnp.sum(dist_xy) / (n * m)
 343            term2 = jnp.sum(dist_xx) / (n * n)
 344            term3 = jnp.sum(dist_yy) / (m * m)
 345            return float(term1 - term2 - term3)
 346        else:
 347            # NumPy implementation
 348            dist_xx = cdist(u, u, metric="euclidean")
 349            dist_yy = cdist(v, v, metric="euclidean")
 350            dist_xy = cdist(u, v, metric="euclidean")
 351            term1 = 2 * np.sum(dist_xy) / (n * m)
 352            term2 = np.sum(dist_xx) / (n * n)
 353            term3 = np.sum(dist_yy) / (m * m)
 354            return term1 - term2 - term3

 373    def fit(self, Y, n_train=None, metric="energy", n_trials=50, **kwargs):
 374        """
 375        Fit the data generator to match the distribution of Y.
 376
 377        Parameters:
 378        -----------
 379        Y : array-like, shape (n_samples, n_features)
 380            Target multivariate data to emulate
 381        n_train : int, default=None
 382            Number of training samples (default: n_samples // 2)
 383        metric : str, default='energy'
 384            Distance metric for optimization ('energy', 'mmd', or 'wasserstein')
 385        n_trials : int, default=50
 386            Number of Optuna optimization trials
 387        **kwargs : dict
 388            Additional arguments for Optuna optimization
 389
 390        Returns:
 391        --------
 392        self : object
 393            Returns self
 394        """
 395        if Y.ndim == 1:
 396            Y = Y.reshape(-1, 1)
 397
 398        n, d = Y.shape
 399        self.n_features_ = d
 400        # Determine whether to use RFF
 401        if self.use_rff == "auto":
 402            self.actual_use_rff_ = n >= self.force_rff_threshold
 403        else:
 404            self.actual_use_rff_ = self.use_rff
 405        # Auto-enable RFF for large datasets with component determination
 406        if self.actual_use_rff_:
 407            self.actual_rff_components_ = self._determine_components(n)
 408            if self.use_rff == "auto":
 409                print(
 410                    f"Large dataset detected (n={n}). Auto-enabling {self.kernel_approximation.upper()} for scalability."
 411                )
 412
 413        if n_train is None:
 414            n_train = n // 2
 415        # Store the input distribution function
 416        self.X_dist = np.random.normal(0, 1, (n, d))
 417        # Create stratified train-test split
 418        train_idx, test_idx = self._stratified_train_test_split(Y, n_train)
 419        Y_train = Y[train_idx]
 420        Y_test = Y[test_idx]
 421        X_train = self.X_dist[:n_train]
 422
 423        def objective(trial):
 424            sigma = trial.suggest_float("sigma", 0.01, 10, log=True)
 425            lambd = trial.suggest_float("lambd", 1e-5, 1, log=True)
 426            gamma = 1 / (2 * sigma**2)
 427            # Create model with current parameters
 428            model = self._create_model(gamma, lambd)
 429            model.fit(X_train, Y_train)
 430            preds_train = model.predict(X_train)
 431            if preds_train.ndim == 1:
 432                preds_train = preds_train.reshape(-1, 1)
 433            res = Y_train - preds_train
 434            Y_sim = self._generate_pseudo_with_model(model, res, len(Y_test))
 435            if metric == "energy":
 436                dist_val = self._custom_energy_distance(Y_test, Y_sim)
 437            elif metric == "mmd":
 438                dist_val = self._mmd(Y_test, Y_sim)
 439            elif metric == "wasserstein" and d == 1:
 440                dist_val = stats.wasserstein_distance(
 441                    Y_test.flatten(), Y_sim.flatten()
 442                )
 443            else:
 444                raise ValueError("Invalid metric for dimension")
 445            return dist_val
 446
 447        # Optimize hyperparameters
 448        study = optuna.create_study(direction="minimize")
 449        study.optimize(objective, n_trials=n_trials, **kwargs)
 450        # Store best parameters and fit final model
 451        self.best_params_ = study.best_params
 452        self.best_score_ = study.best_value
 453        sigma = self.best_params_["sigma"]
 454        lambd = self.best_params_["lambd"]
 455        gamma = 1 / (2 * sigma**2)
 456        # Fit final model with best parameters
 457        self.model = self._create_model(gamma, lambd)
 458        self.model.fit(X_train, Y_train)
 459        # Compute residuals
 460        preds_train = self.model.predict(X_train)
 461        if preds_train.ndim == 1:
 462            preds_train = preds_train.reshape(-1, 1)
 463        self.residuals_ = Y_train - preds_train
 464        # Fit the residual sampler
 465        self._fit_residual_sampler()
 466        self.is_fitted = True
 467        # Print final configuration
 468        if self.actual_use_rff_:
 469            print(
 470                f"  Using {self.kernel_approximation.upper()} with {self.actual_rff_components_} components"
 471            )
 472        else:
 473            print(f"  Using standard kernel method")
 474        return self

 683    def _perm_test(self, Y_orig, Y_sim, stat_func, n_perm=1000):
 684        """Permutation test for distribution comparison."""
 685        if Y_orig.ndim == 1:
 686            Y_orig = Y_orig.reshape(-1, 1)
 687        if Y_sim.ndim == 1:
 688            Y_sim = Y_sim.reshape(-1, 1)
 689
 690        obs = stat_func(Y_orig, Y_sim)
 691        combined = np.vstack((Y_orig, Y_sim))
 692        n1 = Y_orig.shape[0]
 693        perms = np.zeros(n_perm)
 694
 695        for i in range(n_perm):
 696            idx = np.random.permutation(combined.shape[0])
 697            p1 = combined[idx[:n1]]
 698            p2 = combined[idx[n1:]]
 699            perms[i] = stat_func(p1, p2)
 700
 701        pval = (np.sum(perms >= obs) + 1) / (n_perm + 1)
 702        return obs, pval

 704    def _fisher_z_test(self, r1, r2, n1, n2):
 705        """Fisher z-test for comparing correlation coefficients."""
 706        z1 = np.arctanh(r1)
 707        z2 = np.arctanh(r2)
 708        z = (z1 - z2) / np.sqrt(1 / (n1 - 3) + 1 / (n2 - 3))
 709        p = 2 * (1 - stats.norm.cdf(np.abs(z)))
 710        return z, p

 200                    ("approx", approximator),
 201                    ("ridge", Ridge(alpha=alpha)),

 205        return KernelRidge(kernel=self.kernel, gamma=gamma, alpha=alpha)

 207    def _fit_residual_sampler(self, **kwargs):

 224                n_components=min(self.gmm_components, len(self.residuals_)),
 225                random_state=self.random_state,

 230    def _sample_residuals(self, num_samples):
 231        """Sample residuals using the chosen method."""
 232        if self.residuals_ is None:
 233            raise ValueError("No residuals available for sampling")
 234
 235        if self.residual_sampling == "bootstrap":
 236            # Original bootstrap method
 237            n = self.residuals_.shape[0]
 238            idx = np.random.choice(n, num_samples, replace=True)
 239            return self.residuals_[idx]
 240
 241        elif self.residual_sampling == "kde":
 242            # Kernel Density Estimation sampling
 243            if self.kde_model_ is None:
 244                raise ValueError(
 245                    "KDE model not fitted. Call _fit_residual_sampler first."
 246                )
 247            # Sample from KDE
 248            return self.kde_model_.sample(num_samples)
 249
 250        elif self.residual_sampling == "gmm":
 251            # Gaussian Mixture Model sampling
 252            if self.gmm_model_ is None:
 253                raise ValueError(
 254                    "GMM model not fitted. Call _fit_residual_sampler first."
 255                )
 256            # Sample from GMM
 257            return self.gmm_model_.sample(num_samples)[0]
 258
 259        else:
 260            # Should not reach here due to validation in __init__
 261            raise ValueError(
 262                f"Unknown sampling method: {self.residual_sampling}"
 263            )

 265    def _pairwise_sq_dists(self, X1, X2):
 266        """Compute pairwise squared Euclidean distances."""
 267        if self.backend in ["gpu", "tpu"] and JAX_AVAILABLE:
 268            X1_jax = jnp.array(X1)
 269            X2_jax = jnp.array(X2)
 270            result = self._pairwise_sq_dists_jax(X1_jax, X2_jax)
 271            return np.array(result)
 272        else:
 273            X1 = np.atleast_2d(X1)
 274            X2 = np.atleast_2d(X2)
 275            return (
 276                np.sum(X1**2, axis=1)[:, np.newaxis]
 277                + np.sum(X2**2, axis=1)[np.newaxis, :]
 278                - 2 * X1 @ X2.T
 279            )

 310    def _mmd(self, u, v, kernel_sigma=1):
 311        """Maximum Mean Discrepancy between two distributions."""
 312        if u.ndim == 1:
 313            u = u.reshape(-1, 1)
 314        if v.ndim == 1:
 315            v = v.reshape(-1, 1)
 316
 317        def kmat(A, B):
 318            return np.exp(
 319                -self._pairwise_sq_dists(A, B) / (2 * kernel_sigma**2)
 320            )
 321
 322        return (
 323            np.mean(kmat(u, u)) + np.mean(kmat(v, v)) - 2 * np.mean(kmat(u, v))
 324        )

 326    def _custom_energy_distance(self, u, v):
 327        """Energy distance between two distributions."""
 328        if u.ndim == 1:
 329            u = u.reshape(-1, 1)
 330        if v.ndim == 1:
 331            v = v.reshape(-1, 1)
 332
 333        n, d = u.shape
 334        m = v.shape[0]
 335
 336        if self.backend in ["gpu", "tpu"] and JAX_AVAILABLE:
 337            # JAX implementation
 338            u_jax = jnp.array(u)
 339            v_jax = jnp.array(v)
 340            dist_xx = self._cdist_jax(u_jax, u_jax)
 341            dist_yy = self._cdist_jax(v_jax, v_jax)
 342            dist_xy = self._cdist_jax(u_jax, v_jax)
 343            term1 = 2 * jnp.sum(dist_xy) / (n * m)
 344            term2 = jnp.sum(dist_xx) / (n * n)
 345            term3 = jnp.sum(dist_yy) / (m * m)
 346            return float(term1 - term2 - term3)
 347        else:
 348            # NumPy implementation
 349            dist_xx = cdist(u, u, metric="euclidean")
 350            dist_yy = cdist(v, v, metric="euclidean")
 351            dist_xy = cdist(u, v, metric="euclidean")
 352            term1 = 2 * np.sum(dist_xy) / (n * m)
 353            term2 = np.sum(dist_xx) / (n * n)
 354            term3 = np.sum(dist_yy) / (m * m)
 355            return term1 - term2 - term3

 374    def fit(self, Y, n_train=None, metric="energy", n_trials=50, **kwargs):
 375        """
 376        Fit the data generator to match the distribution of Y.
 377
 378        Parameters:
 379        -----------
 380        Y : array-like, shape (n_samples, n_features)
 381            Target multivariate data to emulate
 382        n_train : int, default=None
 383            Number of training samples (default: n_samples // 2)
 384        metric : str, default='energy'
 385            Distance metric for optimization ('energy', 'mmd', or 'wasserstein')
 386        n_trials : int, default=50
 387            Number of Optuna optimization trials
 388        **kwargs : dict
 389            Additional arguments for Optuna optimization
 390
 391        Returns:
 392        --------
 393        self : object
 394            Returns self
 395        """
 396        if Y.ndim == 1:
 397            Y = Y.reshape(-1, 1)
 398
 399        n, d = Y.shape
 400        self.n_features_ = d
 401        # Determine whether to use RFF
 402        if self.use_rff == "auto":
 403            self.actual_use_rff_ = n >= self.force_rff_threshold
 404        else:
 405            self.actual_use_rff_ = self.use_rff
 406        # Auto-enable RFF for large datasets with component determination
 407        if self.actual_use_rff_:
 408            self.actual_rff_components_ = self._determine_components(n)
 409            if self.use_rff == "auto":
 410                print(
 411                    f"Large dataset detected (n={n}). Auto-enabling {self.kernel_approximation.upper()} for scalability."
 412                )
 413
 414        if n_train is None:
 415            n_train = n // 2
 416        # Store the input distribution function
 417        self.X_dist = np.random.normal(0, 1, (n, d))
 418        # Create stratified train-test split
 419        train_idx, test_idx = self._stratified_train_test_split(Y, n_train)
 420        Y_train = Y[train_idx]
 421        Y_test = Y[test_idx]
 422        X_train = self.X_dist[:n_train]
 423
 424        def objective(trial):
 425            sigma = trial.suggest_float("sigma", 0.01, 10, log=True)
 426            lambd = trial.suggest_float("lambd", 1e-5, 1, log=True)
 427            gamma = 1 / (2 * sigma**2)
 428            # Create model with current parameters
 429            model = self._create_model(gamma, lambd)
 430            model.fit(X_train, Y_train)
 431            preds_train = model.predict(X_train)
 432            if preds_train.ndim == 1:
 433                preds_train = preds_train.reshape(-1, 1)
 434            res = Y_train - preds_train
 435            Y_sim = self._generate_pseudo_with_model(model, res, len(Y_test))
 436            if metric == "energy":
 437                dist_val = self._custom_energy_distance(Y_test, Y_sim)
 438            elif metric == "mmd":
 439                dist_val = self._mmd(Y_test, Y_sim)
 440            elif metric == "wasserstein" and d == 1:
 441                dist_val = stats.wasserstein_distance(
 442                    Y_test.flatten(), Y_sim.flatten()
 443                )
 444            else:
 445                raise ValueError("Invalid metric for dimension")
 446            return dist_val
 447
 448        # Optimize hyperparameters
 449        study = optuna.create_study(direction="minimize")
 450        study.optimize(objective, n_trials=n_trials, **kwargs)
 451        # Store best parameters and fit final model
 452        self.best_params_ = study.best_params
 453        self.best_score_ = study.best_value
 454        sigma = self.best_params_["sigma"]
 455        lambd = self.best_params_["lambd"]
 456        gamma = 1 / (2 * sigma**2)
 457        # Fit final model with best parameters
 458        self.model = self._create_model(gamma, lambd)
 459        self.model.fit(X_train, Y_train)
 460        # Compute residuals
 461        preds_train = self.model.predict(X_train)
 462        if preds_train.ndim == 1:
 463            preds_train = preds_train.reshape(-1, 1)
 464        self.residuals_ = Y_train - preds_train
 465        # Fit the residual sampler
 466        self._fit_residual_sampler()
 467        self.is_fitted = True
 468        # Print final configuration
 469        if self.actual_use_rff_:
 470            print(
 471                f"  Using {self.kernel_approximation.upper()} with {self.actual_rff_components_} components"
 472            )
 473        else:
 474            print(f"  Using standard kernel method")
 475        return self

 684    def _perm_test(self, Y_orig, Y_sim, stat_func, n_perm=1000):
 685        """Permutation test for distribution comparison."""
 686        if Y_orig.ndim == 1:
 687            Y_orig = Y_orig.reshape(-1, 1)
 688        if Y_sim.ndim == 1:
 689            Y_sim = Y_sim.reshape(-1, 1)
 690
 691        obs = stat_func(Y_orig, Y_sim)
 692        combined = np.vstack((Y_orig, Y_sim))
 693        n1 = Y_orig.shape[0]
 694        perms = np.zeros(n_perm)
 695
 696        for i in range(n_perm):
 697            idx = np.random.permutation(combined.shape[0])
 698            p1 = combined[idx[:n1]]
 699            p2 = combined[idx[n1:]]
 700            perms[i] = stat_func(p1, p2)
 701
 702        pval = (np.sum(perms >= obs) + 1) / (n_perm + 1)
 703        return obs, pval

 705    def _fisher_z_test(self, r1, r2, n1, n2):
 706        """Fisher z-test for comparing correlation coefficients."""
 707        z1 = np.arctanh(r1)
 708        z2 = np.arctanh(r2)
 709        z = (z1 - z2) / np.sqrt(1 / (n1 - 3) + 1 / (n2 - 3))
 710        p = 2 * (1 - stats.norm.cdf(np.abs(z)))
 711        return z, p

 713    def test_similarity(self, Y_orig, Y_sim, n_perm=1000):
 714        """
 715        Test statistical similarity between original and synthetic data.
 716
 717        Parameters:
 718        -----------
 719        Y_orig : array-like
 720            Original data
 721        Y_sim : array-like
 722            Synthetic data
 723        n_perm : int, default=1000
 724            Number of permutations for permutation tests
 725
 726        Returns:
 727        --------
 728        results : dict
 729            Dictionary containing test results
 730        """
 731        if Y_orig.ndim == 1:
 732            Y_orig = Y_orig.reshape(-1, 1)
 733        if Y_sim.ndim == 1:
 734            Y_sim = Y_sim.reshape(-1, 1)
 735
 736        d = Y_orig.shape[1]
 737        results = {}
 738        # Test 1: Perm with energy
 739        results["energy_perm"] = self._perm_test(
 740            Y_orig, Y_sim, self._custom_energy_distance, n_perm
 741        )
 742        # Test 2: Perm with MMD
 743        results["mmd_perm"] = self._perm_test(
 744            Y_orig, Y_sim, lambda u, v: self._mmd(u, v), n_perm
 745        )
 746
 747        # Test 3: Perm with avg Wasserstein on margins
 748        def avg_wass(u, v):
 749            return np.mean(
 750                [stats.wasserstein_distance(u[:, i], v[:, i]) for i in range(d)]
 751            )
 752
 753        results["avg_wass_perm"] = self._perm_test(
 754            Y_orig, Y_sim, avg_wass, n_perm
 755        )
 756        # Test 4: Min p-value from marginal KS tests
 757        ps_ks = [
 758            stats.ks_2samp(Y_orig[:, i], Y_sim[:, i]).pvalue for i in range(d)
 759        ]
 760        results["min_marginal_ks_p"] = min(ps_ks)
 761        # Test 5: Min p-value from marginal Anderson-Darling tests
 762        ps_ad = [
 763            stats.anderson_ksamp([Y_orig[:, i], Y_sim[:, i]]).significance_level
 764            for i in range(d)
 765        ]
 766        results["min_marginal_ad_p"] = min(ps_ad)
 767        # Test 6: Min p-value from marginal Cramer-von Mises tests
 768        ps_cvm = [
 769            stats.cramervonmises_2samp(Y_orig[:, i], Y_sim[:, i]).pvalue
 770            for i in range(d)
 771        ]
 772        results["min_marginal_cvm_p"] = min(ps_cvm)
 773        # Correlation test: Compare all pairwise correlations
 774        corr_results = {}
 775        pairs = [(i, j) for i in range(d) for j in range(i + 1, d)]
 776        for i, j in pairs:
 777            r_orig = stats.pearsonr(Y_orig[:, i], Y_orig[:, j])[0]
 778            r_sim = stats.pearsonr(Y_sim[:, i], Y_sim[:, j])[0]
 779            z, p = self._fisher_z_test(r_orig, r_sim, len(Y_orig), len(Y_sim))
 780            corr_results[f"corr_dim{i+1}_dim{j+1}"] = (r_orig, r_sim, z, p)
 781        results["corr_tests"] = corr_results
 782        return results
