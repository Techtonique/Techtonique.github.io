 80    def _setup_jax_backend(self):
 81        if not JAX_AVAILABLE:
 82            raise ImportError("JAX is required for GPU/TPU backend")
 83
 84        @jit
 85        def pairwise_sq_dists_jax(X1, X2):
 86            X1_sq = jnp.sum(X1**2, axis=1)[:, jnp.newaxis]
 87            X2_sq = jnp.sum(X2**2, axis=1)[jnp.newaxis, :]
 88            return X1_sq + X2_sq - 2 * X1 @ X2.T
 89
 90        @jit
 91        def cdist_jax(X1, X2):
 92            return vmap(
 93                lambda x: vmap(lambda y: jnp.sqrt(jnp.sum((x - y) ** 2)))(X2)
 94            )(X1)
 95
 96        self._pairwise_sq_dists_jax = pairwise_sq_dists_jax
 97        self._cdist_jax = cdist_jax

106    def _fit_residual_sampler(self, **kwargs):
107        if self.residuals_ is None or len(self.residuals_) == 0:
108            raise ValueError("No residuals available for fitting sampler")
109
110        if self.residual_sampling == "kde":
111            kernel_bandwidths = {"bandwidth": np.logspace(-6, 6, 150)}
112            grid = GridSearchCV(
113                KernelDensity(kernel=self.kde_kernel, **kwargs),
114                param_grid=kernel_bandwidths,
115            )
116            grid.fit(self.residuals_)
117            self.kde_model_ = grid.best_estimator_
118            self.kde_model_.fit(self.residuals_)

120        elif self.residual_sampling == "gmm":
121            self.gmm_model_ = GaussianMixture(
122                n_components=min(self.gmm_components, len(self.residuals_)),
123                random_state=self.random_state,
124                covariance_type="full",
125            )
126            self.gmm_model_.fit(self.residuals_)

128    def _sample_residuals(self, num_samples, random_state=123):
129        if self.residuals_ is None:
130            raise ValueError("No residuals available for sampling")
131
132        n = len(self.residuals_)
133
134        if self.residual_sampling == "bootstrap":
135            np.random.seed(random_state)
136            if num_samples <= n:
137                idx = np.random.choice(n, num_samples, replace=True)
138                return self.residuals_[idx]
139            else:
140                n_repeats = (num_samples // n) + 1
141                tiled = np.tile(self.residuals_, (n_repeats, 1))
142                idx = np.random.choice(len(tiled), num_samples, replace=False)
143                return tiled[idx]
144
145        elif self.residual_sampling == "kde":
146            if self.kde_model_ is None:
147                raise ValueError(
148                    "KDE model not fitted. Call _fit_residual_sampler first."
149                )
150
151            samples = self.kde_model_.sample(
152                num_samples, random_state=random_state
153            )
154
155            if samples.ndim == 1:
156                samples = samples.reshape(-1, 1)
157            return samples
158
159        elif self.residual_sampling == "gmm":
160            if self.gmm_model_ is None:
161                raise ValueError(
162                    "GMM model not fitted. Call _fit_residual_sampler first."
163                )
164
165            # Set random state before sampling
166            np.random.seed(random_state)
167            samples = self.gmm_model_.sample(num_samples)[0]
168
169            if samples.ndim == 1:
170                samples = samples.reshape(-1, 1)
171            return samples
172
173        elif self.residual_sampling == "me-bootstrap":
174            meb = MaximumEntropyBootstrap(random_state=random_state)
175            residuals = self.residuals_.flatten()
176            if residuals.shape[0] < num_samples:
177                repeats = int(np.ceil(num_samples / residuals.shape[0]))
178                residuals = np.tile(residuals, repeats)[:num_samples]
179            else:
180                residuals = residuals[:num_samples]
181            meb.fit(residuals)
182            samples = meb.sample(1)[:, 0].reshape(-1, 1)
183            # Ensure we have exactly num_samples
184            if len(samples) < num_samples:
185                n_repeats = (num_samples // len(samples)) + 1
186                samples = np.tile(samples, (n_repeats, 1))[:num_samples]
187            return samples
188
189        elif self.residual_sampling == "block-bootstrap":
190            samples = bootstrap(
191                self.residuals_,
192                num_samples,
193                block_size=self.block_size,
194                seed=random_state,
195            )
196            # Ensure correct shape
197            if samples.ndim == 1:
198                samples = samples.reshape(-1, 1)
199            return samples
200
201        else:
202            raise ValueError(
203                f"Unknown sampling method: {self.residual_sampling}"
204            )

206    def _pairwise_sq_dists(self, X1, X2):
207        if self.backend in ["gpu", "tpu"] and JAX_AVAILABLE:
208            X1_jax = jnp.array(X1)
209            X2_jax = jnp.array(X2)
210            result = self._pairwise_sq_dists_jax(X1_jax, X2_jax)
211            return np.array(result)
212        else:
213            X1 = np.atleast_2d(X1)
214            X2 = np.atleast_2d(X2)
215            return (
216                np.sum(X1**2, axis=1)[:, np.newaxis]
217                + np.sum(X2**2, axis=1)[np.newaxis, :]
218                - 2 * X1 @ X2.T
219            )

221    def _mmd(self, u, v, kernel_sigma=1):
222        if u.ndim == 1:
223            u = u.reshape(-1, 1)
224        if v.ndim == 1:
225            v = v.reshape(-1, 1)
226
227        def kmat(A, B):
228            return np.exp(
229                -self._pairwise_sq_dists(A, B) / (2 * kernel_sigma**2)
230            )
231
232        return (
233            np.mean(kmat(u, u)) + np.mean(kmat(v, v)) - 2 * np.mean(kmat(u, v))
234        )

236    def _custom_energy_distance(self, u, v):
237        if u.ndim == 1:
238            u = u.reshape(-1, 1)
239        if v.ndim == 1:
240            v = v.reshape(-1, 1)
241
242        n, d = u.shape
243        m = v.shape[0]
244
245        if self.backend in ["gpu", "tpu"] and JAX_AVAILABLE:
246            u_jax = jnp.array(u)
247            v_jax = jnp.array(v)
248            dist_xx = self._cdist_jax(u_jax, u_jax)
249            dist_yy = self._cdist_jax(v_jax, v_jax)
250            dist_xy = self._cdist_jax(u_jax, v_jax)
251            term1 = 2 * jnp.sum(dist_xy) / (n * m)
252            term2 = jnp.sum(dist_xx) / (n * n)
253            term3 = jnp.sum(dist_yy) / (m * m)
254            return float(term1 - term2 - term3)
255        else:
256            dist_xx = cdist(u, u, metric="euclidean")
257            dist_yy = cdist(v, v, metric="euclidean")
258            dist_xy = cdist(u, v, metric="euclidean")
259            term1 = 2 * np.sum(dist_xy) / (n * m)
260            term2 = np.sum(dist_xx) / (n * n)
261            term3 = np.sum(dist_yy) / (m * m)
262            return term1 - term2 - term3

264    def _generate_pseudo_single(self, random_state=123):
265        """
266        Generate a single synthetic realization.
267
268        Returns original data (structure) + resampled residuals (noise)
269        Each call produces a different realization due to different residual samples.
270        """
271        if not self.is_fitted:
272            raise ValueError("Model not fitted. Call fit() first.")
273
274        n_rows = self.n_samples_
275
276        # Base: original time series structure
277        base = self.Y_.copy()
278        if base.ndim == 1:
279            base = base.reshape(-1, 1)
280
281        # Noise: sample new residuals from learned distribution
282        residuals = self._sample_residuals(n_rows, random_state)
283
284        # Ensure residuals match the size of base
285        # This is needed because model residuals may be shorter due to lags
286        if residuals.shape[0] < n_rows:
287            n_repeats = (n_rows // residuals.shape[0]) + 1
288            residuals = np.tile(residuals, (n_repeats, 1))[:n_rows]
289        elif residuals.shape[0] > n_rows:
290            residuals = residuals[:n_rows]
291
292        # Return: structure + new noise realization
293        return base + residuals

295    def fit(self, Y, metric="energy", n_trials=50, **kwargs):
296        if Y.ndim == 1:
297            Y = Y.reshape(-1, 1)
298
299        n, d = Y.shape
300        self.n_features_ = d
301        self.n_samples_ = n
302        self.Y_ = Y  # Store once before optimization
303
304        self.X_dist = np.random.normal(0, 1, (n, d))
305
306        def objective(trial):
307            sigma = trial.suggest_float("sigma", 0.01, 10, log=True)
308            lambd = trial.suggest_float("lambd", 1e-5, 1, log=True)
309            lags = trial.suggest_int("lags", 1, 50)
310            n_hidden_features = trial.suggest_int("n_hidden_features", 1, 20)
311            gamma = 1 / (2 * sigma**2)
312
313            model = self._create_model(gamma, lambd, lags, n_hidden_features)
314            model.fit(Y)
315
316            # Generate synthetic sample using this model's residuals
317            Y_sim = self._generate_pseudo_with_model(
318                model, model.residuals_, n, random_state=trial.number
319            )
320
321            if metric == "energy":
322                dist_val = self._custom_energy_distance(Y, Y_sim)
323            elif metric == "mmd":
324                dist_val = self._mmd(Y, Y_sim)
325            elif metric == "wasserstein" and d == 1:
326                dist_val = stats.wasserstein_distance(
327                    Y.flatten(), Y_sim.flatten()
328                )
329            else:
330                raise ValueError("Invalid metric for dimension")
331
332            return dist_val
333
334        study = optuna.create_study(direction="minimize")
335        study.optimize(objective, n_trials=n_trials, **kwargs)
336
337        self.best_params_ = study.best_params
338        self.best_score_ = study.best_value
339        sigma = self.best_params_["sigma"]
340        lambd = self.best_params_["lambd"]
341        lags = self.best_params_["lags"]
342        n_hidden_features = self.best_params_["n_hidden_features"]
343        gamma = 1 / (2 * sigma**2)
344
345        self.model = self._create_model(gamma, lambd, lags, n_hidden_features)
346        self.model.fit(Y)
347
348        self.residuals_ = self.model.residuals_
349
350        self._fit_residual_sampler()
351        self.is_fitted = True
352
353        print(f"  Best energy distance: {self.best_score_:.6f}")
354        print(f"  Best lags: {lags}, n_hidden_features: {n_hidden_features}")
355
356        return self

358    def _generate_pseudo_with_model(
359        self, model, residuals, num_samples, random_state=None
360    ):
361        """Helper function for optimization - temporarily uses different residuals"""
362        # Temporarily store and swap residual models
363        original_residuals = self.residuals_
364        original_kde = self.kde_model_
365        original_gmm = self.gmm_model_
366
367        self.residuals_ = residuals
368
369        # Only fit if using kde or gmm
370        if self.residual_sampling in ["kde", "gmm"]:
371            self._fit_residual_sampler()
372
373        # Length of actual residuals from the model
374        residual_len = len(residuals)
375
376        # Use provided random state or generate one
377        if random_state is None:
378            random_state = np.random.randint(0, 10000)
379
380        # Sample residuals matching the residual length
381        sampled_residuals = self._sample_residuals(
382            residual_len, random_state=random_state
383        )
384
385        # Restore original state
386        self.residuals_ = original_residuals
387        self.kde_model_ = original_kde
388        self.gmm_model_ = original_gmm
389
390        # The model with lags produces residuals shorter than original data
391        # We need to align: use only the portion of Y that corresponds to residuals
392        # Typically, if lags=L, residuals start from index L
393        y_slice = self.Y_[-residual_len:]  # Take the last residual_len points
394
395        # Return: aligned data + resampled residuals
396        return y_slice + sampled_residuals

398    def sample(self, n_samples=1):
399        """
400        Generate synthetic samples via distribution matching.
401
402        Each sample is: original structure + resampled residuals
403
404        Parameters:
405        -----------
406        n_samples : int, default=1
407            Number of synthetic realizations to generate
408
409        Returns:
410        --------
411        samples : ndarray
412            - If Y was univariate (n_rows, 1): returns shape (n_rows, n_samples)
413            - If Y was multivariate (n_rows, n_features): returns shape (n_features, n_rows, n_samples)
414        """
415        if not self.is_fitted:
416            raise ValueError("Model not fitted. Call fit() first.")
417
418        # Generate n_samples realizations, each with shape (n_rows, n_features)
419        samples_list = []
420        for i, _ in enumerate(range(n_samples)):
421            sample = self._generate_pseudo_single(
422                random_state=1000 + i
423            )  # Shape: (n_rows, n_features)
424            samples_list.append(sample)
425
426        # Stack to get shape (n_samples, n_rows, n_features)
427        stacked = np.stack(samples_list, axis=0)
428
429        # If univariate (n_features == 1), return (n_rows, n_samples)
430        if self.n_features_ == 1:
431            result = stacked.squeeze(
432                axis=2
433            ).T  # (n_samples, n_rows) -> (n_rows, n_samples)
434        else:
435            # If multivariate, return (n_features, n_rows, n_samples)
436            result = stacked.transpose(
437                2, 1, 0
438            )  # (n_samples, n_rows, n_features) -> (n_features, n_rows, n_samples)
439
440        return result

 81    def _setup_jax_backend(self):
 82        if not JAX_AVAILABLE:
 83            raise ImportError("JAX is required for GPU/TPU backend")
 84
 85        @jit
 86        def pairwise_sq_dists_jax(X1, X2):
 87            X1_sq = jnp.sum(X1**2, axis=1)[:, jnp.newaxis]
 88            X2_sq = jnp.sum(X2**2, axis=1)[jnp.newaxis, :]
 89            return X1_sq + X2_sq - 2 * X1 @ X2.T
 90
 91        @jit
 92        def cdist_jax(X1, X2):
 93            return vmap(
 94                lambda x: vmap(lambda y: jnp.sqrt(jnp.sum((x - y) ** 2)))(X2)
 95            )(X1)
 96
 97        self._pairwise_sq_dists_jax = pairwise_sq_dists_jax
 98        self._cdist_jax = cdist_jax

100    def _create_model(self, gamma, alpha, lags=20, n_hidden_features=5):
101        return ns.MTS(
102            obj=KernelRidge(kernel=self.kernel, gamma=gamma, alpha=alpha),
103            lags=lags,
104            n_hidden_features=n_hidden_features,
105        )

107    def _fit_residual_sampler(self, **kwargs):
108        if self.residuals_ is None or len(self.residuals_) == 0:
109            raise ValueError("No residuals available for fitting sampler")
110
111        if self.residual_sampling == "kde":
112            kernel_bandwidths = {"bandwidth": np.logspace(-6, 6, 150)}
113            grid = GridSearchCV(
114                KernelDensity(kernel=self.kde_kernel, **kwargs),
115                param_grid=kernel_bandwidths,
116            )
117            grid.fit(self.residuals_)
118            self.kde_model_ = grid.best_estimator_
119            self.kde_model_.fit(self.residuals_)
120
121        elif self.residual_sampling == "gmm":
122            self.gmm_model_ = GaussianMixture(
123                n_components=min(self.gmm_components, len(self.residuals_)),
124                random_state=self.random_state,
125                covariance_type="full",
126            )
127            self.gmm_model_.fit(self.residuals_)

129    def _sample_residuals(self, num_samples, random_state=123):
130        if self.residuals_ is None:
131            raise ValueError("No residuals available for sampling")
132
133        n = len(self.residuals_)
134
135        if self.residual_sampling == "bootstrap":
136            np.random.seed(random_state)
137            if num_samples <= n:
138                idx = np.random.choice(n, num_samples, replace=True)
139                return self.residuals_[idx]
140            else:
141                n_repeats = (num_samples // n) + 1
142                tiled = np.tile(self.residuals_, (n_repeats, 1))
143                idx = np.random.choice(len(tiled), num_samples, replace=False)
144                return tiled[idx]
145
146        elif self.residual_sampling == "kde":
147            if self.kde_model_ is None:
148                raise ValueError(
149                    "KDE model not fitted. Call _fit_residual_sampler first."
150                )
151
152            samples = self.kde_model_.sample(
153                num_samples, random_state=random_state
154            )
155
156            if samples.ndim == 1:
157                samples = samples.reshape(-1, 1)
158            return samples
159
160        elif self.residual_sampling == "gmm":
161            if self.gmm_model_ is None:
162                raise ValueError(
163                    "GMM model not fitted. Call _fit_residual_sampler first."
164                )
165
166            # Set random state before sampling
167            np.random.seed(random_state)
168            samples = self.gmm_model_.sample(num_samples)[0]
169
170            if samples.ndim == 1:
171                samples = samples.reshape(-1, 1)
172            return samples
173
174        elif self.residual_sampling == "me-bootstrap":
175            meb = MaximumEntropyBootstrap(random_state=random_state)
176            residuals = self.residuals_.flatten()
177            if residuals.shape[0] < num_samples:
178                repeats = int(np.ceil(num_samples / residuals.shape[0]))
179                residuals = np.tile(residuals, repeats)[:num_samples]
180            else:
181                residuals = residuals[:num_samples]
182            meb.fit(residuals)
183            samples = meb.sample(1)[:, 0].reshape(-1, 1)
184            # Ensure we have exactly num_samples
185            if len(samples) < num_samples:
186                n_repeats = (num_samples // len(samples)) + 1
187                samples = np.tile(samples, (n_repeats, 1))[:num_samples]
188            return samples
189
190        elif self.residual_sampling == "block-bootstrap":
191            samples = bootstrap(
192                self.residuals_,
193                num_samples,
194                block_size=self.block_size,
195                seed=random_state,
196            )
197            # Ensure correct shape
198            if samples.ndim == 1:
199                samples = samples.reshape(-1, 1)
200            return samples
201
202        else:
203            raise ValueError(
204                f"Unknown sampling method: {self.residual_sampling}"
205            )

207    def _pairwise_sq_dists(self, X1, X2):
208        if self.backend in ["gpu", "tpu"] and JAX_AVAILABLE:
209            X1_jax = jnp.array(X1)
210            X2_jax = jnp.array(X2)
211            result = self._pairwise_sq_dists_jax(X1_jax, X2_jax)
212            return np.array(result)
213        else:
214            X1 = np.atleast_2d(X1)
215            X2 = np.atleast_2d(X2)
216            return (
217                np.sum(X1**2, axis=1)[:, np.newaxis]
218                + np.sum(X2**2, axis=1)[np.newaxis, :]
219                - 2 * X1 @ X2.T
220            )
221
222    def _mmd(self, u, v, kernel_sigma=1):
223        if u.ndim == 1:
224            u = u.reshape(-1, 1)
225        if v.ndim == 1:
226            v = v.reshape(-1, 1)
227
228        def kmat(A, B):
229            return np.exp(
230                -self._pairwise_sq_dists(A, B) / (2 * kernel_sigma**2)
231            )
232
233        return (
234            np.mean(kmat(u, u)) + np.mean(kmat(v, v)) - 2 * np.mean(kmat(u, v))
235        )
236
237    def _custom_energy_distance(self, u, v):
238        if u.ndim == 1:
239            u = u.reshape(-1, 1)
240        if v.ndim == 1:
241            v = v.reshape(-1, 1)
242
243        n, d = u.shape
244        m = v.shape[0]
245
246        if self.backend in ["gpu", "tpu"] and JAX_AVAILABLE:
247            u_jax = jnp.array(u)
248            v_jax = jnp.array(v)
249            dist_xx = self._cdist_jax(u_jax, u_jax)
250            dist_yy = self._cdist_jax(v_jax, v_jax)
251            dist_xy = self._cdist_jax(u_jax, v_jax)
252            term1 = 2 * jnp.sum(dist_xy) / (n * m)
253            term2 = jnp.sum(dist_xx) / (n * n)
254            term3 = jnp.sum(dist_yy) / (m * m)
255            return float(term1 - term2 - term3)
256        else:
257            dist_xx = cdist(u, u, metric="euclidean")
258            dist_yy = cdist(v, v, metric="euclidean")
259            dist_xy = cdist(u, v, metric="euclidean")
260            term1 = 2 * np.sum(dist_xy) / (n * m)
261            term2 = np.sum(dist_xx) / (n * n)
262            term3 = np.sum(dist_yy) / (m * m)
263            return term1 - term2 - term3

265    def _generate_pseudo_single(self, random_state=123):
266        """
267        Generate a single synthetic realization.
268
269        Returns original data (structure) + resampled residuals (noise)
270        Each call produces a different realization due to different residual samples.
271        """
272        if not self.is_fitted:
273            raise ValueError("Model not fitted. Call fit() first.")
274
275        n_rows = self.n_samples_
276
277        # Base: original time series structure
278        base = self.Y_.copy()
279        if base.ndim == 1:
280            base = base.reshape(-1, 1)
281
282        # Noise: sample new residuals from learned distribution
283        residuals = self._sample_residuals(n_rows, random_state)
284
285        # Ensure residuals match the size of base
286        # This is needed because model residuals may be shorter due to lags
287        if residuals.shape[0] < n_rows:
288            n_repeats = (n_rows // residuals.shape[0]) + 1
289            residuals = np.tile(residuals, (n_repeats, 1))[:n_rows]
290        elif residuals.shape[0] > n_rows:
291            residuals = residuals[:n_rows]
292
293        # Return: structure + new noise realization
294        return base + residuals

296    def fit(self, Y, metric="energy", n_trials=50, **kwargs):
297        if Y.ndim == 1:
298            Y = Y.reshape(-1, 1)
299
300        n, d = Y.shape
301        self.n_features_ = d
302        self.n_samples_ = n
303        self.Y_ = Y  # Store once before optimization
304
305        self.X_dist = np.random.normal(0, 1, (n, d))
306
307        def objective(trial):
308            sigma = trial.suggest_float("sigma", 0.01, 10, log=True)
309            lambd = trial.suggest_float("lambd", 1e-5, 1, log=True)
310            lags = trial.suggest_int("lags", 1, 50)
311            n_hidden_features = trial.suggest_int("n_hidden_features", 1, 20)
312            gamma = 1 / (2 * sigma**2)
313
314            model = self._create_model(gamma, lambd, lags, n_hidden_features)
315            model.fit(Y)
316
317            # Generate synthetic sample using this model's residuals
318            Y_sim = self._generate_pseudo_with_model(
319                model, model.residuals_, n, random_state=trial.number
320            )
321
322            if metric == "energy":
323                dist_val = self._custom_energy_distance(Y, Y_sim)
324            elif metric == "mmd":
325                dist_val = self._mmd(Y, Y_sim)
326            elif metric == "wasserstein" and d == 1:
327                dist_val = stats.wasserstein_distance(
328                    Y.flatten(), Y_sim.flatten()
329                )
330            else:
331                raise ValueError("Invalid metric for dimension")
332
333            return dist_val
334
335        study = optuna.create_study(direction="minimize")

399    def sample(self, n_samples=1):
400        """
401        Generate synthetic samples via distribution matching.
402
403        Each sample is: original structure + resampled residuals
404
405        Parameters:
406        -----------
407        n_samples : int, default=1
408            Number of synthetic realizations to generate
409
410        Returns:
411        --------
412        samples : ndarray
413            - If Y was univariate (n_rows, 1): returns shape (n_rows, n_samples)
414            - If Y was multivariate (n_rows, n_features): returns shape (n_features, n_rows, n_samples)
415        """
416        if not self.is_fitted:
417            raise ValueError("Model not fitted. Call fit() first.")
418
419        # Generate n_samples realizations, each with shape (n_rows, n_features)
420        samples_list = []
421        for i, _ in enumerate(range(n_samples)):
422            sample = self._generate_pseudo_single(
423                random_state=1000 + i
424            )  # Shape: (n_rows, n_features)
425            samples_list.append(sample)
426
427        # Stack to get shape (n_samples, n_rows, n_features)
428        stacked = np.stack(samples_list, axis=0)
429
430        # If univariate (n_features == 1), return (n_rows, n_samples)
431        if self.n_features_ == 1:
432            result = stacked.squeeze(
433                axis=2
434            ).T  # (n_samples, n_rows) -> (n_rows, n_samples)
435        else:
436            # If multivariate, return (n_features, n_rows, n_samples)
437            result = stacked.transpose(
438                2, 1, 0
439            )  # (n_samples, n_rows, n_features) -> (n_features, n_rows, n_samples)
440
441        return result
