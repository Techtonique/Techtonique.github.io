 16class Progbar(object):
 17    """Displays a progress bar.
 18
 19    # Arguments
 20        target: Total number of steps expected, None if unknown.
 21        width: Progress bar width on screen.
 22        verbose: Verbosity mode, 0 (silent), 1 (verbose), 2 (semi-verbose)
 23        stateful_metrics: Iterable of string names of metrics that
 24            should *not* be averaged over time. Metrics in this list
 25            will be displayed as-is. All others will be averaged
 26            by the progbar before display.
 27        interval: Minimum visual progress update interval (in seconds).
 28    """
 29
 30    def __init__(
 31        self,
 32        target,
 33        width=30,
 34        verbose=1,
 35        interval=0.05,
 36        stateful_metrics=None,
 37    ):
 38        self.target = target
 39        self.width = width
 40        self.verbose = verbose
 41        self.interval = interval
 42        if stateful_metrics:
 43            self.stateful_metrics = set(stateful_metrics)
 44        else:
 45            self.stateful_metrics = set()
 46
 47        self._dynamic_display = (
 48            hasattr(sys.stdout, "isatty") and sys.stdout.isatty()
 49        ) or "ipykernel" in sys.modules
 50        self._total_width = 0
 51        self._seen_so_far = 0
 52        self._values = collections.OrderedDict()
 53        self._start = time.time()
 54        self._last_update = 0
 55
 56    def update(self, current, values=None):
 57        """Updates the progress bar.
 58
 59        # Arguments
 60            current: Index of current step.
 61            values: List of tuples:
 62                `(name, value_for_last_step)`.
 63                If `name` is in `stateful_metrics`,
 64                `value_for_last_step` will be displayed as-is.
 65                Else, an average of the metric over time will be displayed.
 66        """
 67        values = values or []
 68        for k, v in values:
 69            if k not in self.stateful_metrics:
 70                if k not in self._values:
 71                    self._values[k] = [
 72                        v * (current - self._seen_so_far),
 73                        current - self._seen_so_far,
 74                    ]
 75                else:
 76                    self._values[k][0] += v * (current - self._seen_so_far)
 77                    self._values[k][1] += current - self._seen_so_far
 78            else:
 79                # Stateful metrics output a numeric value.  This representation
 80                # means "take an average from a single value" but keeps the
 81                # numeric formatting.
 82                self._values[k] = [v, 1]
 83        self._seen_so_far = current
 84
 85        now = time.time()
 86        info = " - %.0fs" % (now - self._start)
 87        if self.verbose == 1:
 88            if (
 89                now - self._last_update < self.interval
 90                and self.target is not None
 91                and current < self.target
 92            ):
 93                return
 94
 95            prev_total_width = self._total_width
 96            if self._dynamic_display:
 97                sys.stdout.write("\b" * prev_total_width)
 98                sys.stdout.write("\r")
 99            else:
100                sys.stdout.write("\n")
101
102            if self.target is not None:
103                numdigits = int(np.floor(np.log10(self.target))) + 1
104                barstr = "%%%dd/%d [" % (
105                    numdigits,
106                    self.target,
107                )
108                bar = barstr % current
109                prog = float(current) / self.target
110                prog_width = int(self.width * prog)
111                if prog_width > 0:
112                    # bar += ('=' * (prog_width - 1))
113                    bar += "" * (prog_width - 1)
114                    if current < self.target:
115                        # bar += '>'
116                        bar += ""
117                    else:
118                        # bar += '='
119                        bar += ""
120                bar += "." * (self.width - prog_width)
121                bar += "]"
122            else:
123                bar = "%7d/Unknown" % current
124
125            self._total_width = len(bar)
126            sys.stdout.write(bar)
127
128            if current:
129                time_per_unit = (now - self._start) / current
130            else:
131                time_per_unit = 0
132            if self.target is not None and current < self.target:
133                eta = time_per_unit * (self.target - current)
134                if eta > 3600:
135                    eta_format = "%d:%02d:%02d" % (
136                        eta // 3600,
137                        (eta % 3600) // 60,
138                        eta % 60,
139                    )
140                elif eta > 60:
141                    eta_format = "%d:%02d" % (
142                        eta // 60,
143                        eta % 60,
144                    )
145                else:
146                    eta_format = "%ds" % eta
147
148                info = " - ETA: %s" % eta_format
149            else:
150                if time_per_unit >= 1:
151                    info += " %.0fs/step" % time_per_unit
152                elif time_per_unit >= 1e-3:
153                    info += " %.0fms/step" % (time_per_unit * 1e3)
154                else:
155                    info += " %.0fus/step" % (time_per_unit * 1e6)
156
157            for k in self._values:
158                info += " - %s:" % k
159                if isinstance(self._values[k], list):
160                    avg = np.mean(
161                        self._values[k][0] / max(1, self._values[k][1])
162                    )
163                    if abs(avg) > 1e-3:
164                        info += " %.4f" % avg
165                    else:
166                        info += " %.4e" % avg
167                else:
168                    info += " %s" % self._values[k]
169
170            self._total_width += len(info)
171            if prev_total_width > self._total_width:
172                info += " " * (prev_total_width - self._total_width)
173
174            if self.target is not None and current >= self.target:
175                info += "\n"
176
177            sys.stdout.write(info)
178            sys.stdout.flush()
179
180        elif self.verbose == 2:
181            if self.target is None or current >= self.target:
182                for k in self._values:
183                    info += " - %s:" % k
184                    avg = np.mean(
185                        self._values[k][0] / max(1, self._values[k][1])
186                    )
187                    if avg > 1e-3:
188                        info += " %.4f" % avg
189                    else:
190                        info += " %.4e" % avg
191                info += "\n"
192
193                sys.stdout.write(info)
194                sys.stdout.flush()
195
196        self._last_update = now

198    def add(self, n, values=None):
199        self.update(self._seen_so_far + n, values)
