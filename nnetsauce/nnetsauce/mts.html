288    def fit(self, X, xreg=None, **kwargs):
289        """Fit MTS model to training data X, with optional regressors xreg
290
291        Parameters:
292
293        X: {array-like}, shape = [n_samples, n_features]
294            Training time series, where n_samples is the number
295            of samples and n_features is the number of features;
296            X must be in increasing order (most recent observations last)
297
298        xreg: {array-like}, shape = [n_samples, n_features_xreg]
299            Additional (external) regressors to be passed to self.obj
300            xreg must be in 'increasing' order (most recent observations last)
301
302        **kwargs: for now, additional parameters to be passed to for kernel density estimation, when needed (see sklearn.neighbors.KernelDensity)
303
304        Returns:
305
306        self: object
307        """
308
309        if (
310            isinstance(X, pd.DataFrame) is False
311        ):  # input data set is a numpy array
312
313            if xreg is None:
314                X = pd.DataFrame(X)
315                self.series_names = [
316                    "series" + str(i) for i in range(X.shape[1])
317                ]
318            else:  # xreg is not None
319                X = mo.cbind(X, xreg)
320                self.xreg_ = xreg
321
322        else:  # input data set is a DataFrame with column names
323
324            X_index = None
325            if X.index is not None:
326                X_index = X.index
327            if xreg is None:
328                X = copy.deepcopy(mo.convert_df_to_numeric(X))
329            else:
330                X = copy.deepcopy(mo.cbind(mo.convert_df_to_numeric(X), xreg))
331                self.xreg_ = xreg
332            if X_index is not None:
333                X.index = X_index
334            self.series_names = X.columns.tolist()
335
336        if isinstance(X, pd.DataFrame):
337            self.df_ = X
338            X = X.values
339            self.df_.columns = self.series_names
340            self.input_dates = ts.compute_input_dates(self.df_)
341        else:
342            self.df_ = pd.DataFrame(X, columns=self.series_names)
343            self.input_dates = ts.compute_input_dates(self.df_)
344
345        try:
346            # multivariate time series
347            n, p = X.shape
348        except:
349            # univariate time series
350            n = X.shape[0]
351            p = 1
352
353        self.n_obs_ = n
354
355        rep_1_n = np.repeat(1, n)
356
357        self.y_ = None
358        self.X_ = None
359        self.n_series = p
360        self.fit_objs_.clear()
361        self.y_means_.clear()
362        residuals_ = []
363        self.residuals_ = None
364        self.residuals_sims_ = None
365        self.kde_ = None
366        self.sims_ = None
367        self.scaled_Z_ = None
368        self.centered_y_is_ = []
369
370        if p > 1:
371            # multivariate time series
372            mts_input = ts.create_train_inputs(X[::-1], self.lags)
373        else:
374            # univariate time series
375            mts_input = ts.create_train_inputs(
376                X.reshape(-1, 1)[::-1], self.lags
377            )
378
379        self.y_ = mts_input[0]
380
381        self.X_ = mts_input[1]
382
383        dummy_y, scaled_Z = self.cook_training_set(y=rep_1_n, X=self.X_)
384
385        self.scaled_Z_ = scaled_Z
386
387        # loop on all the time series and adjust self.obj.fit
388        if self.verbose > 0:
389            print(
390                f"\n Adjusting {type(self.obj).__name__} to multivariate time series... \n "
391            )
392
393        if self.show_progress is True:
394            iterator = tqdm(range(p))
395        else:
396            iterator = range(p)
397
398        if self.type_pi in ("kde", "bootstrap", "block-bootstrap"):
399            for i in iterator:
400                y_mean = np.mean(self.y_[:, i])
401                self.y_means_[i] = y_mean
402                centered_y_i = self.y_[:, i] - y_mean
403                self.centered_y_is_.append(centered_y_i)
404                self.obj.fit(X=scaled_Z, y=centered_y_i)
405                self.fit_objs_[i] = deepcopy(self.obj)
406                residuals_.append(
407                    (
408                        centered_y_i - self.fit_objs_[i].predict(scaled_Z)
409                    ).tolist()
410                )
411
412        if self.type_pi in (
413            "scp-kde",
414            "scp-bootstrap",
415            "scp-block-bootstrap",
416            "scp2-kde",
417            "scp2-bootstrap",
418            "scp2-block-bootstrap",
419        ):
420            # split conformal prediction
421            for i in iterator:
422                n_y = self.y_.shape[0]
423                n_y_half = n_y // 2
424                first_half_idx = range(0, n_y_half)
425                second_half_idx = range(n_y_half, n_y)
426                y_mean_temp = np.mean(self.y_[first_half_idx, i])
427                centered_y_i_temp = self.y_[first_half_idx, i] - y_mean_temp
428                self.obj.fit(X=scaled_Z[first_half_idx, :], y=centered_y_i_temp)
429                # calibrated residuals actually
430                residuals_.append(
431                    (
432                        self.y_[second_half_idx, i]
433                        - (
434                            y_mean_temp
435                            + self.obj.predict(scaled_Z[second_half_idx, :])
436                        )
437                    ).tolist()
438                )
439                # fit on the second half
440                y_mean = np.mean(self.y_[second_half_idx, i])
441                self.y_means_[i] = y_mean
442                centered_y_i = self.y_[second_half_idx, i] - y_mean
443                self.obj.fit(X=scaled_Z[second_half_idx, :], y=centered_y_i)
444                self.fit_objs_[i] = deepcopy(self.obj)
445
446        self.residuals_ = np.asarray(residuals_).T
447        if self.type_pi in (
448            "scp2-kde",
449            "scp2-bootstrap",
450            "scp2-block-bootstrap",
451        ):
452            # Calculate mean and standard deviation for each column
453            data_mean = np.mean(self.residuals_, axis=0)
454            self.residuals_std_dev_ = np.std(self.residuals_, axis=0)
455            # Center and scale the array using broadcasting
456            self.residuals_ = (
457                self.residuals_ - data_mean[np.newaxis, :]
458            ) / self.residuals_std_dev_[np.newaxis, :]
459
460        if self.replications != None and self.type_pi in (
461            "kde",
462            "scp-kde",
463            "scp2-kde",
464        ):
465            if self.verbose > 0:
466                print(f"\n Simulate residuals using {self.kernel} kernel... \n")
467            assert self.kernel in (
468                "gaussian",
469                "tophat",
470            ), "currently, 'kernel' must be either 'gaussian' or 'tophat'"
471            kernel_bandwidths = {"bandwidth": np.logspace(-6, 6, 150)}
472            grid = GridSearchCV(
473                KernelDensity(kernel=self.kernel, **kwargs),
474                param_grid=kernel_bandwidths,
475            )
476            grid.fit(self.residuals_)
477
478            if self.verbose > 0:
479                print(
480                    f"\n Best parameters for {self.kernel} kernel: {grid.best_params_} \n"
481                )
482
483            self.kde_ = grid.best_estimator_
484
485        return self

487    def predict(self, h=5, level=95, **kwargs):
488        """Forecast all the time series, h steps ahead
489
490        Parameters:
491
492        h: {integer}
493            Forecasting horizon
494
495        level: {integer}
496            Level of confidence (if obj has option 'return_std' and the
497            posterior is gaussian)
498
499        new_xreg: {array-like}, shape = [n_samples = h, n_new_xreg]
500            New values of additional (deterministic) regressors on horizon = h
501            new_xreg must be in increasing order (most recent observations last)
502
503        **kwargs: additional parameters to be passed to
504                self.cook_test_set
505
506        Returns:
507
508        model predictions for horizon = h: {array-like}, data frame or tuple.
509        Standard deviation and prediction intervals are returned when
510        `obj.predict` can return standard deviation
511
512        """
513
514        self.output_dates_, frequency = ts.compute_output_dates(self.df_, h)
515
516        self.level_ = level
517
518        self.return_std_ = False  # do not remove (/!\)
519
520        self.mean_ = None  # do not remove (/!\)
521
522        self.mean_ = deepcopy(self.y_)  # do not remove (/!\)
523
524        self.lower_ = None  # do not remove (/!\)
525
526        self.upper_ = None  # do not remove (/!\)
527
528        self.sims_ = None  # do not remove (/!\)
529
530        y_means_ = np.asarray([self.y_means_[i] for i in range(self.n_series)])
531
532        n_features = self.n_series * self.lags
533
534        self.alpha_ = 100 - level
535
536        pi_multiplier = norm.ppf(1 - self.alpha_ / 200)
537
538        if "return_std" in kwargs:  # bayesian forecasting
539            self.return_std_ = True
540            self.preds_std_ = []
541
542            DescribeResult = namedtuple(
543                "DescribeResult", ("mean", "lower", "upper")
544            )  # to be updated
545
546        if self.kde_ != None and self.type_pi in (
547            "kde",
548            "scp-kde",
549            "scp2-kde",
550        ):  # kde
551            self.residuals_sims_ = tuple(
552                self.kde_.sample(n_samples=h, random_state=self.seed + 100 * i)
553                for i in tqdm(range(self.replications))
554            )
555
556        if self.type_pi in ("bootstrap", "scp-bootstrap", "scp2-bootstrap"):
557            assert self.replications is not None and isinstance(
558                self.replications, int
559            ), "'replications' must be provided and be an integer"
560            self.residuals_sims_ = tuple(
561                ts.bootstrap(
562                    self.residuals_,
563                    h=h,
564                    block_size=None,
565                    seed=self.seed + 100 * i,
566                )
567                for i in tqdm(range(self.replications))
568            )
569
570        if self.type_pi in (
571            "block-bootstrap",
572            "scp-block-bootstrap",
573            "scp2-block-bootstrap",
574        ):
575            assert self.replications is not None and isinstance(
576                self.replications, int
577            ), "'replications' must be provided and be an integer"
578            self.residuals_sims_ = tuple(
579                ts.bootstrap(
580                    self.residuals_, h=h, block_size=h, seed=self.seed + 100 * i
581                )
582                for i in tqdm(range(self.replications))
583            )
584
585        for _ in range(h):
586
587            new_obs = ts.reformat_response(self.mean_, self.lags)
588
589            new_X = new_obs.reshape(1, n_features)
590
591            cooked_new_X = self.cook_test_set(new_X, **kwargs)
592
593            if "return_std" in kwargs:
594                self.preds_std_.append(
595                    [
596                        np.asarray(
597                            self.fit_objs_[i].predict(
598                                cooked_new_X, return_std=True
599                            )[1]
600                        ).item()
601                        for i in range(self.n_series)
602                    ]
603                )
604
605            predicted_cooked_new_X = np.asarray(
606                [
607                    np.asarray(self.fit_objs_[i].predict(cooked_new_X)).item()
608                    for i in range(self.n_series)
609                ]
610            )
611
612            preds = np.asarray(y_means_ + predicted_cooked_new_X)
613
614            self.mean_ = mo.rbind(preds, self.mean_)  # preallocate?
615
616        # function's return ----------------------------------------------------------------------
617        self.mean_ = pd.DataFrame(
618            self.mean_[0:h, :][::-1],
619            columns=self.df_.columns,
620            index=self.output_dates_,
621        )
622        if "return_std" not in kwargs:
623
624            if self.replications is None:
625                return self.mean_
626
627            # if "return_std" not in kwargs and self.replications is not None
628            meanf = []
629            lower = []
630            upper = []
631
632            if self.type_pi in (
633                "scp2-kde",
634                "scp2-bootstrap",
635                "scp2-block-bootstrap",
636            ):
637                self.sims_ = tuple(
638                    (
639                        self.mean_
640                        + self.residuals_sims_[i]
641                        * self.residuals_std_dev_[np.newaxis, :]
642                        for i in tqdm(range(self.replications))
643                    )
644                )
645            else:
646                self.sims_ = tuple(
647                    (
648                        self.mean_ + self.residuals_sims_[i]
649                        for i in tqdm(range(self.replications))
650                    )
651                )
652
653            DescribeResult = namedtuple(
654                "DescribeResult", ("mean", "sims", "lower", "upper")
655            )
656            for ix in range(self.n_series):
657                sims_ix = getsims(self.sims_, ix)
658                if self.agg == "mean":
659                    meanf.append(np.mean(sims_ix, axis=1))
660                else:
661                    meanf.append(np.median(sims_ix, axis=1))
662                lower.append(np.quantile(sims_ix, q=self.alpha_ / 200, axis=1))
663                upper.append(
664                    np.quantile(sims_ix, q=1 - self.alpha_ / 200, axis=1)
665                )
666
667            self.mean_ = pd.DataFrame(
668                np.asarray(meanf).T,
669                columns=self.series_names,  # self.df_.columns,
670                index=self.output_dates_,
671            )
672
673            self.lower_ = pd.DataFrame(
674                np.asarray(lower).T,
675                columns=self.series_names,  # self.df_.columns,
676                index=self.output_dates_,
677            )
678
679            self.upper_ = pd.DataFrame(
680                np.asarray(upper).T,
681                columns=self.series_names,  # self.df_.columns,
682                index=self.output_dates_,
683            )
684
685            res = DescribeResult(
686                self.mean_, self.sims_, self.lower_, self.upper_
687            )
688
689            if self.xreg_ is not None:
690
691                if len(self.xreg_.shape) > 1:
692
693                    res2 = mx.tuple_map(
694                        res,
695                        lambda x: mo.delete_last_columns(
696                            x, num_columns=self.xreg_.shape[1]
697                        ),
698                    )
699
700                else:
701
702                    res2 = mx.tuple_map(
703                        res, lambda x: mo.delete_last_columns(x, num_columns=1)
704                    )
705
706                return res2
707
708            else:
709
710                return res
711
712        if "return_std" in kwargs:
713            DescribeResult = namedtuple(
714                "DescribeResult", ("mean", "lower", "upper")
715            )
716
717            self.mean_ = pd.DataFrame(
718                np.asarray(self.mean_),
719                columns=self.series_names,  # self.df_.columns,
720                index=self.output_dates_,
721            )
722
723            self.preds_std_ = np.asarray(self.preds_std_)
724
725            self.lower_ = pd.DataFrame(

Univariate and multivariate time series (MTS) forecasting with Quasi-Randomized networks

Parameters:

obj: object.
    any object containing a method fit (obj.fit()) and a method predict
    (obj.predict()).

n_hidden_features: int.
    number of nodes in the hidden layer.

activation_name: str.
    activation function: 'relu', 'tanh', 'sigmoid', 'prelu' or 'elu'.

a: float.
    hyperparameter for 'prelu' or 'elu' activation function.

nodes_sim: str.
    type of simulation for the nodes: 'sobol', 'hammersley', 'halton',
    'uniform'.

bias: boolean.
    indicates if the hidden layer contains a bias term (True) or not
    (False).

dropout: float.
    regularization parameter; (random) percentage of nodes dropped out
    of the training.

direct_link: boolean.
    indicates if the original predictors are included (True) in model's fitting or not (False).

n_clusters: int.
    number of clusters for 'kmeans' or 'gmm' clustering (could be 0: no clustering).

cluster_encode: bool.
    defines how the variable containing clusters is treated (default is one-hot)
    if `False`, then labels are used, without one-hot encoding.

type_clust: str.
    type of clustering method: currently k-means ('kmeans') or Gaussian
    Mixture Model ('gmm').

type_scaling: a tuple of 3 strings.
    scaling methods for inputs, hidden layer, and clustering respectively
    (and when relevant).
    Currently available: standardization ('std') or MinMax scaling ('minmax').

lags: int.
    number of lags used for each time series.

    type of prediction interval; currently:

    - "bootstrap": based on independent bootstrap of in-sample residuals
    - "block-bootstrap": based on basic block bootstrap of in-sample residuals
    - "scp-kde": Split conformal prediction with Kernel Density Estimation of calibrated residuals
    - "scp-bootstrap": Split conformal prediction with independent bootstrap of calibrated residuals
    - "scp-block-bootstrap": Split conformal prediction with basic block bootstrap of calibrated residuals
    - "scp2-kde": Split conformal prediction with Kernel Density Estimation of standardized calibrated residuals
    - "scp2-bootstrap": Split conformal prediction with independent bootstrap of standardized calibrated residuals
    - "scp2-block-bootstrap": Split conformal prediction with basic block bootstrap of standardized calibrated residuals

    number of replications (if needed, for predictive simulation). Default is 'None'.

    the kernel to use for residuals density estimation (used for predictive simulation). Currently, either 'gaussian' or 'tophat'.

agg: str.
    either "mean" or "median" for simulation of bootstrap aggregating

    reproducibility seed for nodes_sim=='uniform' or predictive simulation.

backend: str.
    "cpu" or "gpu" or "tpu".

verbose: int.
    0: not printing; 1: printing

show_progress: bool.
    True: progress bar when fitting each series; False: no progress bar when fitting each series
