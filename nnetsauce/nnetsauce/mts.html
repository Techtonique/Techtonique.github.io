 989        """Train on training_index, score on testing_index."""
 990
 991        assert (
 992            bool(set(training_index).intersection(set(testing_index))) == False

 996        try:

1008        else:

1010            X_test = X[testing_index]

1015            len(training_index) + h

1039            "explained_variance": skm2.explained_variance_score,
