255    def fit(self, X, xreg=None, **kwargs):
256        """Fit MTS model to training data X, with optional regressors xreg
257
258        Parameters:
259
260            X: {array-like}, shape = [n_samples, n_features]
261                Training time series, where n_samples is the number
262                of samples and n_features is the number of features;
263                X must be in increasing order (most recent observations last)
264
265            xreg: {array-like}, shape = [n_samples, n_features_xreg]
266                Additional (external) regressors to be passed to self.obj
267                xreg must be in 'increasing' order (most recent observations last)
268
269            **kwargs: for now, additional parameters to be passed to for kernel density estimation, when needed (see sklearn.neighbors.KernelDensity)
270
271        Returns:
272
273            self: object
274        """
275
276        if (
277            isinstance(X, pd.DataFrame) is False
278        ):  # input data set is a numpy array
279
280            if xreg is None:
281                X = pd.DataFrame(X)
282                self.series_names = [
283                    "series" + str(i) for i in range(X.shape[1])
284                ]
285            else:  # xreg is not None
286                X = mo.cbind(X, xreg)
287                self.xreg_ = xreg
288
289        else:  # input data set is a DataFrame with column names
290
291            X_index = None
292            if X.index is not None:
293                X_index = X.index
294            if xreg is None:
295                X = copy.deepcopy(mo.convert_df_to_numeric(X))
296            else:
297                X = copy.deepcopy(mo.cbind(mo.convert_df_to_numeric(X), xreg))
298                self.xreg_ = xreg
299            if X_index is not None:
300                X.index = X_index
301            self.series_names = X.columns.tolist()
302
303        if isinstance(X, pd.DataFrame):
304            self.df_ = X
305            X = X.values
306            self.df_.columns = self.series_names
307            self.input_dates = ts.compute_input_dates(self.df_)
308        else:
309            self.df_ = pd.DataFrame(X, columns=self.series_names)
310            self.input_dates = ts.compute_input_dates(self.df_)
311
312        try:
313            # multivariate time series
314            n, p = X.shape
315        except:
316            # univariate time series
317            n = X.shape[0]
318            p = 1
319
320        rep_1_n = np.repeat(1, n)
321
322        self.y_ = None
323        self.X_ = None
324        self.n_series = p
325        self.fit_objs_.clear()
326        self.y_means_.clear()
327        residuals_ = []
328        self.residuals_ = None
329        self.residuals_sims_ = None
330        self.kde_ = None
331        self.sims_ = None
332        self.scaled_Z_ = None
333        self.centered_y_is_ = []
334
335        if p > 1:
336            # multivariate time series
337            mts_input = ts.create_train_inputs(X[::-1], self.lags)
338        else:
339            # univariate time series
340            mts_input = ts.create_train_inputs(
341                X.reshape(-1, 1)[::-1], self.lags
342            )
343
344        self.y_ = mts_input[0]
345
346        self.X_ = mts_input[1]
347
348        dummy_y, scaled_Z = self.cook_training_set(y=rep_1_n, X=self.X_)
349
350        self.scaled_Z_ = scaled_Z
351
352        # loop on all the time series and adjust self.obj.fit
353        if self.verbose > 0:
354            print(
355                f"\n Adjusting {type(self.obj).__name__} to multivariate time series... \n "
356            )
357
358        if self.show_progress is True:
359            iterator = tqdm(range(p))
360        else:
361            iterator = range(p)
362
363        for i in iterator:
364            y_mean = np.mean(self.y_[:, i])
365            self.y_means_[i] = y_mean
366            centered_y_i = self.y_[:, i] - y_mean
367            self.centered_y_is_.append(centered_y_i)
368            self.obj.fit(X=scaled_Z, y=centered_y_i)
369            self.fit_objs_[i] = deepcopy(self.obj)
370            residuals_.append(
371                (centered_y_i - self.fit_objs_[i].predict(scaled_Z)).tolist()
372            )
373
374        self.residuals_ = np.asarray(residuals_).T
375
376        if self.replications != None and self.type_pi == "kde":
377            if self.verbose > 0:
378                print(f"\n Simulate residuals using {self.kernel} kernel... \n")
379            assert self.kernel in (
380                "gaussian",
381                "tophat",
382            ), "currently, 'kernel' must be either 'gaussian' or 'tophat'"
383            kernel_bandwidths = {"bandwidth": np.logspace(-6, 6, 150)}
384            grid = GridSearchCV(

387            )
388            grid.fit(self.residuals_)
389
390            if self.verbose > 0:

392                    f"\n Best parameters for {self.kernel} kernel: {grid.best_params_} \n"
393                )
394
395            self.kde_ = grid.best_estimator_
396
397        return self

399    def predict(self, h=5, level=95, **kwargs):
400        """Forecast all the time series, h steps ahead
401
402        Parameters:
403
404            h: {integer}
405                Forecasting horizon
406
407            level: {integer}
408                Level of confidence (if obj has option 'return_std' and the
409                posterior is gaussian)
410
411            new_xreg: {array-like}, shape = [n_samples = h, n_new_xreg]
412                New values of additional (deterministic) regressors on horizon = h
413                new_xreg must be in increasing order (most recent observations last)
414
415            **kwargs: additional parameters to be passed to
416                    self.cook_test_set
417
418        Returns:
419
420            model predictions for horizon = h: {array-like}, data frame or tuple.
421            Standard deviation and prediction intervals are returned when
422            `obj.predict` can return standard deviation
423        """
424
425        self.output_dates_, frequency = ts.compute_output_dates(self.df_, h)
426
427        self.return_std_ = False  # do not remove (/!\)
428
429        self.mean_ = None  # do not remove (/!\)
430
431        self.mean_ = deepcopy(self.y_)  # do not remove (/!\)
432
433        self.lower_ = None  # do not remove (/!\)
434
435        self.upper_ = None  # do not remove (/!\)
436
437        self.sims_ = None  # do not remove (/!\)
438
439        y_means_ = np.asarray([self.y_means_[i] for i in range(self.n_series)])
440
441        n_features = self.n_series * self.lags
442
443        self.alpha_ = 100 - level
444
445        if "return_std" in kwargs:
446            self.return_std_ = True
447            self.preds_std_ = []
448            pi_multiplier = norm.ppf(1 - self.alpha_ / 200)
449            DescribeResult = namedtuple(
450                "DescribeResult", ("mean", "lower", "upper")
451            )  # to be updated
452
453        if self.kde_ != None and self.type_pi == "kde":
454            pi_multiplier = norm.ppf(1 - self.alpha_ / 200)
455            self.residuals_sims_ = tuple(
456                self.kde_.sample(n_samples=h, random_state=self.seed + 100 * i)
457                for i in tqdm(range(self.replications))
458            )
459
460        for _ in range(h):
461
462            new_obs = ts.reformat_response(self.mean_, self.lags)
463
464            new_X = new_obs.reshape(1, n_features)
465
466            cooked_new_X = self.cook_test_set(new_X, **kwargs)
467
468            if "return_std" in kwargs:
469                self.preds_std_.append(
470                    [
471                        np.asarray(
472                            self.fit_objs_[i].predict(
473                                cooked_new_X, return_std=True
474                            )[1]
475                        ).item()
476                        for i in range(self.n_series)
477                    ]
478                )
479
480            predicted_cooked_new_X = np.asarray(
481                [
482                    np.asarray(self.fit_objs_[i].predict(cooked_new_X)).item()
483                    for i in range(self.n_series)
484                ]
485            )
486
487            preds = np.asarray(y_means_ + predicted_cooked_new_X)
488
489            self.mean_ = mo.rbind(preds, self.mean_)  # preallocate?
490
491        # function's return ----------------------------------------------------------------------
492        self.mean_ = pd.DataFrame(
493            self.mean_[0:h, :][::-1],
494            columns=self.df_.columns,
495            index=self.output_dates_,
496        )
497        if "return_std" not in kwargs:
498
499            if self.kde_ is None:
500                return self.mean_
501
502            # if "return_std" not in kwargs and self.kde_ is not None
503            meanf = []
504            lower = []
505            upper = []
506
507            self.sims_ = tuple(
508                (
509                    self.mean_ + self.residuals_sims_[i]
510                    for i in tqdm(range(self.replications))
511                )
512            )
513
514            DescribeResult = namedtuple(
515                "DescribeResult", ("mean", "sims", "lower", "upper")
516            )
517            for ix in range(self.n_series):
518                sims_ix = getsims(self.sims_, ix)
519                if self.agg == "mean":
520                    meanf.append(np.mean(sims_ix, axis=1))
521                else:
522                    meanf.append(np.median(sims_ix, axis=1))
523                lower.append(np.quantile(sims_ix, q=self.alpha_ / 200, axis=1))
524                upper.append(
525                    np.quantile(sims_ix, q=1 - self.alpha_ / 200, axis=1)
526                )
527
528            self.mean_ = pd.DataFrame(
529                np.asarray(meanf).T,
530                columns=self.series_names,  # self.df_.columns,
531                index=self.output_dates_,
532            )
533
534            self.lower_ = pd.DataFrame(
535                np.asarray(lower).T,
536                columns=self.series_names,  # self.df_.columns,
537                index=self.output_dates_,
538            )
539
540            self.upper_ = pd.DataFrame(
541                np.asarray(upper).T,
542                columns=self.series_names,  # self.df_.columns,
543                index=self.output_dates_,
544            )
545
546            res = DescribeResult(
547                self.mean_, self.sims_, self.lower_, self.upper_
548            )
549
550            if self.xreg_ is not None:
551
552                if len(self.xreg_.shape) > 1:
553
554                    res2 = mx.tuple_map(
555                        res,
556                        lambda x: mo.delete_last_columns(
557                            x, num_columns=self.xreg_.shape[1]
558                        ),
559                    )
560
561                else:
562
563                    res2 = mx.tuple_map(
564                        res, lambda x: mo.delete_last_columns(x, num_columns=1)
565                    )
566
567                return res2
568
569            else:
570
571                return res
572
573        if "return_std" in kwargs:
574            DescribeResult = namedtuple(
575                "DescribeResult", ("mean", "lower", "upper")
576            )
577
578            self.mean_ = pd.DataFrame(
579                np.asarray(self.mean_),
580                columns=self.series_names,  # self.df_.columns,
581                index=self.output_dates_,
582            )
583
584            self.preds_std_ = np.asarray(self.preds_std_)
585
586            self.lower_ = pd.DataFrame(
587                self.mean_.values - pi_multiplier * self.preds_std_,
588                columns=self.series_names,  # self.df_.columns,
589                index=self.output_dates_,
590            )
591
592            self.upper_ = pd.DataFrame(
593                self.mean_.values + pi_multiplier * self.preds_std_,
594                columns=self.series_names,  # self.df_.columns,
595                index=self.output_dates_,
596            )
597            res = DescribeResult(self.mean_, self.lower_, self.upper_)
598
599            if self.xreg_ is not None:
600                if len(self.xreg_.shape) > 1:
601                    res2 = mx.tuple_map(
602                        res,
603                        lambda x: mo.delete_last_columns(
604                            x, num_columns=self.xreg_.shape[1]
605                        ),
606                    )
607                else:
608                    res2 = mx.tuple_map(
609                        res, lambda x: mo.delete_last_columns(x, num_columns=1)
610                    )
611                return DescribeResult(res2[0], res2[1], res2[2])
612
613            return res

615    def score(self, X, training_index, testing_index, scoring=None, **kwargs):
616        """Train on training_index, score on testing_index."""
617
618        assert (
619            bool(set(training_index).intersection(set(testing_index))) == False
620        ), "Non-overlapping 'training_index' and 'testing_index' required"
621
622        # Dimensions
623        try:
624            # multivariate time series
625            n, p = X.shape
626        except:
627            # univariate time series
628            n = X.shape[0]
629            p = 1
630
631        # Training and testing sets
632        if p > 1:
633            X_train = X[training_index, :]
634            X_test = X[testing_index, :]
635        else:
636            X_train = X[training_index]
637            X_test = X[testing_index]
638
639        # Horizon
640        h = len(testing_index)
641        assert (
642            len(training_index) + h
643        ) <= n, "Please check lengths of training and testing windows"
644
645        # Fit and predict
646        self.fit(X_train, **kwargs)
647        preds = self.predict(h=h, **kwargs)
648
649        if scoring is None:
650            scoring = "neg_root_mean_squared_error"
651
652        # check inputs
653        assert scoring in (
654            "explained_variance",
655            "neg_mean_absolute_error",
656            "neg_mean_squared_error",
657            "neg_root_mean_squared_error",
658            "neg_mean_squared_log_error",
659            "neg_median_absolute_error",
660            "r2",
661        ), "'scoring' should be in ('explained_variance', 'neg_mean_absolute_error', \
662                               'neg_mean_squared_error', 'neg_root_mean_squared_error', 'neg_mean_squared_log_error', \
663                               'neg_median_absolute_error', 'r2')"
664
665        scoring_options = {
666            "explained_variance": skm2.explained_variance_score,
667            "neg_mean_absolute_error": skm2.mean_absolute_error,
668            "neg_mean_squared_error": skm2.mean_squared_error,
669            "neg_root_mean_squared_error": lambda x, y: np.sqrt(
670                skm2.mean_squared_error(x, y)
671            ),
672            "neg_mean_squared_log_error": skm2.mean_squared_log_error,
673            "neg_median_absolute_error": skm2.median_absolute_error,
674            "r2": skm2.r2_score,
675        }
676
677        # if p > 1:
678        #     return tuple(
679        #         [
680        #             scoring_options[scoring](
681        #                 X_test[:, i], preds[:, i]#, **kwargs
682        #             )
683        #             for i in range(p)
684        #         ]
685        #     )
686        # else:
687        return scoring_options[scoring](X_test, preds)
