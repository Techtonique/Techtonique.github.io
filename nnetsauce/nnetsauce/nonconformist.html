class MarginErrFunc(nnetsauce.nonconformist.nc.ClassificationErrFunc): 
116class MarginErrFunc(ClassificationErrFunc):
117    """
118    Calculates the margin error.
119    """
120
121    def __init__(self):
122        super(MarginErrFunc, self).__init__()
123
124    def apply(self, prediction, y):
125        prob = np.zeros(y.size, dtype=np.float32)
126        for i, y_ in enumerate(y):
127            if y_ >= prediction.shape[1]:
128                prob[i] = 0
129            else:
130                prob[i] = prediction[i, int(y_)]
131                prediction[i, int(y_)] = -np.inf
132        return 0.5 - ((prob - prediction.max(axis=1)) / 2)

class QuantileRegErrFunc(nnetsauce.nonconformist.nc.RegressionErrFunc): 
177class QuantileRegErrFunc(RegressionErrFunc):
178    """Calculates conformalized quantile regression error."""
179
180    def __init__(self):
181        super(QuantileRegErrFunc, self).__init__()
182
183    def apply(self, prediction, y):
184        y_lower = prediction[:, 0]
185        y_upper = prediction[:, -1]
186        error_low = y_lower - y
187        error_high = y - y_upper
188        err = np.maximum(error_high, error_low)
189        return err
190
191    def apply_inverse(self, nc, significance):
192        nc = np.sort(nc, 0)
193        index = int(np.ceil((1 - significance) * (nc.shape[0] + 1))) - 1
194        index = min(max(index, 0), nc.shape[0] - 1)
195        return np.vstack([nc[index], nc[index]])

Base class for all estimators in scikit-learn.

Inheriting from this class provides default implementations of:

  • setting and getting parameters used by GridSearchCV and friends;
  • textual and HTML representation displayed in terminals and IDEs;
  • estimator serialization;
  • parameters validation;
  • data validation;
  • feature names validation.

Read more in the :ref:User Guide <rolling_your_own_estimator>.

Notes

All estimators should specify all the parameters that can be set at the class level in their __init__ as explicit keyword arguments (no *args or **kwargs).

468class RegressorNc(BaseModelNc):
469    """Nonconformity scorer using an underlying regression model.
470
471    Parameters
472    ----------
473    model : RegressorAdapter
474        Underlying regression model used for calculating nonconformity scores.
475
476    err_func : RegressionErrFunc
477        Error function object.
478
479    normalizer : BaseScorer
480        Normalization model.
481
482    beta : float
483        Normalization smoothing parameter. As the beta-value increases,
484        the normalized nonconformity function approaches a non-normalized
485        equivalent.
486
487    Attributes
488    ----------
489    model : RegressorAdapter
490        Underlying model object.
491
492    err_func : RegressionErrFunc
493        Scorer function used to calculate nonconformity scores.
494
495    See also
496    --------
497    ProbEstClassifierNc, NormalizedRegressorNc
498    """
499
500    def __init__(
501        self, model, err_func=AbsErrorErrFunc(), normalizer=None, beta=1e-6
502    ):
503        super(RegressorNc, self).__init__(model, err_func, normalizer, beta)
504
505    def predict(self, x, nc, significance=None):
506        """Constructs prediction intervals for a set of test examples.
507
508        Predicts the output of each test pattern using the underlying model,
509        and applies the (partial) inverse nonconformity function to each
510        prediction, resulting in a prediction interval for each test pattern.
511
512        Parameters
513        ----------
514        x : numpy array of shape [n_samples, n_features]
515            Inputs of patters for which to predict output values.
516
517        significance : float
518            Significance level (maximum allowed error rate) of predictions.
519            Should be a float between 0 and 1. If ``None``, then intervals for
520            all significance levels (0.01, 0.02, ..., 0.99) are output in a
521            3d-matrix.
522
523        Returns
524        -------
525        p : numpy array of shape [n_samples, 2] or [n_samples, 2, 99]
526            If significance is ``None``, then p contains the interval (minimum
527            and maximum boundaries) for each test pattern, and each significance
528            level (0.01, 0.02, ..., 0.99). If significance is a float between
529            0 and 1, then p contains the prediction intervals (minimum and
530            maximum	boundaries) for the set of test patterns at the chosen
531            significance level.
532        """
533        n_test = x.shape[0]
534        prediction = self.model.predict(x)
535        if self.normalizer is not None:
536            norm = self.normalizer.score(x) + self.beta
537        else:
538            norm = np.ones(n_test)
539
540        if significance:
541            intervals = np.zeros((x.shape[0], 2))
542            err_dist = self.err_func.apply_inverse(nc, significance)
543            err_dist = np.hstack([err_dist] * n_test)
544            if prediction.ndim > 1:  # CQR
545                intervals[:, 0] = prediction[:, 0] - err_dist[0, :]
546                intervals[:, 1] = prediction[:, -1] + err_dist[1, :]
547            else:  # regular conformal prediction
548                err_dist *= norm
549                intervals[:, 0] = prediction - err_dist[0, :]
550                intervals[:, 1] = prediction + err_dist[1, :]
551
552            return intervals
553        else:  # Not tested for CQR
554            significance = np.arange(0.01, 1.0, 0.01)
555            intervals = np.zeros((x.shape[0], 2, significance.size))
556
557            for i, s in enumerate(significance):
558                err_dist = self.err_func.apply_inverse(nc, s)
559                err_dist = np.hstack([err_dist] * n_test)
560                err_dist *= norm
561
562                intervals[:, 0, i] = prediction - err_dist[0, :]
563                intervals[:, 1, i] = prediction + err_dist[0, :]
564
565            return intervals

386                level (0.01, 0.02, ..., 0.99). If significance is a float between

388                maximum	boundaries) for the set of test patterns at the chosen

396
397        if n_significance > 1:
398            prediction = np.zeros((x.shape[0], 2, n_significance))

414                else:
