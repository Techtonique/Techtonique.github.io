def nearestPD(A): 
15def nearestPD(A):
16    """Find the nearest positive-definite matrix to input
17
18    A Python/Numpy port of John D'Errico's `nearestSPD` MATLAB code [1], which
19    credits [2].
20
21    [1] https://www.mathworks.com/matlabcentral/fileexchange/42885-nearestspd
22
23    [2] N.J. Higham, "Computing a nearest symmetric positive semidefinite
24    matrix" (1988): https://doi.org/10.1016/0024-3795(88)90223-6
25    """
26
27    B = (A + A.T) / 2
28    _, s, V = la.svd(B)
29
30    H = np.dot(V.T, np.dot(np.diag(s), V))
31
32    A2 = (B + H) / 2
33
34    A3 = (A2 + A2.T) / 2
35
36    if isPD(A3):
37        return A3
38
39    spacing = np.spacing(la.norm(A))
40    # The above is different from [1]. It appears that MATLAB's `chol` Cholesky
41    # decomposition will accept matrixes with exactly 0-eigenvalue, whereas
42    # Numpy's will not. So where [1] uses `eps(mineig)` (where `eps` is Matlab
43    # for `np.spacing`), we use the above definition. CAVEAT: our `spacing`
44    # will be much larger than [1]'s `eps(mineig)`, since `mineig` is usually on
45    # the order of 1e-16, and `eps(1e-16)` is on the order of 1e-34, whereas
46    # `spacing` will, for Gaussian random matrixes of small dimension, be on
47    # othe order of 1e-16. In practice, both ways converge, as the unit test
48    # below suggests.
49    I = np.eye(A.shape[0])
50    k = 1
51    while not isPD(A3):
52        mineig = np.min(np.real(la.eigvals(A3)))
53        A3 += I * (-mineig * k**2 + spacing)
54        k += 1
55
56    return A3

 52        """Updates the progress bar.
